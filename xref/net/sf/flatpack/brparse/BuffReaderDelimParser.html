87      @Override
88      protected DataSet doParse() {
89          final DataSet ds = new BuffReaderDataSet(getPzMetaData(), this);
90          try {
91              // gather the conversion properties
92              ds.setPZConvertProps(ParserUtils.loadConvertProperties());
93  
94              br = new BufferedReader(getDataSourceReader());
95  
96              return ds;
97  
98          } catch (final IOException ex) {
99              LOGGER.error("error accessing/creating inputstream", ex);
100         }
101 
102         return null;
103     }

105     /**
106      * Reads in the next record on the file and return a row
107      *
108      * @param ds
109      * @return Row
110      */
111     @Override
112     public Row buildRow(final DefaultDataSet ds) {
113         /** loop through each line in the file */
114         while (true) {
115             String line;
116             try {
117                 line = fetchNextRecord(br, getQualifier(), getDelimiter());
118             } catch (final IOException e) {
119                 throw new RuntimeException("Error Fetching Record From File...", e);
120             }
121 
122             if (line == null) {
123                 return null;
124             }
125 
126             // check to see if the user has elected to skip the first record
127             if (!processedFirst && isIgnoreFirstRecord()) {
128                 processedFirst = true;
129                 continue;
130             } else if (!processedFirst && shouldCreateMDFromFile()) {
131                 processedFirst = true;
132                 setPzMetaData(ParserUtils.getPZMetaDataFromFile(line, getDelimiter(), getQualifier(), this, isAddSuffixToDuplicateColumnNames()));
133                 ds.setMetaData(getPzMetaData());
134                 continue;
135             }
136 
137             // TODO
138             // seems like we may want to try doing something like this. I have my reservations because
139             // it is possible that we don't get a "detail" id and this might generate NPE
140             // is it going to create too much overhead to do a null check here as well???
141             List<String> columns = ParserUtils.splitLine(line, getDelimiter(), getQualifier(), FPConstants.SPLITLINE_SIZE_INIT,
142                     isPreserveLeadingWhitespace(), isPreserveTrailingWhitespace());
143             final String mdkey = ParserUtils.getCMDKeyForDelimitedFile(getPzMetaData(), columns);
144             final List<ColumnMetaData> cmds = ParserUtils.getColumnMetaData(mdkey, getPzMetaData());
145             final int columnCount = cmds.size();
146             // DEBUG
147 
148             // Incorrect record length on line log the error. Line
149             // will not be included in the dataset
150             if (columns.size() > columnCount) {
151                 if (isIgnoreExtraColumns()) {
152                     // user has choosen to ignore the fact that we have too many columns in the data from
153                     // what the mapping has described. sublist the array to remove un-needed columns
154                     columns = columns.subList(0, columnCount);
155                     addError(ds, "TRUNCATED LINE TO CORRECT NUMBER OF COLUMNS", getLineCount(), 1);
156                 } else {
157                     // log the error
158                     addError(ds, "TOO MANY COLUMNS WANTED: " + columnCount + " GOT: " + columns.size(), getLineCount(), 2,
159                             isStoreRawDataToDataError() ? line : null);
160                     continue;
161                 }
162             } else if (columns.size() < columnCount) {
163                 if (isHandlingShortLines()) {
164                     // We can pad this line out
165                     while (columns.size() < columnCount) {
166                         columns.add("");
167                     }
168 
169                     // log a warning
170                     addError(ds, "PADDED LINE TO CORRECT NUMBER OF COLUMNS", getLineCount(), 1);
171 
172                 } else {
173                     addError(ds, "TOO FEW COLUMNS WANTED: " + columnCount + " GOT: " + columns.size(), getLineCount(), 2,
174                             isStoreRawDataToDataError() ? line : null);
175                     continue;
176                 }
177             }
178 
179             final Row row = new Row();
180             row.setMdkey(mdkey.equals(FPConstants.DETAIL_ID) ? null : mdkey); // try
181             // to limit the memory use
182             row.setCols(columns);
183             row.setRowNumber(getLineCount());
184 
185             if (isFlagEmptyRows()) {
186                 // user has elected to have the parser flag rows that are empty
187                 row.setEmpty(ParserUtils.isListElementsEmpty(columns));
188             }
189             if (isStoreRawDataToDataSet()) {
190                 // user told the parser to keep a copy of the raw data in the row
191                 // WARNING potential for high memory usage here
192                 row.setRawData(line);
193             }
194 
195             return row;
196         }
197     }
