43  /**
44   * Provides a PZParser which obtains records directly from
45   * a BufferedReader as an alternative to reading the
46   * entire file into memory.
47   *
48   * Database column mappings are not supported by this factory
49   * at the present time.  This class is meant to mimic the LargeDataSet
50   * class of pre 3.0 versions, which did not support database mappings
51   * either.  A RuntimeExcpetion will be thrown if trying to obtain a parser
52   * for a database map.
53   *
54   * @author Paul Zepernick
55   */
56  public class BuffReaderParseFactory implements ParserFactory {
57      private static final BuffReaderParseFactory INSTANCE = new BuffReaderParseFactory();
58  
59      public static ParserFactory getInstance() {
60          return INSTANCE;
61      }
62  
63      /**
64       * Not supported at this time.
65       */
66      @Override
67      public Parser newFixedLengthParser(final Connection con, final File dataSource, final String dataDefinition) {
68          throw new UnsupportedOperationException("Not supported...");
69      }
70  
71      /**
72       * Not supported at this time.
73       */
74      @Override
75      public Parser newFixedLengthParser(final Connection con, final InputStream dataSourceStream, final String dataDefinition) {
76          throw new UnsupportedOperationException("Not supported...");
77      }
78  
79      /*
80       * (non-Javadoc)
81       *
82       * @see net.sf.flatpack.PZParserFactory#newParser(java.io.File,
83       *      java.io.File)
84       */
85      @Override
86      public Parser newFixedLengthParser(final File pzmapXML, final File dataSource) {
87          return new BuffReaderFixedParser(pzmapXML, dataSource);
88      }
89  
90      @Override
91      public Parser newFixedLengthParser(final Connection con, final Reader dataSource, final String dataDefinition) {
92          return new DBBuffReaderFixedParser(con, dataSource, dataDefinition);
93      }
94  
95      @Override
96      public Parser newFixedLengthParser(final Reader pzmapXMLStream, final Reader dataSource) {
97          return new BuffReaderFixedParser(pzmapXMLStream, dataSource);
98      }
99

101      * (non-Javadoc)
102      *
103      * @see net.sf.flatpack.PZParserFactory#newParser(java.io.InputStream,
104      *      java.io.InputStream)

106     @Override
107     public Parser newFixedLengthParser(final InputStream pzmapXMLStream, final InputStream dataSourceStream) {
108         return new BuffReaderFixedParser(pzmapXMLStream, dataSourceStream);
109     }

111     @Override

113             final char qualifier, final boolean ignoreFirstRecord) {
