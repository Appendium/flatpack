46      public static void createFile(final int cols, final int rows, final String filename) {
47          FileWriter fw = null;
48          PrintWriter out = null;
49          try {
50  
51              fw = new FileWriter(filename);
52              out = new PrintWriter(fw);
53  
54              // write the column names across the top of the file
55              for (int i = 1; i <= cols; i++) {
56                  if (i > 1) {
57                      out.write(",");
58                  }
59                  out.write("\"column " + i + "\"");
60              }
61              out.write("\r\n");
62              out.flush();
63  
64              // write the rows
65              for (int i = 1; i <= rows; i++) {
66                  for (int j = 1; j <= cols; j++) {
67                      if (j > 1) {
68                          out.write(",");
69                      }
70                      out.write("\"data " + j + "\"");
71                  }
72  
73                  out.write("\r\n");
74                  out.flush();
75                  if (i % 100000 == 0) {
76                      System.out.print(".");
77                  }
78              }
79  
80          } catch (final Exception ex) {
81              LOG.error("Issue", ex);
82          } finally {
83              try {
84                  if (out != null) {
85                      out.close();
86                  }
87                  if (fw != null) {
88                      fw.close();
89                  }
90              } catch (final Exception ignore) {
91              }
92  
93          }
94      }
