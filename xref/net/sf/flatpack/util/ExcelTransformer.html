34  /**
35   * Converts a DataSet object into an excel spreadsheet.
36   *
37   * Only detail records will be contained in the export.
38   * Header / Trailer records defined by &lt;record&gt; tags in the
39   * pzmap will be ignored on the export.
40   *
41   * @author Paul Zepernick
42   */
43  public class ExcelTransformer {
44      private final DataSet ds;
45  
46      private final File xlsFile;
47  
48      private String[] exportOnlyColumns;
49  
50      private String[] excludeFromExportColumns;
51  
52      private String[] numericColumns;
53  
54      /**
55       * Constructs a new Excel transformer
56       *
57       * @param ds
58       *            DataSet to convert
59       * @param xlsFile
60       *            Excel file to be created
61       */
62      public ExcelTransformer(final DataSet ds, final File xlsFile) {
63          this.ds = ds;
64          this.xlsFile = xlsFile;
65      }
66  
67      /**
68       * Writes the Excel file to disk
69       *
70       * @throws IOException
71       * @throws WriteException
72       */
73      public void writeExcelFile() throws IOException, WriteException {
74          WritableWorkbook excelWrkBook = null;
75          int curDsPointer = 0;
76  
77          try {
78              final String[] columnNames = ds.getColumns();
79              final List<String> exportOnlyColumnsList = getExportOnlyColumns() != null ? Arrays.asList(exportOnlyColumns) : null;
80              final List<String> excludeFromExportColumnsList = getExcludeFromExportColumns() != null ? Arrays.asList(excludeFromExportColumns) : null;
81              final List<String> numericColumnList = getNumericColumns() != null ? Arrays.asList(getNumericColumns()) : new ArrayList<String>();
82              // get the current position of the DataSet. We have to go to the top
83              // to do this write,
84              // and we will put the pionter back where it was after we are done
85              curDsPointer = ds.getIndex();
86              ds.goTop();
87  
88              excelWrkBook = Workbook.createWorkbook(xlsFile);
89              final WritableSheet wrkSheet = excelWrkBook.createSheet("results", 0);
90  
91              final WritableFont times10ptBold = new WritableFont(WritableFont.TIMES, 10, WritableFont.BOLD);
92              final WritableFont times10pt = new WritableFont(WritableFont.TIMES, 10, WritableFont.NO_BOLD);
93              // write the column headings in the spreadsheet
94              WritableCellFormat cellFormat = new WritableCellFormat(times10ptBold);
95              int colOffset = 0;
96              for (int i = 0; i < columnNames.length; i++) {
97                  if (exportOnlyColumnsList != null && !exportOnlyColumnsList.contains(columnNames[i])
98                          || excludeFromExportColumnsList != null && excludeFromExportColumnsList.contains(columnNames[i])) {
99                      colOffset++;
100                     continue;
101                 }
102 
103                 final Label xlsTextLbl = new Label(i - colOffset, 0, columnNames[i], cellFormat);
104                 wrkSheet.addCell(xlsTextLbl);
105             }
106 
107             cellFormat = new WritableCellFormat(times10pt);
108             int row = 1;
109             while (ds.next()) {
110                 if (!ds.isRecordID(FPConstants.DETAIL_ID)) {
111                     continue;
112                 }
113 
114                 colOffset = 0;
115                 for (int i = 0; i < columnNames.length; i++) {
116                     if (exportOnlyColumnsList != null && !exportOnlyColumnsList.contains(columnNames[i])
117                             || excludeFromExportColumnsList != null && excludeFromExportColumnsList.contains(columnNames[i])) {
118                         colOffset++;
119                         continue;
120                     }
121 
122                     WritableCell wc = null;
123                     if (numericColumnList.contains(columnNames[i])) {
124                         wc = new Number(i - colOffset, row, ds.getDouble(columnNames[i]), cellFormat);
125                     } else {
126                         wc = new Label(i - colOffset, row, ds.getString(columnNames[i]), cellFormat);
127                     }
128 
129                     wrkSheet.addCell(wc);
130                 }
131 
132                 row++;
133             }
134 
135             excelWrkBook.write();
136 
137         } finally {
138             if (curDsPointer > -1) {
139                 ds.absolute(curDsPointer);
140             }
141             if (excelWrkBook != null) {
142                 excelWrkBook.close();
143             }
144         }
145 
146     }

148     /**
149      * The columns names contained in the array will be igored if
150      * setExportOnlyColumns() is called.
151      *
152      * Any columns names contained in this list will be excluded from
153      * the export in Excel.
154      *
155      * @param excludeFromExportColumns the excludeFromExportColumns to set
156      */
157     public void setExcludeFromExportColumns(final String[] excludeFromExportColumns) {
158         if (excludeFromExportColumns != null) {
159             this.excludeFromExportColumns = excludeFromExportColumns.clone();
160         } else {
161             this.excludeFromExportColumns = null;
162         }
163     }
164 
165     /**
166      * When set, only columns contained in the String[] will
167      * be exported out to Excel.
168      *
169      * @param exportOnlyColumns the exportOnlyColumns to set
170      */
171     public void setExportOnlyColumns(final String[] exportOnlyColumns) {
172         if (exportOnlyColumns != null) {
173             this.exportOnlyColumns = exportOnlyColumns.clone();
174         } else {
175             this.exportOnlyColumns = null;
176         }
177     }

182     public String[] getNumericColumns() {

186     /**

189      * @param numericColumns the numericColumns to set
190      */
191     public void setNumericColumns(final String[] numericColumns) {
192         this.numericColumns = numericColumns;
193     }

198     public String[] getExportOnlyColumns() {
