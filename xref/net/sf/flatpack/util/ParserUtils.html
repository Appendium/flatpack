77      /**
78       * Returns an ArrayList of items in a delimited string. If there is no
79       * qualifier around the text, the qualifier parameter can be left null, or
80       * empty. There should not be any line breaks in the string. Each line of
81       * the file should be passed in individually.
82       * Elements which are not qualified will have leading and trailing white
83       * space removed.  This includes unqualified elements, which may be
84       * contained in an unqualified parse: "data",  data  ,"data"
85       *
86       * Special thanks to Benoit for contributing this much improved speedy parser :0)
87       *
88       * @author Benoit Xhenseval
89       * @param line -
90       *            String of data to be parsed
91       * @param delimiter -
92       *            Delimiter separating each element
93       * @param qualifier -
94       *            qualifier which is surrounding the text
95       * @param initialSize -
96       *            initial capacity of the List size
97       * @param preserveLeadingWhitespace
98       *            Keep any leading spaces
99       * @param preserveTrailingWhitespace
100      *            Keep any trailing spaces
101      * @return List
102      */
103     public static List<String> splitLine(final String line, final char delimiter, final char qualifier, final int initialSize,
104             final boolean preserveLeadingWhitespace, final boolean preserveTrailingWhitespace) {
105         final List<String> list = new ArrayList<String>(initialSize);
106 
107         if (delimiter == 0) {
108             list.add(line);
109             return list;
110         } else if (line == null) {
111             return list;
112         }
113 
114         String trimmedLine;
115         if (delimiter == '\t' || delimiter == ' ') {
116             // skip the trim for these delimiters, doing the trim will mess up the parse
117             // on empty records which contain just the delimiter
118             trimmedLine = line;
119         } else {
120             trimmedLine = line;
121             if (!preserveLeadingWhitespace) {
122                 trimmedLine = ParserUtils.lTrim(line);
123             }
124             if (!preserveTrailingWhitespace) {
125                 trimmedLine = ParserUtils.rTrim(line);
126             }
127         }
128 
129         final int size = trimmedLine.length();
130 
131         if (size == 0) {
132             list.add("");
133             return list;
134         }
135 
136         boolean insideQualifier = false;
137         char previousChar = 0;
138         int startBlock = 0;
139         int endBlock = 0;
140         boolean blockWasInQualifier = false;
141 
142         final String doubleQualifier = "" + qualifier + qualifier;
143         for (int i = 0; i < size; i++) {
144 
145             final char currentChar = trimmedLine.charAt(i);
146             if (currentChar != delimiter && currentChar != qualifier) {
147                 previousChar = currentChar;
148                 endBlock = i + 1;
149                 continue;
150             }
151 
152             if (currentChar == delimiter) {
153                 // we've found the delimiter (eg ,)
154                 if (!insideQualifier) {
155                     String trimmed = trimmedLine.substring(startBlock, endBlock > startBlock ? endBlock : startBlock + 1);
156                     if (!blockWasInQualifier) {
157                         if (!preserveLeadingWhitespace) {
158                             trimmed = ParserUtils.lTrim(trimmed);
159                         }
160                         if (!preserveTrailingWhitespace) {
161                             trimmed = ParserUtils.rTrim(trimmed);
162                         }
163                     }
164 
165                     if (trimmed == null || trimmed.length() == 1 && (trimmed.charAt(0) == delimiter || trimmed.charAt(0) == qualifier)) {
166                         list.add("");
167                     } else {
168                         trimmed = trimmed.replace(doubleQualifier, String.valueOf(qualifier));
169                         list.add(trimmed);
170                     }
171                     blockWasInQualifier = false;
172                     startBlock = i + 1;
173                 }
174             } else if (currentChar == qualifier) {
175                 if (!insideQualifier && previousChar != qualifier) {
176                     if (previousChar == delimiter || previousChar == 0 || previousChar == ' ') {
177                         insideQualifier = true;
178                         startBlock = i + 1;
179                     } else {
180                         endBlock = i + 1;
181                     }
182                 } else {
183                     if (i + 1 < size && delimiter != ' ') {
184                         // this is used to allow unescaped qualifiers to be contained within the element
185                         // do not run this check is a space is being used as a delimiter
186                         // we don't want to trim the delimiter off
187                         // loop until we find a char that is not a space, or we reach the end of the line.
188                         int start = i + 1;
189                         char charToCheck = trimmedLine.charAt(start);
190                         while (charToCheck == ' ') {
191                             start++;
192                             if (start == size) {
193                                 break;
194                             }
195                             charToCheck = trimmedLine.charAt(start);
196                         }
197 
198                         if (charToCheck != delimiter) {
199                             previousChar = currentChar;
200                             endBlock = i + 1;
201                             continue;
202                         }
203 
204                     }
205                     insideQualifier = false;
206                     blockWasInQualifier = true;
207                     endBlock = i;
208                     // last column (e.g. finishes with ")
209                     if (i == size - 1) {
210                         String str = trimmedLine.substring(startBlock, size - 1);
211                         str = str.replace(doubleQualifier, String.valueOf(qualifier));
212                         list.add(str);
213                         startBlock = i + 1;
214                     }
215                 }
216             }
217             previousChar = currentChar;
218         }
219 
220         if (startBlock < size) {
221             String str = trimmedLine.substring(startBlock, size);
222             str = str.replace(doubleQualifier, String.valueOf(qualifier));
223             if (blockWasInQualifier) {
224                 if (str.charAt(str.length() - 1) == qualifier) {
225                     list.add(str.substring(0, str.length() - 1));
226                 } else {
227                     list.add(str);
228                 }
229             } else {
230                 String s = str;
231                 if (!preserveLeadingWhitespace) {
232                     s = ParserUtils.lTrim(s);
233                 }
234                 if (!preserveTrailingWhitespace) {
235                     s = ParserUtils.rTrim(s);
236                 }
237                 list.add(s);
238             }
239         } else if (trimmedLine.charAt(size - 1) == delimiter) {
240             list.add("");
241         }
242 
243         return list;
244     }

263     /**
264      * Removes empty space from the beginning of a string
265      *
266      * @param value -
267      *            to be trimmed
268      * @return String
269      */
270     public static String lTrim(final String value) {
271         if (value == null) {
272             return null;
273         }
274 
275         String trimmed = value;
276         int offset = 0;
277         final int maxLength = value.length();
278         while (offset < maxLength && (value.charAt(offset) == ' ' || value.charAt(offset) == '\t')) {
279             offset++;
280         }
281 
282         if (offset > 0) {
283             trimmed = value.substring(offset);
284         }
285 
286         return trimmed;
287     }

315     /**
316      * Removes empty space from the end of a string
317      *
318      * @param value -
319      *            to be trimmed
320      * @return String
321      */
322     public static String rTrim(final String value) {
323         if (value == null) {
324             return null;
325         }
326 
327         String trimmed = value;
328         int offset = value.length() - 1;
329         while (offset > -1 && (value.charAt(offset) == ' ' || value.charAt(offset) == '\t')) {
330             offset--;
331         }
332 
333         if (offset < value.length() - 1) {
334             trimmed = value.substring(0, offset + 1);
335         }
336 
337         return trimmed;
338     }

369         final StringBuilder s = new StringBuilder();
370         for (int i = 0; i < theString.length(); i++) {
371             final char currentChar = theString.charAt(i);
372             if (currentChar != theChar) {
373                 s.append(currentChar);
374             }
375         }
376 
377         return s.toString();

380     /**
381      * Returns a list of ColumnMetaData objects. This is for use with delimited
382      * files. The first line of the file which contains data will be used as the
383      * column names
384      *
385      * @param line
386      * @param delimiter
387      * @param qualifier
388      * @param p
389      *          PZParser used to specify additional option when working with the ColumnMetaData. Can be null
390      * @param addSuffixToDuplicateColumnNames 
391      * @return PZMetaData
392      */
393     public static MetaData getPZMetaDataFromFile(final String line, final char delimiter, final char qualifier, final Parser p,
394             boolean addSuffixToDuplicateColumnNames) {
395         final List<ColumnMetaData> results = new ArrayList<ColumnMetaData>();
396         final Set<String> dupCheck = new HashSet<String>();
397 
398         final List<String> lineData = splitLine(line, delimiter, qualifier, FPConstants.SPLITLINE_SIZE_INIT, false, false);
399         for (final String colName : lineData) {
400             final ColumnMetaData cmd = new ColumnMetaData();
401             String colNameToUse = colName;
402             if (dupCheck.contains(colNameToUse)) {
403                 if (!addSuffixToDuplicateColumnNames) {
404                     throw new FPException("Duplicate Column Name In File: " + colNameToUse);
405                 } else {
406                     int count = 2;
407                     while (dupCheck.contains(colNameToUse + count)) {
408                         count++;
409                     }
410                     colNameToUse = colName + count;
411                 }
412             }
413             cmd.setColName(colNameToUse);
414             results.add(cmd);
415             dupCheck.add(cmd.getColName());
416         }
417 
418         return new MetaData(results, buidColumnIndexMap(results, p));
419     }

421     /**
422      * Determines if the given line is the first part of a multiline record.  It does this by verifying that the
423      * qualifer on the last element is not closed
424      *
425      * @param chrArry -
426      *            char data of the line
427      * @param delimiter -
428      *            delimiter being used
429      * @param qualifier -
430      *            qualifier being used
431      * @return boolean
432      */
433     public static boolean isMultiLine(final char[] chrArry, final char delimiter, final char qualifier) {
434 
435         // check if the last char is the qualifier, if so then this a good
436         // chance it is not multiline
437         if (chrArry[chrArry.length - 1] != qualifier) {
438             // could be a potential line break
439             boolean qualiFound = false;
440             for (int i = chrArry.length - 1; i >= 0; i--) {
441                 if (chrArry[i] == ' ') {
442                     continue;
443                 }
444 
445                 // check to see if we can find a qualifier followed by a
446                 // delimiter
447                 // remember we are working are way backwards on the line
448                 if (qualiFound) {
449                     if (chrArry[i] == delimiter) {
450                         // before deciding if this is the begining of a qualified new line
451                         // I think we have to go back to the beginning of the line and see if we are inside a qualified
452                         // field or not?
453                         // return true;
454                         boolean qualifiedContent = chrArry[0] == qualifier;
455                         for (int index = 0; index < chrArry.length; index++) {
456                             char currentChar = chrArry[index];
457                             qualifiedContent = currentChar == qualifier;
458                             if (qualifiedContent) {
459                                 // go until first occurence of closing qualifierdelimiter combination
460                                 for (; index < chrArry.length; index++) {
461                                     if (chrArry[index] == delimiter && chrArry[++index] == qualifier) {
462                                         qualifiedContent = false;
463                                     }
464                                 }
465                             }
466                         }
467                         return qualifiedContent;
468                     }
469                     // guard against multiple qualifiers in the sequence [ ,""We ]
470                     qualiFound = chrArry[i] == qualifier;
471                 } else if (chrArry[i] == delimiter) {
472                     // if we have a delimiter followed by a qualifier, then we
473                     // have moved on to a new element and this could not be multiline.
474                     // start a new loop here in case there is
475                     // space between the delimiter and qualifier
476                     for (int j = i - 1; j >= 0; j--) {
477                         if (chrArry[j] == ' ') {
478                             continue;
479                         } else if (chrArry[j] == qualifier) {
480                             return false;
481                         }
482                         break;
483                     }
484 
485                 } else if (chrArry[i] == qualifier) {
486                     qualiFound = true;
487                 }
488             }
489         } else {
490             // we have determined that the last char on the line is a qualifier.
491             // This most likely means
492             // that this is not multiline, however we must account for the
493             // following scenario
494             // data,data,"
495             // data
496             // /data"
497             for (int i = chrArry.length - 1; i >= 0; i--) {
498                 if (i == chrArry.length - 1 || chrArry[i] == ' ') {
499                     // skip the first char, or any spaces we come across between
500                     // the delimiter and qualifier
501                     continue;
502                 }
503                 if (chrArry[i] == delimiter) {
504                     // before deciding if this is the begining of a qualified new line
505                     // I think we have to go back to the beginning of the line and see if we are inside a qualified
506                     // field or not?
507                     // return true;
508                     boolean qualifiedContent = chrArry[0] == qualifier;
509                     for (int index = 0; index < chrArry.length; index++) {
510                         char currentChar = chrArry[index];
511                         qualifiedContent = currentChar == qualifier;
512                         if (qualifiedContent) {
513                             // go until first occurence of closing qualifierdelimiter combination
514                             for (; index < chrArry.length; index++) {
515                                 if (chrArry[index] == delimiter && chrArry[++index] == qualifier) {
516                                     qualifiedContent = false;
517                                 }
518                             }
519                         }
520                     }
521                     return qualifiedContent;
522                 }
523                 break;
524             }
525         }
526 
527         return false;
528     }

530     public static Map<String, Integer> calculateRecordLengths(final MetaData columnMD) {
531         final Map<String, Integer> recordLengths = new HashMap<String, Integer>();
532 
533         // first the basic columns
534         int recordLength = 0;
535         for (final ColumnMetaData cmd : columnMD.getColumnsNames()) {
536             recordLength += cmd.getColLength();
537         }
538 
539         recordLengths.put(FPConstants.DETAIL_ID, recordLength);
540 
541         final Iterator<Entry<String, XMLRecordElement>> columnMDIt = columnMD.xmlRecordIterator();
542         while (columnMDIt.hasNext()) {
543             final Entry<String, XMLRecordElement> entry = columnMDIt.next();
544             final List<ColumnMetaData> cmds = entry.getValue().getColumns();
545 
546             recordLength = 0;
547             for (final ColumnMetaData cmd : cmds) {
548                 recordLength += cmd.getColLength();
549             }
550 
551             recordLengths.put(entry.getKey(), recordLength);
552         }
553 
554         return recordLengths;
555 
556     }

558     public static String getCMDKeyForDelimitedFile(final MetaData columnMD, final List<String> lineElements) {
559         if (!columnMD.isAnyRecordFormatSpecified()) {
560             // no <RECORD> elments were specifed for this parse, just return the
561             // detail id
562             return FPConstants.DETAIL_ID;
563         }
564         final Iterator<Entry<String, XMLRecordElement>> mapEntries = columnMD.xmlRecordIterator();
565         // loop through the XMLRecordElement objects and see if we need a
566         // different MD object
567         while (mapEntries.hasNext()) {
568             final Entry<String, XMLRecordElement> entry = mapEntries.next();
569             final XMLRecordElement recordXMLElement = entry.getValue();
570 
571             if (recordXMLElement.getElementCount() > 0 && recordXMLElement.getElementCount() == lineElements.size()) {
572                 // determing which <record> mapping to use by the number of elements
573                 // contained on the line
574                 return entry.getKey();
575             } else if (recordXMLElement.getElementNumber() > lineElements.size()) {
576                 // make sure the element referenced in the mapping exists
577                 continue;
578             }
579 
580             final String lineElement = lineElements.get(recordXMLElement.getElementNumber() - 1);
581             if (lineElement.equals(recordXMLElement.getIndicator())) {
582                 // we found the MD object we want to return
583                 return entry.getKey();
584             }
585 
586         }
587 
588         // must be a detail line
589         return FPConstants.DETAIL_ID;
590     }

600     public static int getColumnIndex(final String key, final MetaData columnMD, final String colName, final boolean columNameCaseSensitive) {
601         int idx = -1;
602         String column = colName;
603         if (!columNameCaseSensitive) {
604             column = colName.toLowerCase(Locale.getDefault());
605         }
606         idx = columnMD.getColumnIndex(key, column);
607 
608         if (idx < 0) {
609             throw new NoSuchElementException(
610                     "Column [" + column + "] does not exist, check case/spelling." + (key != null ? " key:[" + key + "]" : ""));
611         }
612         return idx;
613     }

662     /**
663      * <p>
664      * Returns padding using the specified delimiter repeated to a given length.
665      * </p>
666      *
667      * <pre>
668      *                PZStringUtils.padding(0, 'e')  = &quot;&quot;
669      *                PZStringUtils.padding(3, 'e')  = &quot;eee&quot;
670      *                PZStringUtils.padding(-2, 'e') = IndexOutOfBoundsException
671      * </pre>
672      *
673      * <p>
674      * Note: this method doesn't not support padding with <a
675      * href="http://www.unicode.org/glossary/#supplementary_character">Unicode
676      * Supplementary Characters</a> as they require a pair of <code>char</code>s
677      * to be represented.
678      * </p>
679      *
680      * @param repeat
681      *            number of times to repeat delim
682      * @param padChar
683      *            character to repeat
684      * @return String with repeated character
685      * @throws IndexOutOfBoundsException
686      */
687     public static String padding(final int repeat, final char padChar) {
688         if (repeat < 0) {
689             throw new IndexOutOfBoundsException("Cannot pad a negative amount: " + repeat);
690         }
691         final char[] buf = new char[repeat];
692         for (int i = 0; i < buf.length; i++) {
693             buf[i] = padChar;
694         }
695         return new String(buf);
696     }

698     /**
699      * Build a map of name/position based on a list of ColumnMetaData.
700      *
701      * @author Benoit Xhenseval
702      * @author Paul Zepernick
703      * @param columns
704      * @param p
705      *         Reference to Parser which can provide additional options on how the
706      *         map should be build.  This can be NULL.
707      * @return a new Map
708      */
709     public static Map<String, Integer> buidColumnIndexMap(final List<ColumnMetaData> columns, final Parser p) {
710         Map<String, Integer> map = null;
711         if (columns != null && !columns.isEmpty()) {
712             map = new HashMap<String, Integer>();
713             int idx = 0;
714             for (final ColumnMetaData meta : columns) {
715                 String colName = meta.getColName();
716                 if (p != null && !p.isColumnNamesCaseSensitive()) {
717                     // user has selected to make column names case sensitive
718                     // on lookups
719                     colName = colName.toLowerCase(Locale.getDefault());
720                 }
721                 map.put(colName, idx++);
722             }
723         }
724         return map;

730      *
731      *      PZStringUtils.stripNonLongChars("1000.25") = "1000"
732      *
733      * Method will truncate everything to the right of the decimal
734      * place when encountered.
735      *
736      * @param value
737      * @return String
738      */
739     public static String stripNonLongChars(final String value) {
740         final StringBuilder newString = new StringBuilder();
741 
742         for (int i = 0; i < value.length(); i++) {
743             final char c = value.charAt(i);
744             if (c == '.') {
745                 // stop if we hit a decimal point
746                 break;
747             } else if (c >= '0' && c <= '9' || c == '-') {
748                 newString.append(c);
749             }
750         }
751         // check to make sure we do not have a single length string with
752         // just a minus sign
753         final int sLen = newString.length();

756             return "0";
757         }

867     /**
868      * Returns a definition of pz column metadata from a given
869      * pz datastructure held in an SQL database
870      *
871      * @param con
872      *          Database connection containing the Datafile and Datastructure
873      *          tables
874      * @param dataDefinition
875      *          Name of the data definition stored in the Datafile table
876      * @param parser
877      * 			Instance of the parser being used for the file.  It will be checked to get the table names
878      * 			for the DATASTRUCTURE table and DATAFILE table.
879      * @throws SQLException
880      * @return List
881      */
882     public static List<ColumnMetaData> buildMDFromSQLTable(final Connection con, final String dataDefinition, final Parser parser)
883             throws SQLException {
884         final List<ColumnMetaData> cmds = new ArrayList<ColumnMetaData>();
885         final String dfTbl = parser != null ? parser.getDataFileTable() : "DATAFILE";
886         final String dsTbl = parser != null ? parser.getDataStructureTable() : "DATASTRUCTURE";
887         final StringBuilder sqlSb = new StringBuilder();
888 
889         sqlSb.append("SELECT * FROM ").append(dfTbl).append(" INNER JOIN ").append(dsTbl).append(" ON ").append(dfTbl).append(".DATAFILE_NO = ")
890                 .append(dsTbl).append(".DATAFILE_NO " + "WHERE DATAFILE_DESC = ? ORDER BY DATASTRUCTURE_COL_ORDER");
891 
892         try (PreparedStatement stmt = con.prepareStatement(sqlSb.toString())) { // always use PreparedStatement
893             // as the DB can do clever things.
894             stmt.setString(1, dataDefinition);
895             try (ResultSet rs = stmt.executeQuery()) {
896 
897                 int recPosition = 1;
898                 // put array of columns together. These will be used to put together
899                 // the dataset when reading in the file
900                 while (rs.next()) {
901 
902                     final ColumnMetaData column = new ColumnMetaData();
903                     column.setColName(rs.getString("DATASTRUCTURE_COLUMN"));
904                     column.setColLength(rs.getInt("DATASTRUCTURE_LENGTH"));
905                     column.setStartPosition(recPosition);
906                     column.setEndPosition(recPosition + rs.getInt("DATASTRUCTURE_LENGTH") - 1);
907                     recPosition += rs.getInt("DATASTRUCTURE_LENGTH");
908 
909                     cmds.add(column);
910                 }
911             }
912             if (cmds.isEmpty()) {
913                 throw new FPException("Data File Key [" + dataDefinition + "] Is Not In The database OR No Columns Specified In Table");
914             }
915         }
916         return cmds;
917     }
