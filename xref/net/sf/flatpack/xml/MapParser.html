203     private static List<ColumnMetaData> parseColumnElements(NodeList nodeList) {
204         final Set<String> columnNames = new HashSet<>();
205         final List<ColumnMetaData> columnResults = new ArrayList<>();
206         for (int i = 0; i < nodeList.getLength(); i++) {
207             Node node = nodeList.item(i);
208 
209             final String nodeName = node.getNodeName();
210             // final String nodeValue = node.getNodeValue();
211 
212             switch (nodeName) {
213             case "COLUMN":
214                 final NamedNodeMap attributes = node.getAttributes();
215                 System.out.println("COL:" + attributes.getNamedItem("name"));
216                 final ColumnMetaData cmd = new ColumnMetaData();
217 
218                 // make sure the name attribute is present on the column
219                 final String columnName = attributes.getNamedItem("name").getTextContent();
220                 if (columnName == null) {
221                     throw new IllegalArgumentException("Name attribute is required on the column tag!");
222                 }
223 
224                 // make sure the names in columnInfo are unique
225                 if (columnNames.contains(columnName)) {
226                     throw new IllegalArgumentException("Duplicate name column '" + columnName + "'");
227                 }
228 
229                 cmd.setColName(columnName);
230                 columnNames.add(columnName);
231 
232                 // check to see if the column length can be set
233                 if (attributes.getNamedItem("length") != null) {
234                     try {
235                         cmd.setColLength(Integer.parseInt(attributes.getNamedItem("length").getTextContent()));
236                     } catch (final Exception ex) {
237                         throw new IllegalArgumentException(
238                                 "LENGTH ATTRIBUTE ON COLUMN ELEMENT MUST BE AN INTEGER.  GOT: " + attributes.getNamedItem("length").getTextContent(),
239                                 ex);
240                     }
241                 }
242                 columnResults.add(cmd);
243                 break;
244             }
245         }
246         return columnResults;
247     }

412     /**
413      * New method based on Reader. Reads the XMLDocument for a PZMetaData
414      * file from an InputStream, WebStart compatible. Parses the XML file, and
415      * returns a Map containing Lists of ColumnMetaData.
416      *
417      * @param xmlStreamReader
418      * @param pzparser
419      *          Can be null.  Allows additional opts to be set during the XML map read
420      * @return Map &lt;records&gt; with their corresponding
421      * @throws IOException
422      * @throws SAXException 
423      * @throws ParserConfigurationException 
424      */
425     public static MetaData parseMap(final Reader xmlStreamReader, final Parser pzparser)
426             throws IOException, ParserConfigurationException, SAXException {
427         final Map map = parse2(xmlStreamReader, pzparser);
428 
429         final List<ColumnMetaData> col = (List<ColumnMetaData>) map.get(FPConstants.DETAIL_ID);
430         map.remove(FPConstants.DETAIL_ID);
431 
432         final Map m = (Map) map.get(FPConstants.COL_IDX);
433         map.remove(FPConstants.COL_IDX);
434 
435         // loop through the map and remove anything else that is an index of FPConstancts.COL_IDX + _
436         // these were put in for the writer.
437         // TODO maybe these shoudld be thrown into the MetaData instead of just discarded, but they are unused
438         // in the Reader the moment. This parseMap is not utilized in the writer so it is safe to remove them here
439         final Iterator entrySetIt = map.entrySet().iterator();
440         while (entrySetIt.hasNext()) {
441             final Entry e = (Entry) entrySetIt.next();
442             if (((String) e.getKey()).startsWith(FPConstants.COL_IDX + "_")) {
443                 entrySetIt.remove();
444             }
445         }
446 
447         return new MetaData(col, m, map);
448     }
