101     /**
102      * New method based on Reader. Reads the XMLDocument for a PZMetaData
103      * file from an InputStream, WebStart compatible. Parses the XML file, and
104      * returns a Map containing Lists of ColumnMetaData.
105      *
106      * @param xmlStreamReader
107      * @param pzparser
108      *          Can be null.  Allows additional opts to be set during the XML map read
109      * @return Map &lt;records&gt; with their corresponding
110      * @throws IOException
111      * @throws JDOMException
112      */
113     public static Map<String, Object> parse(final Reader xmlStreamReader, final Parser pzparser) throws JDOMException, IOException {
114         if (xmlStreamReader == null) {
115             throw new NullPointerException("XML Reader Is Not Allowed To Be Null...");
116         }
117         final SAXBuilder builder = new SAXBuilder();
118         builder.setValidation(true);
119         // handle the ability to pull DTD from Jar if needed
120         builder.setEntityResolver(new ResolveLocalDTD());
121 
122         // JDOM started to blow up on the parse if the system id param was not
123         // specified
124         // not sure why this started to happen now. Was not making to
125         // EntityResolver to pull
126         // dtd out of the jar if needed
127         final Document document = builder.build(xmlStreamReader, "file:///");
128 
129         final Element root = document.getRootElement();
130 
131         // lets first get all of the columns that are declared directly under
132         // the PZMAP
133         List<ColumnMetaData> columns = getColumnChildren(root);
134         final Map<String, Object> mdIndex = new LinkedHashMap<String, Object>(); // retain the same order
135         // specified in the mapping
136         mdIndex.put(FPConstants.DETAIL_ID, columns); // always force detail
137         // to the top of
138         // the map no matter what
139         mdIndex.put(FPConstants.COL_IDX, ParserUtils.buidColumnIndexMap(columns, pzparser));
140 
141         // get all of the "record" elements and the columns under them
142         final Iterator<Element> recordDescriptors = root.getChildren("RECORD").iterator();
143         while (recordDescriptors.hasNext()) {
144             final Element xmlElement = recordDescriptors.next();
145 
146             if (xmlElement.getAttributeValue("id").equals(FPConstants.DETAIL_ID)) {
147                 // make sure the id attribute does not have a value of "detail" this
148                 // is the harcoded
149                 // value we are using to mark columns specified outside of a
150                 // <RECORD> element
151                 throw new IllegalArgumentException("The ID 'detail' on the <RECORD> element is reserved, please select another id");
152             }
153 
154             columns = getColumnChildren(xmlElement);
155             final XMLRecordElement xmlre = new XMLRecordElement();
156             xmlre.setColumns(columns, pzparser);
157             xmlre.setIndicator(xmlElement.getAttributeValue("indicator"));
158             xmlre.setElementNumber(convertAttributeToInt(xmlElement.getAttribute("elementNumber")));
159             xmlre.setStartPosition(convertAttributeToInt(xmlElement.getAttribute("startPosition")));
160             xmlre.setEndPositition(convertAttributeToInt(xmlElement.getAttribute("endPosition")));
161             xmlre.setElementCount(convertAttributeToInt(xmlElement.getAttribute("elementCount")));
162             mdIndex.put(xmlElement.getAttributeValue("id"), xmlre);
163             // make a column index for non detail records
164             mdIndex.put(FPConstants.COL_IDX + "_" + xmlElement.getAttributeValue("id"), ParserUtils.buidColumnIndexMap(columns, pzparser));
165         }
166 
167         if (showDebug) {
168             setShowDebug(mdIndex);
169         }
170 
171         return mdIndex;
172     }

174     // helper to convert to integer
175     private static int convertAttributeToInt(final Attribute attribute) {
176         if (attribute == null) {
177             return 0;
178         }
179 
180         try {
181             return attribute.getIntValue();
182         } catch (final Exception ignore) {
183             return 0;
184         }
185     }

187     // helper to retrieve the "COLUMN" elements from the given parent
188     private static List<ColumnMetaData> getColumnChildren(final Element parent) {
189         final List<ColumnMetaData> columnResults = new ArrayList<ColumnMetaData>();
190         final Set<String> columnNames = new HashSet<String>();
191         final Iterator<Element> xmlChildren = parent.getChildren("COLUMN").iterator();
192 
193         while (xmlChildren.hasNext()) {
194             final ColumnMetaData cmd = new ColumnMetaData();
195             final Element xmlColumn = xmlChildren.next();
196 
197             // make sure the name attribute is present on the column
198             final String columnName = xmlColumn.getAttributeValue("name");
199             if (columnName == null) {
200                 throw new IllegalArgumentException("Name attribute is required on the column tag!");
201             }
202 
203             // make sure the names in columnInfo are unique
204             if (columnNames.contains(columnName)) {
205                 throw new IllegalArgumentException("Duplicate name column '" + columnName + "'");
206             }
207 
208             cmd.setColName(columnName);
209             columnNames.add(columnName);
210 
211             // check to see if the column length can be set
212             if (xmlColumn.getAttributeValue("length") != null) {
213                 try {
214                     cmd.setColLength(Integer.parseInt(xmlColumn.getAttributeValue("length")));
215                 } catch (final Exception ex) {
216                     throw new IllegalArgumentException(
217                             "LENGTH ATTRIBUTE ON COLUMN ELEMENT MUST BE AN INTEGER.  GOT: " + xmlColumn.getAttributeValue("length"), ex);
218                 }
219             }
220             columnResults.add(cmd);
221         }
222 
223         return columnResults;
224     }

236     private static void setShowDebug(final Map<String, Object> xmlResults) {
237         for (final Entry<String, Object> entry : xmlResults.entrySet()) {
238             XMLRecordElement xmlrecEle = null;
239             final String recordID = entry.getKey();
240             List<ColumnMetaData> columns = null;
241             if (recordID.equals(FPConstants.DETAIL_ID)) {
242                 columns = (List<ColumnMetaData>) entry.getValue();
243             } else {
244                 xmlrecEle = (XMLRecordElement) entry.getValue();
245                 columns = xmlrecEle.getColumns();
246             }
247 
248             LOGGER.debug(">>>>Column MD Id: " + recordID);
249             if (xmlrecEle != null) {
250                 LOGGER.debug("Start Position: " + xmlrecEle.getStartPosition() + " " + "End Position: " + xmlrecEle.getEndPositition() + " "
251                         + "Element Number: " + xmlrecEle.getElementNumber() + " " + "Indicator: " + xmlrecEle.getIndicator());
252             }
253             for (final ColumnMetaData cmd : columns) {
254                 LOGGER.debug("Column Name: " + cmd.getColName() + " LENGTH: " + cmd.getColLength());
255 
256             }
257         }

260     /**
261      * New method based on Reader. Reads the XMLDocument for a PZMetaData
262      * file from an InputStream, WebStart compatible. Parses the XML file, and
263      * returns a Map containing Lists of ColumnMetaData.
264      *
265      * @param xmlStreamReader
266      * @param pzparser
267      *          Can be null.  Allows additional opts to be set during the XML map read
268      * @return Map &lt;records&gt; with their corresponding
269      * @throws IOException
270      * @throws JDOMException
271      */
272     public static MetaData parseMap(final Reader xmlStreamReader, final Parser pzparser) throws JDOMException, IOException {
273         final Map map = parse(xmlStreamReader, pzparser);
274 
275         final List<ColumnMetaData> col = (List<ColumnMetaData>) map.get(FPConstants.DETAIL_ID);
276         map.remove(FPConstants.DETAIL_ID);
277 
278         final Map m = (Map) map.get(FPConstants.COL_IDX);
279         map.remove(FPConstants.COL_IDX);
280 
281         // loop through the map and remove anything else that is an index of FPConstancts.COL_IDX + _
282         // these were put in for the writer.
283         // TODO maybe these shoudld be thrown into the MetaData instead of just discarded, but they are unused
284         // in the Reader the moment. This parseMap is not utilized in the writer so it is safe to remove them here
285         final Iterator entrySetIt = map.entrySet().iterator();
286         while (entrySetIt.hasNext()) {
287             final Entry e = (Entry) entrySetIt.next();
288             if (((String) e.getKey()).startsWith(FPConstants.COL_IDX + "_")) {
289                 entrySetIt.remove();
290             }
291         }
292 
293         return new MetaData(col, m, map);
294     }
