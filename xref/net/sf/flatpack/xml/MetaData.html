71      public int getColumnIndex(final String key, final String columnName) {
72          int idx = -1;
73          if (key != null && !key.equals(FPConstants.DETAIL_ID) && !key.equals(FPConstants.COL_IDX)) {
74              idx = xmlRecordElements.get(key).getColumnIndex(columnName);
75          } else if (key == null || key.equals(FPConstants.DETAIL_ID)) {
76              final Integer i = (Integer) columnIndexMap.get(columnName);
77              if (i != null) { // happens when the col name does not exist in the
78                  // mapping
79                  idx = i.intValue();
80              }
81          }
82          return idx;
83      }
