112     public void setColumns(final List<ColumnMetaData> columnsToUse, final Parser p) {
113         this.columns = columnsToUse;
114         this.columnIndex = ParserUtils.buidColumnIndexMap(columns, p);
115     }

117     /**
118      * Returns the index of the column name.
119      *
120      * @author Benoit Xhenseval
121      * @param colName
122      * @return -1 if the column name does not exist.
123      */
124     public int getColumnIndex(final String colName) {
125         int idx = -1;
126         if (columnIndex != null) {
127             final Integer i = columnIndex.get(colName);
128             if (i != null) {
129                 idx = i.intValue();
130             }
131         }
132         return idx;
133     }
