33  package net.sf.flatpack;
34  
35  /**
36   * This class holds errors that occurred while parsing or processing a data file.
37   *
38   * @author Paul Zepernick
39   * @version 2.0
40   */
41  public class DataError {
42      private final String errorDesc;
43  
44      private final int lineNo;
45  
46      private final int errorLevel;
47  
48      private final String rawData;
49  
50      /**
51       *
52       * @param errorDesc
53       *          Text description of the error that occurred
54       * @param lineNo
55       *          Line number in the data file the error occurred on
56       * @param errorLevel
57       *          Level of the error (1=warning, 2=moderate, 3=severe)
58       */
59      public DataError(final String errorDesc, final int lineNo, final int errorLevel) {
60          this(errorDesc, lineNo, errorLevel, null);
61      }
62  
63      /**
64       *
65       * @param errorDesc
66       *          Text description of the error that occurred
67       * @param lineNo
68       *          Line number in the data file the error occurred on
69       * @param errorLevel
70       *          Level of the error (1=warning, 2=moderate, 3=severe)
71       * @param rawData
72       *          String of data which the parse failed on
73       */
74      public DataError(final String errorDesc, final int lineNo, final int errorLevel, final String rawData) {
75          this.errorDesc = errorDesc;
76          this.lineNo = lineNo;
77          this.errorLevel = errorLevel;
78          this.rawData = rawData;
79      }
80  
81      /**
82       * Returns the errorDesc.
83       *
84       * @return String
85       */
86      public String getErrorDesc() {
87          return errorDesc;
88      }
89  
90      /**
91       * Returns the errorLevel.
92       *
93       * @return int
94       */
95      public int getErrorLevel() {
96          return errorLevel;
97      }
98  
99      /**
100      * Returns the lineNo.
101      *
102      * @return int
103      */
104     public int getLineNo() {
105         return lineNo;
106     }
107 
108     @Override
109     public String toString() {
110         final StringBuilder buf = new StringBuilder();
111         buf.append("Line:").append(lineNo).append(" Level:").append(errorLevel).append(" Desc:").append(errorDesc)
112         .append(System.getProperty("line.separator"));
113         return buf.toString();
114     }
115 
116     /**
117      * Option must be set on parser, otherwise this is
118      * null by default
119      *
120      * @return the rawData
121      */
122     public String getRawData() {
123         return rawData;
124     }
125 }
