52  /**
53   * @author paul zepernick
54   *
55   * resorts an array of objects. Arrays get sorted by OrderElements. Right now,
56   * this class will only handle string comparisons.
57   *
58   * @version 2.0
59   */
60  public class OrderBy implements Comparator, Serializable {
61      private static final long serialVersionUID = 5622406168247149895L;
62  
63      /** collection of order elements to sort by */
64      private final List<OrderColumn> orderbys = new ArrayList<OrderColumn>();
65  
66      private MetaData metaData;
67  
68      private Parser parser;
69  
70      /**
71       * Adds an order element to the sort.
72       *
73       * @param oc -
74       *            OrderColumn
75       */
76      public void addOrderColumn(final OrderColumn oc) {
77          orderbys.add(oc);
78      }
79  
80      /**
81       * overridden from the Comparator class.
82       *
83       * Performs the sort
84       *
85       * @return int
86       */
87      @Override
88      public int compare(final Object arg0, final Object arg1) {
89          final Row row0 = (Row) arg0;
90          final Row row1 = (Row) arg1;
91          int result = 0;
92  
93          for (int i = 0; i < orderbys.size(); i++) {
94              final OrderColumn oc = orderbys.get(i);
95              // null indicates "detail" record which is what the parser assigns
96              // to <column> 's setup outside of <record> elements
97              final String mdkey0 = row0.getMdkey() == null ? FPConstants.DETAIL_ID : row0.getMdkey();
98              final String mdkey1 = row1.getMdkey() == null ? FPConstants.DETAIL_ID : row1.getMdkey();
99  
100             // shift all non detail records to the bottom of the DataSet
101             if (!mdkey0.equals(FPConstants.DETAIL_ID) && !mdkey1.equals(FPConstants.DETAIL_ID)) {
102                 // keep headers / trailers in the same order at the bottom of
103                 // the DataSet
104                 return 0;
105             } else if (!mdkey0.equals(FPConstants.DETAIL_ID) || !mdkey1.equals(FPConstants.DETAIL_ID)) {
106                 return !mdkey0.equals(FPConstants.DETAIL_ID) ? 1 : 0;
107             }
108 
109             // convert to one type of case so the comparator does not take case
110             // into account when sorting
111             Comparable comp0 = null;
112             Comparable comp1 = null;
113             final String str0 = row0.getValue(
114                     ParserUtils.getColumnIndex(row0.getMdkey(), metaData, oc.getColumnName(), parser.isColumnNamesCaseSensitive())).toLowerCase(
115                             Locale.getDefault());
116             final String str1 = row1.getValue(
117                     ParserUtils.getColumnIndex(row1.getMdkey(), metaData, oc.getColumnName(), parser.isColumnNamesCaseSensitive())).toLowerCase(
118                             Locale.getDefault());
119             switch (oc.getSelectedColType()) {
120             case OrderColumn.COLTYPE_STRING:
121             default:
122                 comp0 = str0;
123                 comp1 = str1;
124                 break;
125             case OrderColumn.COLTYPE_NUMERIC:
126                 comp0 = Double.valueOf(ParserUtils.stripNonDoubleChars(str0));
127                 comp1 = Double.valueOf(ParserUtils.stripNonDoubleChars(str1));
128                 break;
129             case OrderColumn.COLTYPE_DATE:
130                 final SimpleDateFormat sdf = new SimpleDateFormat(oc.getDateFormatPattern());
131                 try {
132                     comp0 = sdf.parse(str0);
133                 } catch (final ParseException e) {
134                     comp0 = getBadDateDefault();
135 
136                 }
137 
138                 try {
139                     comp1 = sdf.parse(str1);
140                 } catch (final ParseException e) {
141                     comp1 = getBadDateDefault();
142                 }
143                 break;
144             }
145 
146             // multiply by the sort indicator to get a ASC or DESC result
147             result = comp0.compareTo(comp1) * oc.getSortIndicator();
148 
149             // if it is = 0 then the primary sort is done, and it can start the
150             // secondary sorts
151             if (result != 0) {
152                 break;
153             }
154         }
155 
156         return result;
157     }
158 
159     private Date getBadDateDefault() {
160         final Calendar defaultBadDt = new GregorianCalendar();
161         defaultBadDt.set(Calendar.MONTH, 1);
162         defaultBadDt.set(Calendar.YEAR, 1900);
163         defaultBadDt.set(Calendar.DAY_OF_MONTH, 1);
164         return defaultBadDt.getTime();
165     }

172     }

178         this.parser = parser;
179     }
