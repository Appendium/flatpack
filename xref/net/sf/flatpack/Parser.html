35  import java.util.stream.Stream;

37  /**
38   * PZParser is ready to parse the data and return an object that can then be
39   * traversed. The default parser should NOT handle short lines, the user can
40   * change it prior to calling parse.
41   *
42   * @author Benoit Xhenseval
43   * @author Paul Zepernick
44   */
45  public interface Parser {
46  
47      /**
48       * Start the parsing. Will return "null" if the parse fails and the DataSet
49       * cannot be created
50       *
51       * @return the data set resulting from parsing
52       */
53      DataSet parse();
54  
55      /**
56       * Parse the data and return an interface where one can extract one record at a time, until
57       * next returns false;
58       * @since 3.4
59       */
60      StreamingDataSet parseAsStream();
61  
62      /**
63       * Parse the data and return a stream or records;
64       * @since 4.0
65       */
66      Stream<Record> stream();
67  
68      /**
69       * @return true, lines with less columns then the amount of column headers
70       *         will be added as empty's instead of producing an error
71       */
72      boolean isHandlingShortLines();
73  
74      /**
75       * @param handleShortLines -
76       *            when flagged as true, lines with less columns then the amount
77       *            of column headers will be added as empty's instead of
78       *            producing an error
79       */
80      Parser setHandlingShortLines(boolean handleShortLines);
81  
82      /**
83       * @param addSuffixToDuplicateColumnNames
84       *          when true, add a count to duplicate colum names. eg the second column called "Asset" will become "Asset2".
85       */
86      Parser setAddSuffixToDuplicateColumnNames(boolean addSuffixToDuplicateColumnNames);
87  
88      /**
89       *
90       * @return true, detail lines with a length or column count &gt; the mapping
91       *         definition will be truncated and the reader will NOT register
92       *         these lines as errors in the DataError collection.
93       */
94      boolean isIgnoreExtraColumns();
95  
96      /**
97       *
98       * @param ignoreExtraColumns
99       *            when true, detail lines with a length or column count &gt; the
100      *            mapping definition will be truncated and the reader will NOT
101      *            register these lines as errors in the DataError collection.
102      */
103     Parser setIgnoreExtraColumns(boolean ignoreExtraColumns);
104 
105     /**
106      * Defaults to <code>true</code>.
107      * @return true, the parser will preserve leading whitespace in each column when splitting a line
108      */
109     boolean isPreserveLeadingWhitespace();
110 
111     /**
112      * Defaults to <code>true</code>.
113      * @param preserveLeadingWhitespace
114      *            when true, the parser will preserve leading whitespace in each column when splitting a line
115      */
116     Parser setPreserveLeadingWhitespace(boolean preserveLeadingWhitespace);
117 
118     /**
119      * Defaults to <code>false</code>.
120      * @return true, the parser will preserve trailing whitespace in each column when splitting a line
121      */
122     boolean isPreserveTrailingWhitespace();
123 
124     /**
125      * Defaults to <code>false</code>.
126      * @param preserveTrailingWhitespace
127      *            when true, the parser will preserve trailing whitespace in each column when splitting a line
128      */
129     Parser setPreserveTrailingWhitespace(boolean preserveTrailingWhitespace);
130 
131     /**
132      * Default is false
133      *
134      * @return true, column names will have to be an exact match when retrieving
135      *         the value of a column. Example when true: Column name =
136      *         AColumnName ; getString("acolumnname") would fail Example when
137      *         false: Column name = AColumnName ; getString("acolumnname") would
138      *         pass
139      */
140     boolean isColumnNamesCaseSensitive();
141 
142     /**
143      * Default is false
144      *
145      * @param columnNamesCaseSensitive
146      *            when true, column names will have to be an exact match when
147      *            retrieving the value of a column. Example when true: Column
148      *            name = AColumnName ; getString("acolumnname") would fail
149      *            Example when false: Column name = AColumnName ;
150      *            getString("acolumnname") would pass
151      */
152     Parser setColumnNamesCaseSensitive(boolean columnNamesCaseSensitive);
153 
154     /**
155      * Default is false
156      *
157      * @return true, warnings encountered during parsing will not be included in
158      *         the DataSet errors
159      */
160     boolean isIgnoreParseWarnings();
161 
162     /**
163      *
164      * @param ignoreParseWarnings
165      *            when true, warnings encountered during parsing will not be
166      *            included in the DataSet errors
167      */
168     Parser setIgnoreParseWarnings(boolean ignoreParseWarnings);
169 
170     /**
171      *
172      * @return true, empty Strings will get returned as NULL when calling
173      *         DataSet.getString()
174      */
175     boolean isNullEmptyStrings();
176 
177     /**
178      *
179      * @param nullEmptyStrings
180      *            when true, empty Strings will get returned as NULL when
181      *            calling DataSet.getString()
182      */
183     Parser setNullEmptyStrings(boolean nullEmptyStrings);
184 
185     /**
186      *
187      * @return flagEmptyRows when true, will analyze the row to see if all
188      *         elements are empty and place a flag on the DataSet indicating if
189      *         the row is empty
190      */
191     boolean isFlagEmptyRows();
192 
193     /**
194      * when true, will analyze the row to see if all elements are empty and
195      * place a flag on the DataSet indicating if the row is empty.  This will slow
196      * down the parse and should only be used when necessary.  It is off by default.
197      *
198      * @param flagEmptyRows
199      */
200     Parser setFlagEmptyRows(boolean flagEmptyRows);
201 
202     /**
203      * @return when true, the parser will place the data of the line which failed the parse and
204      * place it into the DataError object.  DataError.getRawData() can be called to retrieve
205      * the line.
206      */
207     boolean isStoreRawDataToDataError();
208 
209     /**
210      * when true, the parser will place the data of the line which failed the parse and
211      * place it into the DataError object.  DataError.getRawData() can be called to retrieve
212      * the line.
213      *
214      * @param storeRawDataToDataError
215      */
216     Parser setStoreRawDataToDataError(boolean storeRawDataToDataError);
217 
218     /**
219      * @return when true, the parser will place the data of the line into the DataSet object.
220      * DataSet.getRawData() can be called to retrieve the line.
221      */
222     boolean isStoreRawDataToDataSet();
223 
224     /**
225      * WARNING!!  Setting this option has potential to cause high memory usage.
226      *
227      * when true, the parser will place the data of the line into the DataSet object.
228      * DataSet.getRawData() can be called to retrieve the line.
229      *
230      * @param storeRawDataToDataError
231      */
232     Parser setStoreRawDataToDataSet(boolean storeRawDataToDataError);
233 
234     /**
235      * Returns the table name that will be used to read the MetaData from the db.  The
236      * default table name is DATAFILE.  This may be problimatic for some who are using case
237      * sensistive db table names or who wish to provide a different table name in the db.
238      *
239      * This is only applicable when using a database for file mappings.
240      *
241      * @return the DATAFILE table name
242      */
243     String getDataFileTable();
244 
245     /**
246      * Sets the table name that will be used to read the MetaData from the db.  The
247      * default table name is DATAFILE.  This may be problimatic for some who are using case
248      * sensistive db table names or who wish to provide a different table name in the db.
249      *
250      * This is only applicable when using a database for file mappings.
251      *
252      * @param dataFileTable
253      * 			Name of the table name to use in place of "DATAFILE"
254      */
255     Parser setDataFileTable(String dataFileTable);
256 
257     /**
258      * Returns the table name that will be used to read the MetaData from the db.  The
259      * default table name is DATASTRUCTURE.  This may be problimatic for some who are using case
260      * sensistive db table names or who wish to provide a different table name in the db.
261      *
262      * This is only applicable when using a database for file mappings.
263      *
264      * @return the DATASTRUCTURE table name
265      */
266     String getDataStructureTable();
267 
268     /**
269      * Sets the table name that will be used to read the MetaData from the db.  The
270      * default table name is DATASTRUCTURE.  This may be problimatic for some who are using case
271      * sensistive db table names or who wish to provide a different table name in the db.
272      *
273      * This is only applicable when using a database for file mappings.
274      *
275      * @param dataStructureTable
276      * 			Name of the table name to us in placfe of "DATASTRUCTURE"
277      *
278      */
279     Parser setDataStructureTable(String dataStructureTable);
