16  public class FixedLengthWriter extends AbstractWriter {
17      private static final int MAX_CHAR_TO_USE_LOOP = 8;
18      private final Map<?, ?> columnMapping;
19      private final char padChar;

21      protected FixedLengthWriter(final Map parsedMapping, final java.io.Writer output, final char padChar) throws IOException {
22          super(output);
23          this.columnMapping = parsedMapping;
24          this.padChar = padChar;

27      @Override
28      public Writer addRecordEntry(final String columnName, final Object value) {
29          if (value != null) {
30              final ColumnMetaData metaData = this.getColumnMetaData(columnName);
31  
32              final String valueString = value.toString();
33              if (valueString.length() > metaData.getColLength()) {
34                  throw new IllegalArgumentException(valueString + " exceeds the maximum length for column " + columnName + "("
35                          + metaData.getColLength() + ")");
36              }
37          }
38          return super.addRecordEntry(columnName, value);
39      }
40  
41      @Override
42      public Writer nextRecord() throws IOException {
43          for (final ColumnMetaData element : getColumnMetaData()) {
44              final Object value = this.getRowMap().get(element.getColName());
45              this.write(this.formattedValue(value, element));
46          }
47  
48          return super.nextRecord();
49      }
50  
51      protected char[] formattedValue(final Object val, final ColumnMetaData element) {
52          Object value = val;
53          if (value == null) {
54              // TODO DO: maybe have a way to substitute default values here?
55              value = "";
56          }
57  
58          String stringValue;
59          if (value instanceof BigDecimal) {
60              final BigDecimal bd = (BigDecimal) value;
61              stringValue = bd.signum() == 0 ? "0" : bd.toPlainString();
62          } else {
63              stringValue = value.toString();
64          }
65  
66          final int stringLength = stringValue.length();
67  
68          final int columnLength = element.getColLength();
69          final char[] formattedValue = new char[columnLength];
70  
71          /*
72           * Copy the contents of the original string; for Strings up to ~8-10
73           * characters this loop is consistently faster than String.getChars().
74           * Short Strings are usually the majority of values in fixed-width data
75           * columns.
76           */
77          final int numCharacters = Math.min(stringLength, columnLength);
78          if (numCharacters < MAX_CHAR_TO_USE_LOOP) {
79              for (int i = 0; i < numCharacters; i++) {
80                  formattedValue[i] = stringValue.charAt(i);
81              }
82          } else {
83              stringValue.getChars(0, numCharacters, formattedValue, 0);
84          }
85  
86          if (stringLength < columnLength) {
87              // pad if necessary
88              Arrays.fill(formattedValue, stringLength, columnLength, padChar);
89          }
90  
91          return formattedValue;
92      }

95      protected boolean validateColumnTitle(final String columnTitle) {
96          final Map<String, Integer> columnNameToIndex = (Map<String, Integer>) columnMapping.get(FPConstants.COL_IDX);
97          return columnNameToIndex.keySet().contains(columnTitle);
98      }
