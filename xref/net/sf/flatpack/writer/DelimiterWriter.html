24      protected DelimiterWriter(final Map columnMapping, final java.io.Writer output, final char delimiter, final char qualifier,
25              final WriterOptions options) throws IOException {
26          super(output);
27          this.delimiter = delimiter;
28          this.qualifier = qualifier;
29  
30          columnTitles = new ArrayList<String>();
31          final List<ColumnMetaData> columns = (List<ColumnMetaData>) columnMapping.get(FPConstants.DETAIL_ID);
32          for (final ColumnMetaData cmd : columns) {
33              columnTitles.add(cmd.getColName());
34          }
35          // write the column headers
36          if (options.isAutoPrintHeader()) {
37              printHeader();
38          }
39      }

41      protected void writeWithDelimiter(final Object value) throws IOException {
42          this.write(value);
43          this.write(delimiter);
44      }

46      @Override
47      protected void write(final Object value) throws IOException {
48          String stringValue = "";
49  
50          if (value != null) {
51              // TODO DO: format the value
52              if (value instanceof BigDecimal) {
53                  final BigDecimal bd = (BigDecimal) value;
54                  stringValue = bd.signum() == 0 ? "0" : bd.toPlainString();
55              } else {
56                  stringValue = value.toString();
57              }
58          }
59  
60          final boolean needsQuoting = stringValue.indexOf(delimiter) != -1 //
61                  || qualifier != FPConstants.NO_QUALIFIER && stringValue.indexOf(qualifier) != -1 //
62                  || stringValue.indexOf('\n') != -1;
63          // || stringValue.split("\r\n|\r|\n").length > 1;
64  
65          if (needsQuoting) {
66              super.write(qualifier);
67          }
68  
69          super.write(stringValue);
70  
71          if (needsQuoting) {
72              super.write(qualifier);
73          }
74      }
75  
76      protected void writeColumnTitles() throws IOException {
77          final Iterator<String> titleIter = columnTitles.iterator();
78          while (titleIter.hasNext()) {
79              final String title = titleIter.next();
80  
81              if (titleIter.hasNext()) {
82                  this.writeWithDelimiter(title);
83              } else {
84                  this.write(title);
85              }
86          }
87      }
88  
89      protected void writeRow() throws IOException {
90          final Iterator<String> titlesIter = columnTitles.iterator();
91          while (titlesIter.hasNext()) {
92              final String columnTitle = titlesIter.next();
93              if (getRowMap() != null) {
94                  if (titlesIter.hasNext()) {
95                      writeWithDelimiter(getRowMap().get(columnTitle));
96                  } else {
97                      write(getRowMap().get(columnTitle));
98                  }
99              }
100         }
101     }
