114             array = new String[cmds.size()];

312     @Override
313     public boolean isAnError(final int lineNo) {
314         for (int i = 0; i < errors.size(); i++) {

316                 return true;
317             }
318         }

322     /*
323      * (non-Javadoc)
324      *
325      * @see net.sf.flatpack.DataSet#next()
326      */
327     @Override
328     public boolean next() {
329         if (pointer < rows.size() && pointer + 1 != rows.size()) {
330             pointer++;
331             currentRecord = new RowRecord(rows.get(pointer), metaData, parser.isColumnNamesCaseSensitive(), pzConvertProps, strictNumericParse,
332                     upperCase, lowerCase, parser.isNullEmptyStrings());
333             return true;
334         }
335         currentRecord = null;
336         return false;
337     }

366         if (pointer <= 0) {

371         currentRecord = new RowRecord(rows.get(pointer), metaData, parser.isColumnNamesCaseSensitive(), pzConvertProps, strictNumericParse, upperCase,

514      *             Parser.isFlagEmptyRows() must be set to true before using
515      *             this
516      * @throws FPException
517      *             if cursor is on an invalid row
518      */
519     @Override
520     public boolean isRowEmpty() {
521         if (!parser.isFlagEmptyRows()) {
522             // flag empty rows needs to be set for this functionality
523             // throw an exception
524             throw new FPInvalidUsageException("Parser.isFlagEmptyRows(true) must be set before using isRowEmpty()");
525         }
526 
527         if (pointer < 0) {
528             throw new FPException("Cursor on invalid row..  Make sure next() is called and returns true");
529         }
530 
531         return rows.get(pointer).isEmpty();
532     }
