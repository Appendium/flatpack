34  package net.sf.flatpack.converter;

36  import net.sf.flatpack.util.ParserUtils;
37  
38  /**
39   * Returns a Double
40   * Non numeric chars are removed from the string
41   * before converting
42   *
43   * @author Paul Zepernick
44   */
45  public class ConvertDouble implements Converter {
46  
47      /*
48       * (non-Javadoc)
49       *
50       * @see net.sf.flatpack.converter#convertValue(java.lang.String)
51       */
52      @Override
53      public Object convertValue(final String valueToConvert) {
54          return new Double(ParserUtils.stripNonDoubleChars(valueToConvert));
55      }
