33  package net.sf.flatpack.structure;
34  
35  import java.util.ArrayList;
36  import java.util.List;
37  
38  /**
39   * This holds a collection of columns and their values along with the actual
40   * row number they appear in the flat file
41   *
42   * @author Paul Zepernick
43   */
44  public class Row {
45      /** List to hold all columns that exist in the row */
46      private List<String> cols;
47  
48      /** Row number in flat file */
49      private int rowNumber;
50  
51      /** key to the MD for this row, null will indicate it is "detail" MD */
52      private String mdkey;
53  
54      private boolean empty;
55  
56      private String rawData;
57  
58      /**
59       * Constructs a new Row
60       *
61       */
62      public Row() {
63          cols = new ArrayList<String>();
64      }
65  
66      /**
67       * Adds a column to a row
68       *
69       * @param colValue -
70       *            String value to add to the row
71       */
72      public void addColumn(final String colValue) {
73          cols.add(colValue);
74      }
75  
76      /**
77       * Appends the List of Strings to the existing columns in the row
78       *
79       * @param columns -
80       *            List of Strings to append to the row
81       */
82      public void addColumn(final List<String> columns) {
83          if (cols == null) {
84              cols = new ArrayList<String>();
85          }
86          cols.addAll(columns);
87      }
88  
89      /**
90       * Returns the value of a column for a specified column name
91       *
92       * @param colPosition -
93       *            int position of the column in the array
94       * @return String value of column
95       */
96      public String getValue(final int colPosition) {
97          return cols.get(colPosition);
98      }
99  
100     /**
101      * Set the value of a column for a specified column name
102      *
103      * @param columnIndex -
104      *            column number to change
105      * @param value -
106      *            String column value
107      */
108     public void setValue(final int columnIndex, final String value) {
109         cols.set(columnIndex, value);
110     }
111 
112     /**
113      * Returns the rowNumber.
114      *
115      * @return int
116      */
117     public int getRowNumber() {
118         return rowNumber;
119     }
120 
121     /**
122      * Sets the rowNumber.
123      *
124      * @param rowNumber
125      *            The rowNumber to set
126      */
127     public void setRowNumber(final int rowNumber) {
128         this.rowNumber = rowNumber;
129     }
130 
131     /**
132      * Returns the cols for the row.
133      *
134      * @return Vector
135      */
136     public List<String> getCols() {
137         return cols;
138     }
139 
140     /**
141      * Set the columns for the row.
142      *
143      * @param cols -
144      *            Vector of Strings
145      */
146     public void setCols(final List<String> cols) {
147         this.cols = cols;
148     }
149 
150     /**
151      * @return Returns the mdkey.
152      */
153     public String getMdkey() {
154         return mdkey;
155     }
156 
157     /**
158      * @param mdkey
159      *            The mdkey to set.
160      */
161     public void setMdkey(final String mdkey) {
162         this.mdkey = mdkey;
163     }
164 
165     /**
166      * All columns in the row are empty
167      *
168      * @return the empty
169      */
170     public boolean isEmpty() {
171         return empty;
172     }
173 
174     /**
175      * When true, all columns in the row are empty
176      *
177      * @param empty the empty to set
178      */
179     public void setEmpty(final boolean empty) {
180         this.empty = empty;
181     }
182 
183     /**
184      * Raw data used to create the columns for the row
185      *
186      * @return the rawData
187      */
188     public String getRawData() {
189         return rawData;
190     }
191 
192     /**
193      * Raw data used to create the columns for the row
194      *
195      * @param rawData the rawData to set
196      */
197     public void setRawData(final String rawData) {
198         this.rawData = rawData;
199     }
200 }
