001package net.sf.flatpack.parserutils;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import junit.framework.TestCase;
007import net.sf.flatpack.structure.ColumnMetaData;
008import net.sf.flatpack.util.FixedWidthParserUtils;
009import net.sf.flatpack.util.ParserUtils;
010
011/**
012 * Test the functionality of a fixed width parse
013 *
014 * @author Paul Zepernick
015 */
016public class FixedWidthParserUtilsTest extends TestCase {
017
018    /**
019     * Test fixed width text
020     *
021     */
022    public void testFixedParse() {
023        check(new String[] { "test", "test", "test" }, new int[] { 5, 10, 20 }, new String[] { "test", "test", "test" }, true, false);
024
025        check(new String[] { "test with some space", "test", "test" }, new int[] { 300, 10, 20 }, new String[] { "test with some space", "test",
026        "test" }, true, false);
027
028        final String[] textWithLeadingAndTrailing = { "  test with leading and trailing    ", "  test ", "test" };
029        check(textWithLeadingAndTrailing, new int[] { 36, 7, 4 }, textWithLeadingAndTrailing, true, true);
030        check(textWithLeadingAndTrailing, new int[] { 36, 7, 4 }, new String[] { "  test with leading and trailing", "  test", "test" }, true, false);
031        check(textWithLeadingAndTrailing, new int[] { 36, 7, 4 }, new String[] { "test with leading and trailing    ", "test ", "test" }, false, true);
032    }
033
034    private void check(final String[] columnData, final int[] lengths, final String[] expected, final boolean preserveLeading,
035            final boolean preserveTrailing) {
036        final List<ColumnMetaData> columnMetaData = new ArrayList<ColumnMetaData>();
037
038        assertEquals("data and col lengths different size...", columnData.length, lengths.length);
039
040        for (final int length : lengths) {
041            final ColumnMetaData cmd = new ColumnMetaData();
042            cmd.setColLength(length);
043            columnMetaData.add(cmd);
044        }
045
046        final StringBuilder lineToParse = new StringBuilder();
047        for (int i = 0; i < columnData.length; i++) {
048            // padd each column
049            lineToParse.append(columnData[i]).append(ParserUtils.padding(lengths[i] - columnData[i].length(), ' '));
050        }
051
052        final List<String> splitResult = FixedWidthParserUtils.splitFixedText(columnMetaData, lineToParse.toString(), preserveLeading,
053                preserveTrailing);
054
055        // compare the parse results to the expected results
056        assertEquals("did not return correct number of cols...", expected.length, splitResult.size());
057
058        for (int i = 0; i < expected.length; i++) {
059            assertEquals("col values don't match...", expected[i], splitResult.get(i));
060        }
061
062    }
063
064    public static void main(final String[] args) {
065        junit.textui.TestRunner.run(FixedWidthParserUtilsTest.class);
066    }
067}
