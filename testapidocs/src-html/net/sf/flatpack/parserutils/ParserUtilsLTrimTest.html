015package net.sf.flatpack.parserutils;
016
017import junit.framework.TestCase;
018import net.sf.flatpack.util.ParserUtils;
019
020/**
021 * Tests the functionality of the left trim in the ParserUtils
022 *
023 * @author paul zepernick
024 */
025public class ParserUtilsLTrimTest extends TestCase {
026    /**
027     * Make sure all spaces are properly removed from the front of the String
028     *
029     */
030    public void testTrimLeadingSpaces() {
031        final String testS = "     RemoveAll SpacesFromMe";
032        assertEquals("RemoveAll SpacesFromMe", ParserUtils.lTrim(testS));
033    }
034
035    /**
036     * Make sure all spaces are properly removed from the front of the String
037     * leaving the trailing spaces at the end
038     *
039     */
040    public void testTrimLeadingSpacesWithTrailingSpaces() {
041        final String testS = "     RemoveAll SpacesFromMe     ";
042        final String tResult = ParserUtils.lTrim(testS);
043        assertEquals("RemoveAll SpacesFromMe     ", tResult);
044    }
045
046    /**
047     * Make sure all TAB chars are properly removed from the front of the String
048     * leaving the trailing spaces at the end
049     *
050     */
051    public void testTrimLeadingTabs() {
052        final String testS = "\t\t\tRemoveAll SpacesFromMe     ";
053        final String tResult = ParserUtils.lTrim(testS);
054        assertEquals("RemoveAll SpacesFromMe     ", tResult);
055    }
056
057    /**
058     * Make sure all TAB chars are properly removed from the front of the String
059     * leaving the trailing spaces at the end
060     *
061     */
062    public void testKeepLeadingTabs() {
063        final String testS = "     \t\t\tRemoveAll SpacesFromMe     ";
064        final String tResult = ParserUtils.lTrimKeepTabs(testS);
065        assertEquals("\t\t\tRemoveAll SpacesFromMe     ", tResult);
066    }
067
068    /**
069     * Ensure that spaces and tabs in the middle of the string will
070     * not be removed.
071     */
072    public void testWithTabsInMiddleAndEnd() {
073        assertEquals("RemoveAll \tSpaces \t\t", ParserUtils.lTrim("\t \t RemoveAll \tSpaces \t\t"));
074        assertEquals("\t \t RemoveAll \tSpaces \t\t ", ParserUtils.lTrimKeepTabs(" \t \t RemoveAll \tSpaces \t\t "));
075        assertEquals("\t \t RemoveAll \tSpaces \t\t", ParserUtils.lTrimKeepTabs("\t \t RemoveAll \tSpaces \t\t"));
076    }
077
078    /**
079     * Make sure all spaces are properly removed from the end of the String
080     *
081     */
082    public void testTrimTrailingSpaces() {
083        final String testS = "  RemoveAll trailing SpacesFromMe     ";
084        assertEquals("  RemoveAll trailing SpacesFromMe", ParserUtils.rTrim(testS));
085
086        assertEquals("", ParserUtils.rTrim("     "));
087        assertEquals("test", ParserUtils.rTrim("test"));
088    }
089
090    public static void main(final String[] args) {
091        junit.textui.TestRunner.run(ParserUtilsLTrimTest.class);
092    }
093}
