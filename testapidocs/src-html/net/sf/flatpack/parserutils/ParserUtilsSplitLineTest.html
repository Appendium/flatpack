019
020    private static final String[] DELIMITED_DATA_WITH_BREAKS = { "Column 1 \r\n\r\n Test After Break \r\n Another Break", "Column 2",
021        "Column 3 \r\n\r\n Test After Break", "Column 4", "Column 5 \r\n\r\n Test After Break\r\n Another Break" };

029    // 0 = delimiter
030    // 1 = qualifier
031    private static final char[][] DELIM_QUAL_PAIR = { { ',', '\"' }, { '\t', '\"' }, { '|', '\"' }, { '_', '\"' }, { ',', 0 }, { '|', 0 },
032        { '\t', 0 } };

034    /**
035     * Test without any line breaks
036     *
037     */
038    public void testNoLineBreaks() {
039        // loop down all delimiter qualifier pairs to test
040        for (final char[] element : DELIM_QUAL_PAIR) {
041            final char d = element[0];
042            final char q = element[1];
043
044            final String txtToParse = UnitTestUtils.buildDelimString(DELIMITED_DATA_NO_BREAKS, d, q);
045
046            final List splitLineResults = ParserUtils.splitLine(txtToParse, d, q, 10, false, false);
047
048            // check to make sure we have the same amount of elements which were
049            // expected
050            assertEquals("Expected size (d = [" + d + "] q = [" + (q != 0 ? String.valueOf(q) : "") + "] txt [" + txtToParse + "])",
051                    DELIMITED_DATA_NO_BREAKS.length, splitLineResults.size());
052
053            // loop through each value and compare what came back
054            for (int j = 0; j < DELIMITED_DATA_NO_BREAKS.length; j++) {
055                assertEquals("Data Element Value Does Not Match (d = [" + d + "] q = [" + q + "] txt [" + txtToParse + "])",
056                        DELIMITED_DATA_NO_BREAKS[j], (String) splitLineResults.get(j));
057            }
058        }
059    }

062     * Test with any line breaks
063     *
064     */
065    public void testLineBreaks() {
066        // loop down all delimiter qualifier pairs to test
067        for (final char[] element : DELIM_QUAL_PAIR) {
068            final char d = element[0];
069            final char q = element[1];
070

076            // expected

079
080            // loop through each value and compare what came back

082                assertEquals("Data Element Value Does Not Match (d = " + d + " q = " + q + ")", DELIMITED_DATA_WITH_BREAKS[j],
083                        (String) splitLineResults.get(j));
084            }

200    private void check(final String txtToParse, final char delim, final char qualifier, final String[] expected) {
201        final List splitLineResults = ParserUtils.splitLine(txtToParse, delim, qualifier, 10, false, false);
202
203        assertEquals("Did Not Get Amount Of Elements Expected (d = " + delim + " q = " + qualifier + ") txt [" + txtToParse + "]", expected.length,
204                splitLineResults.size());
205
206        for (int i = 0; i < expected.length; i++) {
207            assertEquals("expecting...", expected[i], splitLineResults.get(i));
208        }
209    }
