001package net.sf.flatpack.parserutils;
002
003import java.io.IOException;
004import java.math.BigDecimal;
005import java.util.List;
006import java.util.Properties;
007
008import junit.framework.TestCase;
009import net.sf.flatpack.util.FPConstants;
010import net.sf.flatpack.util.ParserUtils;
011
012/**
013 * Test misc methods in the ParserUtils
014 * class
015 *
016 * @author Paul Zepernick
017 */
018public class ParserUtilsTest extends TestCase {
019
020    public void testStripNonDouble() {
021        checkDoubleStrip("  $10.00   ", "10.00");
022        checkDoubleStrip("random chars  $10.00   more random", "10.00");
023        checkDoubleStrip(" $ 1 0 . 0 0 ", "10.00");
024        checkDoubleStrip("- $ 1 0 . 0 0 ", "-10.00");
025        checkDoubleStrip("1a2b3c4d.01234", "1234.01234");
026        checkDoubleStrip("-", "0");
027        checkDoubleStrip("  -  ", "0");
028        checkDoubleStrip("", "0");
029    }
030
031    private void checkDoubleStrip(final String txtToStrip, final String expected) {
032        final String stripRes = ParserUtils.stripNonDoubleChars(txtToStrip);
033        assertEquals("expecting...", stripRes, expected);
034    }
035
036    public void testStripNonLong() {
037        checkLongStrip("  $10.00   ", "10");
038        checkLongStrip("random chars  $10.00   more random", "10");
039        checkLongStrip(" $ 1 0 . 0 0 ", "10");
040        checkLongStrip("- $ 1 0 . 0 0 ", "-10");
041        checkLongStrip("1a2b3c4d.01234", "1234");
042        checkLongStrip("-", "0");
043        checkLongStrip("  -  ", "0");
044        checkLongStrip("", "0");
045    }
046
047    private void checkLongStrip(final String txtToStrip, final String expected) {
048        final String stripRes = ParserUtils.stripNonLongChars(txtToStrip);
049        assertEquals("expecting...", stripRes, expected);
050    }
051
052    public void testPZConverter() throws IOException {
053        final Properties convertProps = ParserUtils.loadConvertProperties();
054
055        assertEquals(ParserUtils.runPzConverter(convertProps, "$5.00C", Double.class), new Double("5.00"));
056        assertEquals(ParserUtils.runPzConverter(convertProps, "$5.00C", Integer.class), new Integer("5"));
057        assertEquals(ParserUtils.runPzConverter(convertProps, "$5.3556", BigDecimal.class), new BigDecimal("5.3556"));
058    }
059
060    public void testEmptyRow() {
061        final String data = ",,,";
062        final List l = ParserUtils.splitLine(data, ',', FPConstants.NO_QUALIFIER, 4, false, false);
063        assertEquals("list should be empty and is not...", ParserUtils.isListElementsEmpty(l), true);
064    }
065
066    public void testQualifiedNonMultiLine() {
067        final String data = "data 1-1,data 1-2,\"qualified,data 1-3,\"\n";
068         assertEquals(ParserUtils.isMultiLine(data.toCharArray(), ',', '\"'), false);
069    }
070
071    public void testQualifiedMultiLine() {
072        final String data = "data 1-1,data 1-2,\"qualified,data 1-3,\n" +
073                            "qualified data 1-3 continued from previous line\"\n";
074         assertEquals(ParserUtils.isMultiLine(data.toCharArray(), ',', '\"'), true);
075    }
076
077    public void testNonQualifiedNonMultiLine() {
078        final String data = "data 1-1,data 1-2,qualified,data 1-3\n";
079         assertEquals(ParserUtils.isMultiLine(data.toCharArray(), ',', '\"'), false);
080    }
081
082    public void testNonQualifiedMultiLine() {
083        // can't really have multiline without qualifier
084        final String data = "data 1-1,data 1-2,qualified,data 1-3\n" +
085                            "qualified data 1-3 continued from previous line\n";
086         assertEquals(ParserUtils.isMultiLine(data.toCharArray(), ',', '\"'), false);
087    }
088
089    public static void main(final String[] args) {
090        junit.textui.TestRunner.run(ParserUtilsTest.class);
091    }
092}
