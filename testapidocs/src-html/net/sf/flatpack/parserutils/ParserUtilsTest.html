022    public void testStripNonDouble() {
023        checkDoubleStrip("  $10.00   ", "10.00");
024        checkDoubleStrip("random chars  $10.00   more random", "10.00");
025        checkDoubleStrip(" $ 1 0 . 0 0 ", "10.00");
026        checkDoubleStrip("- $ 1 0 . 0 0 ", "-10.00");
027        checkDoubleStrip("1a2b3c4d.01234", "1234.01234");
028        checkDoubleStrip("-", "0");
029        checkDoubleStrip("  -  ", "0");
030        checkDoubleStrip("", "0");
031    }
032
033    private void checkDoubleStrip(final String txtToStrip, final String expected) {
034        final String stripRes = ParserUtils.stripNonDoubleChars(txtToStrip);
035        assertEquals("expecting...", stripRes, expected);
036    }
037
038    public void testStripNonLong() {
039        checkLongStrip("  $10.00   ", "10");
040        checkLongStrip("random chars  $10.00   more random", "10");
041        checkLongStrip(" $ 1 0 . 0 0 ", "10");
042        checkLongStrip("- $ 1 0 . 0 0 ", "-10");
043        checkLongStrip("1a2b3c4d.01234", "1234");
044        checkLongStrip("-", "0");
045        checkLongStrip("  -  ", "0");
046        checkLongStrip("", "0");
047    }
048
049    private void checkLongStrip(final String txtToStrip, final String expected) {
050        final String stripRes = ParserUtils.stripNonLongChars(txtToStrip);
051        assertEquals("expecting...", stripRes, expected);
052    }

054    public void testPZConverter() throws IOException {
055        final Properties convertProps = ParserUtils.loadConvertProperties();
056
057        assertEquals(ParserUtils.runPzConverter(convertProps, "$5.00C", Double.class), new Double("5.00"));
058        assertEquals(ParserUtils.runPzConverter(convertProps, "$5.00C", Integer.class), new Integer("5"));
059        assertEquals(ParserUtils.runPzConverter(convertProps, "$5.3556", BigDecimal.class), new BigDecimal("5.3556"));
060    }

062    public void testEmptyRow() {
063        final String data = ",,,";
064        final List l = ParserUtils.splitLine(data, ',', FPConstants.NO_QUALIFIER, 4, false, false);
065        assertEquals("list should be empty and is not...", ParserUtils.isListElementsEmpty(l), true);
066    }

068    public void testQualifiedNonMultiLine() {
069        final String data = "data 1-1,data 1-2,\"qualified,data 1-3,\"\n";
070        assertEquals(ParserUtils.isMultiLine(data.toCharArray(), ',', '\"'), false);
071    }
072
073    public void testQualifiedMultiLine() {
074        final String data = "data 1-1,data 1-2,\"qualified,data 1-3,\n" + "qualified data 1-3 continued from previous line\"\n";
075        assertEquals(ParserUtils.isMultiLine(data.toCharArray(), ',', '\"'), true);
076    }
077
078    public void testNonQualifiedNonMultiLine() {
079        final String data = "data 1-1,data 1-2,qualified,data 1-3\n";
080        assertEquals(ParserUtils.isMultiLine(data.toCharArray(), ',', '\"'), false);
081    }
082
083    public void testNonQualifiedMultiLine() {
084        // can't really have multiline without qualifier
085        final String data = "data 1-1,data 1-2,qualified,data 1-3\n" + "qualified data 1-3 continued from previous line\n";
086        assertEquals(ParserUtils.isMultiLine(data.toCharArray(), ',', '\"'), false);
087    }

089    private void testCsvSplit(String title, String line, String... expected) {
090        final List<String> splitLine = ParserUtils.splitLine(line, ',', '"', 5, false, false);
091        System.out.println(title + " [" + line + "] ==> " + splitLine);
092        assertThat(splitLine).as(title).containsExactly(expected);
093    }
094
095    private void testFancyCsvSplit(String title, String line, String... expected) {
096        final List<String> splitLine = ParserUtils.splitLine(line, ',', '|', 5, false, false);
097        System.out.println(title + " [" + line + "] ==> " + splitLine);
098        assertThat(splitLine).as(title).containsExactly(expected);
099    }
