003import static org.assertj.core.api.Assertions.assertThat;

005import java.io.IOException;

011import javax.xml.parsers.ParserConfigurationException;

015import junit.framework.TestCase;

021public class MapParserTest extends TestCase {
022    private static final String PZ_FIXED_MAP = "<?xml version='1.0'?>\n" +//
023            "<!DOCTYPE PZMAP SYSTEM  \"flatpack.dtd\" >\n" //
024            + "<PZMAP>\n" //
025            + "    <RECORD id=\"header\" startPosition=\"1\" endPosition=\"7\" indicator=\"HEADER \">\n"//
026            + "        <COLUMN name=\"INDICATOR\" length=\"7\" />\n"//
027            + "        <COLUMN name=\"HEADERDATA\" length=\"19\" />\n" //
028            + "    </RECORD>\n" //
029            + "    <COLUMN name=\"FIRSTNAME\" length=\"35\" />\n" //
030            + "    <COLUMN name=\"LASTNAME\" length=\"35\" />\n" //
031            + "    <COLUMN name=\"ADDRESS\" length=\"100\" />\n" //
032            + "    <COLUMN name=\"CITY\" length=\"100\" />\n" //
033            + "    <COLUMN name=\"STATE\" length=\"2\" />\n" //
034            + "    <COLUMN name=\"ZIP\" length=\"5\" />\n"
035            + "    <RECORD id=\"trailer\" startPosition=\"1\" endPosition=\"7\" indicator=\"TRAILER\">\n"//
036            + "        <COLUMN name=\"INDICATOR\" length=\"7\" />\n" //
037            + "        <COLUMN name=\"TRAILERDATA\" length=\"19\" />\n" //
038            + "    </RECORD>\n"//
039            + "</PZMAP> \n";
040    private static final String PZ_MAP = "<?xml version='1.0'?>\n" +  //
041            "<!DOCTYPE PZMAP SYSTEM\n" + //
042            "    \"flatpack.dtd\" >\n" + //
043            "<PZMAP>\n" + //
044            "    <COLUMN name=\"FIRSTnAME\" />\n" + //
045            "    <COLUMN name=\"LASTNAME\" />\n" + //
046            "    <COLUMN name=\"ADDRESS\" />\n" + //
047            "    <COLUMN name=\"CITY\" />\n" + //
048            "    <COLUMN name=\"STATE\" />\n" + //
049            "    <COLUMN name=\"ZIP\" />\n" + //
050            "</PZMAP> \n";
051    private static final String INVALID_DUPLICATE_COL = "<?xml version='1.0'?>\n" +  //
052            "<!DOCTYPE PZMAP SYSTEM\n" + //
053            "    \"flatpack.dtd\" >\n" + //
054            "<PZMAP>\n" + //
055            "    <COLUMN name=\"LASTNAME\" />\n" + //
056            "    <COLUMN name=\"LASTNAME\" />\n" + //
057            "    <COLUMN name=\"ZIP\" />\n" + //
058            "</PZMAP> \n";
059
060    public void testInvalidMap() throws IOException, ParserConfigurationException, SAXException {
061        try {
062            final Map<String, Object> parse = MapParser
063                    .parse(new InputStreamReader(this.getClass().getClassLoader().getResourceAsStream("BrokenMapping.pzmap.xml")), null);
064            fail("Expected Exception SAXException");
065        } catch (SAXException io) {
066            assertThat(io.getMessage()).contains("The element type \"COLUMN\" must be terminated by the matching end-tag \"</COLUMN>\"");
067        }
068    }
069
070    public void testParseFixedMap() throws IOException, ParserConfigurationException, SAXException {
071        final Map<String, Object> parse = MapParser.parse(new StringReader(PZ_FIXED_MAP), null);
072        assertThat(parse).hasSize(6);
073        final List<ColumnMetaData> details = (List<ColumnMetaData>) parse.get(FPConstants.DETAIL_ID);
074        assertThat(details).isNotNull();
075        assertThat(details).hasSize(6);
076        assertThat(details).extracting("colName").containsOnly("FIRSTNAME", "LASTNAME", "ADDRESS", "CITY", "STATE", "ZIP");
077        assertThat(details).extracting("colLength").containsOnly(35, 100, 2, 5);
078    }
079
080    public void testParseComplexFixedMap() throws IOException, ParserConfigurationException, SAXException {
081        final Map<String, Object> parse = MapParser.parse(new InputStreamReader(getClass().getClassLoader().getResourceAsStream("test-complex-fixed.xml")), null);
082        assertThat(parse).hasSize(8);
083        final XMLRecordElement details = (XMLRecordElement) parse.get("exchange");
084        assertThat(details).isNotNull();
085        assertThat(details.getIndicator()).isEqualToIgnoringCase("TE");
086        assertThat(details.getStartPosition()).isEqualTo(1);
087        assertThat(details.getEndPositition()).isEqualTo(2);
088        assertThat(details.getColumns()).isNotNull();
089        assertThat(details.getColumns()).hasSize(4);
090
091        assertThat(details.getColumns()).extracting("colName").containsOnly("RecordType", "StatusIndicator", "ExchangeCode", "ExchangeName");
092        assertThat(details.getColumns()).extracting("colLength").containsOnly(2, 1, 8, 259);
093    }
094
095    public void testParse() throws IOException, ParserConfigurationException, SAXException {
096        final Map<String, Object> parse = MapParser.parse(new StringReader(PZ_MAP), null);
097        assertThat(parse).hasSize(2);
098        final List<ColumnMetaData> details = (List<ColumnMetaData>) parse.get(FPConstants.DETAIL_ID);
099        assertThat(details).isNotNull();
100        assertThat(details).hasSize(6);
101        assertThat(details).extracting("colName").containsOnly("FIRSTnAME", "LASTNAME", "ADDRESS", "CITY", "STATE", "ZIP");
102    }
103
104    public void testInvalidDuplicate() throws IOException, ParserConfigurationException, SAXException {
105        final Parser parser = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader("hello"), ',', '"');
106        parser.setColumnNamesCaseSensitive(true);
107        try {
108            MapParser.parse(new StringReader(INVALID_DUPLICATE_COL), parser);
109            fail("Expecting exception due to duplicate col");
110        } catch (IllegalArgumentException iae) {
111            assertThat(iae.getMessage()).contains("'LASTNAME'");
112        }
113    }
114
115    public void testParseCaseSensitive() throws IOException, ParserConfigurationException, SAXException {
116        final Parser parser = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader("hello"), ',', '"');
117        parser.setColumnNamesCaseSensitive(true);
118        final Map<String, Object> parse = MapParser.parse(new StringReader(PZ_MAP), parser);
119        assertThat(parse).hasSize(2);
120        final List<ColumnMetaData> details = (List<ColumnMetaData>) parse.get(FPConstants.DETAIL_ID);
121        assertThat(details).isNotNull();
122        assertThat(details).hasSize(6);
123        assertThat(details).extracting("colName").containsOnly("FIRSTnAME", "LASTNAME", "ADDRESS", "CITY", "STATE", "ZIP");
124        final Map<String, Integer> colIndex = (Map<String, Integer>) parse.get(FPConstants.COL_IDX);
125        assertThat(colIndex).isNotEmpty();
126        assertThat(colIndex).containsOnlyKeys("FIRSTnAME", "LASTNAME", "ADDRESS", "CITY", "STATE", "ZIP");
127    }
128
129    public void testParseIgnoreCase() throws SAXException, IOException, ParserConfigurationException {
130        final Parser parser = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader("hello"), ',', '"');
131        parser.setColumnNamesCaseSensitive(false);
132        final Map<String, Object> parse = MapParser.parse(new StringReader(PZ_MAP), parser);
133        assertThat(parse).hasSize(2);
134        final List<ColumnMetaData> details = (List<ColumnMetaData>) parse.get(FPConstants.DETAIL_ID);
135        assertThat(details).isNotNull();
136        assertThat(details).hasSize(6);
137        assertThat(details).extracting("colName").containsOnly("FIRSTnAME", "LASTNAME", "ADDRESS", "CITY", "STATE", "ZIP");
138        final Map<String, Integer> colIndex = (Map<String, Integer>) parse.get(FPConstants.COL_IDX);
139        assertThat(colIndex).isNotEmpty();
140        assertThat(colIndex).containsOnlyKeys("firstname", "lastname", "address", "city", "state", "zip");
141    }
