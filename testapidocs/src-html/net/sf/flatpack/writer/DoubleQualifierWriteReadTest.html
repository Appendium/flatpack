001package net.sf.flatpack.writer;
002
003import java.io.ByteArrayInputStream;
004import java.io.StringWriter;
005import java.util.ArrayList;
006import java.util.HashMap;
007import java.util.LinkedHashSet;
008import java.util.List;
009import java.util.Map;
010import java.util.Optional;
011import java.util.Set;
012
013import org.slf4j.Logger;
014import org.slf4j.LoggerFactory;
015
016import junit.framework.TestCase;
017import net.sf.flatpack.DataSet;
018import net.sf.flatpack.DelimiterParser;
019import net.sf.flatpack.Record;
020
021public class DoubleQualifierWriteReadTest extends TestCase {
022
023    private static final Logger LOG = LoggerFactory.getLogger(DoubleQualifierWriteReadTest.class);
024
025    public void testDelimiter() throws Exception {
026        final java.io.Writer writer = new StringWriter();
027
028        final Set<String> header = new LinkedHashSet<String>();
029        header.add("Header1");
030        header.add("Header2");
031        header.add("Header3");
032
033        final List<Map<String, String>> dataList = new ArrayList<Map<String, String>>();
034        for (int i = 0; i < 2; i++) {
035            final Map<String, String> tempMap = new HashMap<String, String>(header.size());
036            for (final String it : header) {
037                tempMap.put(it, it + "_" + "Data \"\" " + i + " \"");
038            }
039            dataList.add(tempMap);
040        }
041        final DelimiterWriterFactory delimiterWriterFactory = new DelimiterWriterFactory(';', '"');
042        for (final String it : header) {
043            delimiterWriterFactory.addColumnTitle(it);
044        }
045        final Writer createWriter = delimiterWriterFactory.createWriter(writer, WriterOptions.getInstance().autoPrintHeader(true));
046
047        for (final Map<String, String> data : dataList) {
048            for (final Map.Entry<String, String> entry : data.entrySet() ) {
049                createWriter.addRecordEntry(entry.getKey(), entry.getValue());
050            }
051            createWriter.nextRecord();
052        }
053        createWriter.flush();
054        LOG.info("Content \n{}", writer.toString());
055
056
057        final ByteArrayInputStream file = new ByteArrayInputStream(writer.toString().getBytes());
058        final DelimiterParser parser = new DelimiterParser(file, ';', '"', false);
059        final DataSet parse = parser.parse();
060        parse.next();
061        final Optional<Record> record = parse.getRecord();
062        assertEquals("Header2_Data \"\"\"\" 0 \"\"", record.get().getString("Header2"));
063        LOG.info("Header2 \n{}", record.get().getString("Header2"));
064    }
065
066}
