001package net.sf.flatpack.writer;
002
003import junit.framework.TestCase;
004
005/**
006 *
007 * @author Dirk Holmes and Holger Holger Hoffstatte
008 */
009public abstract class PZWriterTestCase extends TestCase {
010    private final String lineSeparator = System.getProperty("line.separator");
011
012    protected String joinLines(final String line1, final String line2) {
013        if (line1 == null) {
014            throw new IllegalArgumentException("parameter string1 may not be null");
015        }
016
017        final StringBuilder result = new StringBuilder(line1);
018        result.append(lineSeparator);
019        if (line2 != null) {
020            result.append(line2);
021            result.append(lineSeparator);
022        }
023
024        return result.toString();
025    }
026
027    protected String normalizeLineEnding(final String line) {
028        return this.joinLines(line, null);
029    }
030}
