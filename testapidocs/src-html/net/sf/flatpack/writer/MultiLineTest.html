001package net.sf.flatpack.writer;
002
003import java.io.StringReader;
004import java.io.StringWriter;
005import java.util.ArrayList;
006import java.util.HashMap;
007import java.util.List;
008import java.util.Map;
009
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013import junit.framework.TestCase;
014import net.sf.flatpack.DataSet;
015import net.sf.flatpack.DefaultParserFactory;
016import net.sf.flatpack.structure.ColumnMetaData;
017import net.sf.flatpack.util.FPConstants;
018
019public class MultiLineTest extends TestCase {
020    private static final String NEW_LINE = System.getProperty("line.separator");
021    private static final String CONTENT = "this is a" + NEW_LINE + " multiline " + NEW_LINE + "....";
022
023    private static final Logger LOG = LoggerFactory.getLogger(MultiLineTest.class);
024    private Map<String, Object> mapping;
025
026    @Override
027    public void setUp() {
028        mapping = new HashMap<>();
029        final List<ColumnMetaData> listColumns = new ArrayList<>();
030        for (final String column : new String[] { "Id", "Description" }) {
031            final ColumnMetaData columnMetadata = new ColumnMetaData();
032            columnMetadata.setColName(column);
033            listColumns.add(columnMetadata);
034        }
035        mapping.put(FPConstants.DETAIL_ID, listColumns);
036    }
037
038    public void testMultipleLine() throws Exception {
039        final DelimiterWriterFactory factory = new DelimiterWriterFactory(mapping, ',', '"');
040
041        final StringWriter strWriter = new StringWriter();
042        final Writer flatMapWriter = factory.createWriter(strWriter, new WriterOptions().autoPrintHeader(false));
043        flatMapWriter.printHeader();
044        flatMapWriter.addRecordEntry("Id", "1");
045        flatMapWriter.addRecordEntry("Description", CONTENT);
046        flatMapWriter.nextRecord();
047        flatMapWriter.flush();
048        flatMapWriter.close();
049        LOG.info("CSV is \n{}", strWriter.toString());
050
051        final String toRead = strWriter.toString();
052        final DataSet ds = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(toRead), ',', '"').parse();
053        if (ds.next()) {
054            LOG.info("Parsed content \n{}", ds.getString("Description"));
055            assertEquals(ds.getString("Description"), CONTENT);
056        }
057
058    }
059
060}
