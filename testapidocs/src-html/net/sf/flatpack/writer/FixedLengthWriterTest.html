016public class FixedLengthWriterTest extends PZWriterTestCase {
017    public void testWriteFixedLength() throws Exception {
018        final StringWriter out = new StringWriter();
019        final Writer writer = new FixedWriterFactory(this.getMapping()).createWriter(out);
020
021        writer.addRecordEntry("LASTNAME", "DOE") // New fluent
022        .addRecordEntry("ADDRESS", "1234 CIRCLE CT") //
023        .addRecordEntry("STATE", "OH") //
024        .addRecordEntry("ZIP", "44035") //
025        .addRecordEntry("FIRSTNAME", "JOHN") //
026        .addRecordEntry("CITY", "ELYRIA") //
027        .nextRecord() //
028        .flush();
029
030        final String expected = this
031                .normalizeLineEnding("JOHN                               DOE                                1234 CIRCLE CT                                                                                      ELYRIA                                                                                              OH44035");
032        Assert.assertEquals(expected, out.toString());
033    }
034
035    public void testWriterWithDifferentFillChar() throws Exception {
036        final StringWriter out = new StringWriter();
037        final Writer writer = new FixedWriterFactory(this.getMapping(), '.').createWriter(out);
038
039        writer.addRecordEntry("LASTNAME", "DOE");
040        writer.addRecordEntry("ADDRESS", "1234 CIRCLE CT");
041        writer.addRecordEntry("STATE", "OH");
042        writer.addRecordEntry("ZIP", "44035");
043        writer.addRecordEntry("FIRSTNAME", "JOHN");
044        writer.addRecordEntry("CITY", "ELYRIA");
045        writer.nextRecord();
046        writer.flush();
047
048        final String expected = this
049                .normalizeLineEnding("JOHN...............................DOE................................1234 CIRCLE CT......................................................................................ELYRIA..............................................................................................OH44035");
050        Assert.assertEquals(expected, out.toString());
051    }
052
053    public void testCreateParserWithMalformedMappingFile() throws Exception {
054        try {
055            final InputStream mapping = this.getClass().getClassLoader().getResourceAsStream("BrokenMapping.pzmap.xml");
056            final InputStreamReader mappingReader = new InputStreamReader(mapping);
057            new FixedWriterFactory(mappingReader);
058            Assert.fail();
059        } catch (final InitialisationException ie) {
060            // this excecption must occur, mapping xml is invalid
061        }
062    }
063
064    public void testWriteStringWiderThanColumnDefinition() throws Exception {
065        final StringWriter out = new StringWriter();
066        final Writer writer = new FixedWriterFactory(this.getMapping()).createWriter(out);
067        try {
068            writer.addRecordEntry("STATE", "THISISTOOLONG");
069            Assert.fail("writing entries that are too long should fail");
070        } catch (final IllegalArgumentException iae) {
071            // expected exception
072        }
073    }
074
075    public void testWriteNullColumn() throws Exception {
076        final StringWriter out = new StringWriter();
077        final Writer writer = new FixedWriterFactory(this.getMapping()).createWriter(out);
078
079        writer.addRecordEntry("LASTNAME", "DOE");
080        writer.addRecordEntry("ADDRESS", "1234 CIRCLE CT");
081        writer.addRecordEntry("STATE", "OH");
082        writer.addRecordEntry("ZIP", "44035");
083        // note that we don't write a firstname
084        writer.addRecordEntry("FIRSTNAME", null);
085        writer.addRecordEntry("CITY", "ELYRIA");
086        writer.nextRecord();
087        writer.flush();
088
089        final String expected = this
090                .normalizeLineEnding("                                   DOE                                1234 CIRCLE CT                                                                                      ELYRIA                                                                                              OH44035");
091        Assert.assertEquals(expected, out.toString());
092    }
093
094    public void DONOTtestWriteDifferentRecords() throws Exception {
095        final String ls = System.getProperty("line.separator");
096        final StringWriter out = new StringWriter();
097        final Writer writer = new FixedWriterFactory(getMappingDiffRecordTypes()).createWriter(out);
098        // writer.setRecordId("header");
099        writer.addRecordEntry("recordtype", "H");
100        writer.addRecordEntry("headerdata1", "header data");
101        writer.nextRecord();
102
103        writer.addRecordEntry("recordtype", "D");
104        writer.addRecordEntry("detaildata1", "detail data");
105        writer.nextRecord();
106        writer.flush();
107
108        final StringBuilder expected = new StringBuilder();
109        expected.append("H");
110        expected.append("header data         ").append(ls);
111        expected.append("D");
112        expected.append("detail data         ").append(ls);
113
114        assertEquals("Checking writer for different record types...", expected.toString(), out.toString());
115
116    }
117
118    private Reader getMapping() {
119        final InputStream mapping = this.getClass().getClassLoader().getResourceAsStream("FixedLength.pzmap.xml");
120        return new InputStreamReader(mapping);
121    }
122
123    private Reader getMappingDiffRecordTypes() {
124        final String xml = "<?xml version=\"1.0\" encoding=\"ISO-8859-1\"?> \r\n" + "<!DOCTYPE PZMAP SYSTEM     \"pzfilereader.dtd\" > \r\n"
125                + "     <PZMAP>\r\n" + "                <RECORD id=\"header\" startPosition=\"1\" endPosition=\"1\" indicator=\"H\">"
126                + "                     <COLUMN name=\"recordtype\" length=\"1\" /> \r\n" + "                   <COLUMN name=\"headerdata1\" length=\"20\" /> \r\n" + "         </RECORD>"
127                + "             <COLUMN name=\"recordtype\" length=\"1\" /> \r\n" + "           <COLUMN name=\"detaildata1\" length=\"20\" /> \r\n" + " </PZMAP>";
128
129        return new StringReader(xml);
130    }
