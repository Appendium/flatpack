001package net.sf.flatpack;
002
003import java.io.StringReader;
004import java.util.List;
005import java.util.stream.Collectors;
006
007import junit.framework.TestCase;
008import net.sf.flatpack.util.FPConstants;
009
010/**
011 * Test methods in the DataSet
012 *
013 * @author Paul Zepernick
014 */
015public class DataSetFunctionalityTest extends TestCase {
016
017    public void testContains() {
018        DataSet ds;
019        final String cols = "column1,column2,column3\r\n value1,value2,value3";
020        final Parser p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
021        ds = p.parse();
022        ds.next();
023        assertEquals("column should NOT be found...", false, ds.contains("shouldnotcontain"));
024
025        assertEquals("column should be found...", true, ds.contains("column1"));
026    }
027
028    public void testContainsForStream() {
029        final String cols = "column1,column2,column3\r\n value1  ,value2,value3";
030        final Parser p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
031        final StreamingDataSet ds = p.parseAsStream();
032        assertTrue(ds.next());
033        assertEquals("column should NOT be found...", false, ds.getRecord().get().contains("shouldnotcontain"));
034        assertEquals("column should be found...", true, ds.getRecord().get().contains("column1"));
035        assertEquals("column should be found...", " value1", ds.getRecord().get().getString("column1"));
036    }
037
038    public void testDoNotPreserveSpace() {
039        final String cols = "column1,column2,column3\r\n value1  , value2,value3   ";
040        final Parser p1 = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
041        p1.setPreserveLeadingWhitespace(false);
042        final StreamingDataSet ds = p1.parseAsStream();
043        ds.next();
044        assertEquals("do not preserve leading space value1...", "value1", ds.getRecord().get().getString("column1"));
045        assertEquals("do not preserve leading space value2...", "value2", ds.getRecord().get().getString("column2"));
046        assertEquals("do not preserve leading space value3...", "value3", ds.getRecord().get().getString("column3"));
047    }
048
049    public void testDoPreserveLeadingSpace() {
050        final String cols = "column1,column2,column3\r\n value1  , value2,value3   ";
051        final Parser p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
052        final StreamingDataSet ds = p.parseAsStream();
053        ds.next();
054        assertEquals("value1...", " value1", ds.getRecord().get().getString("column1"));
055        assertEquals("value2...", " value2", ds.getRecord().get().getString("column2"));
056        assertEquals("value3...", "value3", ds.getRecord().get().getString("column3"));
057    }
058
059    public void testDoPreserveTrailingSpace() {
060        final String cols = "column1,column2,column3\r\n value1  , value2,value3   ";
061        final Parser p1 = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
062        p1.setPreserveLeadingWhitespace(false);
063        p1.setPreserveTrailingWhitespace(true);
064        final StreamingDataSet ds = p1.parseAsStream();
065        ds.next();
066        assertEquals("do preserve trailing space value1...", "value1  ", ds.getRecord().get().getString("column1"));
067        assertEquals("do preserve trailing space value2...", "value2", ds.getRecord().get().getString("column2"));
068        assertEquals("do preserve trailing space value3...", "value3   ", ds.getRecord().get().getString("column3"));
069    }
070
071    public void testDoPreserveBothSpace() {
072        final String cols = "column1,column2,column3\r\n value1  , value2,value3   ";
073        final Parser p1 = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
074        p1.setPreserveLeadingWhitespace(true);
075        p1.setPreserveTrailingWhitespace(true);
076        final StreamingDataSet ds = p1.parseAsStream();
077        ds.next();
078        assertEquals("do preserve both space value1...", " value1  ", ds.getRecord().get().getString("column1"));
079        assertEquals("do preserve both space value2...", " value2", ds.getRecord().get().getString("column2"));
080        assertEquals("do preserve both space value3...", "value3   ", ds.getRecord().get().getString("column3"));
081    }
082
083    public void testContainsWithStream() {
084        final String cols = "column1,column2,column3\r\nvalue1,value2,value3\r\nvalue1a,value2a,value3a";
085        final Parser p = DefaultParserFactory.newCsvParser(new StringReader(cols));
086        final List<Record> collect = p.stream().collect(Collectors.toList());
087        assertEquals("Size", 2, collect.size());
088        assertEquals("column should NOT be found...", false, collect.get(0).contains("shouldnotcontain"));
089        assertEquals("column should be found...", true, collect.get(0).contains("column1"));
090        assertEquals("column should NOT be found...", false, collect.get(0).contains("shouldnotcontain"));
091        assertEquals("column should be found...", true, collect.get(0).contains("column1"));
092        assertEquals("column should be found val1...", "value1", collect.get(0).getString("column1"));
093        assertEquals("column should be found 1a...", "value1a", collect.get(1).getString("column1"));
