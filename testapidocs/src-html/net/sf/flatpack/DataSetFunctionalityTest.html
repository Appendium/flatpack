019    public void testContains() {
020        DataSet ds;
021        final String cols = "column1,column2,column3\r\n value1,value2,value3";
022        final Parser p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
023        ds = p.parse();
024        ds.next();
025        assertEquals("column should NOT be found...", false, ds.contains("shouldnotcontain"));
026
027        assertEquals("column should be found...", true, ds.contains("column1"));
028    }

030    public void testContainsForStream() {
031        final String cols = "column1,column2,column3\r\n value1  ,value2,value3";
032        final Parser p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
033        final StreamingDataSet ds = p.parseAsStream();
034        assertTrue(ds.next());
035        assertEquals("column should NOT be found...", false, ds.getRecord().get().contains("shouldnotcontain"));
036        assertEquals("column should be found...", true, ds.getRecord().get().contains("column1"));
037        assertEquals("column should be found...", " value1", ds.getRecord().get().getString("column1"));
038    }

040    public void testDoNotPreserveSpace() {
041        final String cols = "column1,column2,column3\r\n value1  , value2,value3   ";
042        final Parser p1 = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
043        p1.setPreserveLeadingWhitespace(false);
044        final StreamingDataSet ds = p1.parseAsStream();
045        ds.next();
046        assertEquals("do not preserve leading space value1...", "value1", ds.getRecord().get().getString("column1"));
047        assertEquals("do not preserve leading space value2...", "value2", ds.getRecord().get().getString("column2"));
048        assertEquals("do not preserve leading space value3...", "value3", ds.getRecord().get().getString("column3"));
049    }

159    public void testContainsWithStream() {
160        final String cols = "column1,column2,column3\r\nvalue1,value2,value3\r\nvalue1a,value2a,value3a";
161        final Parser p = DefaultParserFactory.newCsvParser(new StringReader(cols));
162        final List<Record> collect = p.stream().collect(Collectors.toList());
163        assertEquals("Size", 2, collect.size());
164        assertEquals("column should NOT be found...", false, collect.get(0).contains("shouldnotcontain"));
165        assertEquals("column should be found...", true, collect.get(0).contains("column1"));
166        assertEquals("column should NOT be found...", false, collect.get(0).contains("shouldnotcontain"));
167        assertEquals("column should be found...", true, collect.get(0).contains("column1"));
168        assertEquals("column should be found val1...", "value1", collect.get(0).getString("column1"));
169        assertEquals("column should be found 1a...", "value1a", collect.get(1).getString("column1"));
