001package net.sf.flatpack.brparse;
002
003import java.io.StringReader;
004import junit.framework.TestCase;
005import net.sf.flatpack.DataSet;
006import net.sf.flatpack.DefaultDataSet;
007import net.sf.flatpack.DefaultParserFactory;
008import net.sf.flatpack.Parser;
009import net.sf.flatpack.structure.Row;
010
011/**
012 * Test misc methods in the BuffReaderFixedParser class
013 *
014 * @author Tim Zimmerman
015 */
016public class BuffReaderFixedParserTest extends TestCase {
017
018    private static final String PZ_MAP_XML_STRING
019            = "<?xml version=\"1.0\"?>\n"
020            + "<!DOCTYPE PZMAP SYSTEM   \"flatpack.dtd\" >\n"
021            + "<PZMAP>\n"
022            + " <COLUMN name=\"FIELD_ONE\" length=\"10\" />\n"
023            + " <COLUMN name=\"FIELD_TWO\" length=\"10\" />\n"
024            + " <COLUMN name=\"FIELD_THREE\" length=\"10\" />\n"
025            + "</PZMAP> ";
026    private static final String EXACT_LINE_STRING = "exactly   thirty    characters";
027    private static final String SHORT_LINE_STRING = "shorter   than      thirty";
028    private static final String LONG_LINE_STRING = "longer    than      thirty    characters";
029
030    private Parser fixedWidthParser;
031
032    public String parseRawData(String pzMapXML, String dataString) {
033        String rawData = null;
034        StringReader pzReader = new StringReader(pzMapXML);
035        StringReader lineReader = new StringReader(dataString);
036        try {
037            Parser parser = BuffReaderParseFactory.getInstance().newFixedLengthParser(pzReader, lineReader);
038            assertTrue("Parser is not an instance of " + BuffReaderFixedParser.class, parser instanceof BuffReaderFixedParser);
039
040            BuffReaderFixedParser fixedWidthParser = (BuffReaderFixedParser) parser;
041            fixedWidthParser.setIgnoreExtraColumns(true);
042            fixedWidthParser.setHandlingShortLines(true);
043            fixedWidthParser.setStoreRawDataToDataSet(true);
044
045            DefaultDataSet dataset = (DefaultDataSet) fixedWidthParser.parse();
046            Row parsedRow = fixedWidthParser.buildRow(dataset);
047            rawData = parsedRow.getRawData();
048        } finally {
049            pzReader.close();
050            lineReader.close();
051        }
052        return rawData;
053    }
054
055    public void testBuildExactRow() {
056        String rawData = this.parseRawData(PZ_MAP_XML_STRING, EXACT_LINE_STRING);
057        assertTrue("The raw data does not match the orginal line", rawData.equals(EXACT_LINE_STRING));
058    }
059
060    public void testBuildShortRow() {
061        String rawData = this.parseRawData(PZ_MAP_XML_STRING, SHORT_LINE_STRING);
062        assertTrue("The raw data does not match the orginal line", rawData.equals(SHORT_LINE_STRING));
063    }
064
065    public void testBuildLongRow() {
066        String rawData = this.parseRawData(PZ_MAP_XML_STRING, LONG_LINE_STRING);
067        assertTrue("The raw data does not match the orginal line", rawData.equals(LONG_LINE_STRING));
068    }
069
070    public static void main(final String[] args) {
071        junit.textui.TestRunner.run(BuffReaderFixedParserTest.class);
072    }
073}
