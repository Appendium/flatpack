001package net.sf.flatpack.utilities;
002
003/**
004 * Static utility methods for Junit tests
005 *
006 * @author Paul Zepernick
007 */
008public class UnitTestUtils {
009
010    /**
011     * Builds a delimited qualified string containing the elements passed in
012     *
013     * @param elements
014     *            Elements To Include In Delimited String
015     * @param delimiter
016     * @param qualifier
017     * @return String
018     */
019    public static String buildDelimString(final String[] elements, final char delimiter, final char qualifier) {
020        final StringBuilder sb = new StringBuilder();
021        for (int i = 0; i < elements.length; i++) {
022            if (i > 0) {
023                sb.append(delimiter);
024            }
025            if (qualifier != 0) {
026                sb.append(qualifier);
027            }
028            sb.append(elements[i]);
029            if (qualifier != 0) {
030                sb.append(qualifier);
031            }
032        }
033        return sb.toString();
034    }
035}
