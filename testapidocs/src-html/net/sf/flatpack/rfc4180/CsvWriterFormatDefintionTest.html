001package net.sf.flatpack.rfc4180;
002
003import java.io.IOException;
004import java.io.StringWriter;
005
006import net.sf.flatpack.writer.DelimiterWriterFactory;
007import net.sf.flatpack.writer.Rfc4180TestCase;
008import net.sf.flatpack.writer.Writer;
009import net.sf.flatpack.writer.WriterOptions;
010
011import static org.hamcrest.CoreMatchers.is;
012import static org.hamcrest.CoreMatchers.not;
013import static org.hamcrest.CoreMatchers.notNullValue;
014import static org.junit.Assert.assertThat;
015
016public class CsvWriterFormatDefintionTest extends Rfc4180TestCase {
017        private final char FIELD_DELIMITER = ',';
018        private final char FIELD_QUALIFIER = '"';
019        private final String LINE_SEPARATOR = "\r\n";
020
021
022        /*
023         * 2.1 Each record is located on a separate line, delimited by a line break (CRLF).  For example:
024     *
025     * aaa,bbb,ccc CRLF
026     * zzz,yyy,xxx CRLF
027         */
028        public void testLineSeparation() throws IOException {
029                final StringWriter out = new StringWriter();
030
031                try (Writer writer = getWriterForRfc4180(out, false)) {
032                        // write one line of data ... not in the correct order of fields
033                        writer.addRecordEntry("col1", "aaa")
034                                        .addRecordEntry("col2", "bbb")
035                                        .addRecordEntry("col3", "ccc")
036                                        .nextRecord();
037                        writer.addRecordEntry("col3", "xxx")
038                                        .addRecordEntry("col2", "yyy")
039                                        .addRecordEntry("col1", "zzz")
040                                        .nextRecord();
041
042                        writer.flush();
043                }
044
045                String result = out.toString();
046                assertThat(result, is(notNullValue()));
047                assertThat(result.length(), is(26));
048
049                String[] lines = result.split(LINE_SEPARATOR);
050                assertThat(lines, is(notNullValue()));
051                assertThat(lines.length, is(2));
052                assertThat(lines[0], is("aaa,bbb,ccc"));
053                assertThat(lines[1], is("zzz,yyy,xxx"));
054        }
055
056        /*
057         * 2.4a. Each line should contain the same number of fields throughout the file
058         */
059        public void testSameNumberOffFields() throws IOException {
060                final StringWriter out = new StringWriter();
061
062                try (Writer writer = getWriterForRfc4180(out, false)) {
063                        // write one line of data ... not in the correct order of fields
064                        writer.addRecordEntry("col1", "aaa")
065                                        .addRecordEntry("col2", "bbb")
066                                        .addRecordEntry("col3", "ccc")
067                                        .nextRecord();
068                        writer.addRecordEntry("col3", "xxx")
069                                        //.addRecordEntry("col2", "yyy")
070                                        .addRecordEntry("col1", "zzz")
071                                        .nextRecord();
072
073                        writer.flush();
074                }
075
076                String result = out.toString();
077                assertThat(result, is(notNullValue()));
078                assertThat(result.length(), is(23));
079
080                String[] lines = result.split(LINE_SEPARATOR);
081                assertThat(lines, is(notNullValue()));
082                assertThat(lines.length, is(2));
083                assertThat(lines[0], is("aaa,bbb,ccc"));
084                assertThat(lines[0].split(String.valueOf(FIELD_DELIMITER)).length, is(3));
085                assertThat(lines[1], is("zzz,,xxx"));
086                assertThat(lines[1].split(String.valueOf(FIELD_DELIMITER)).length, is(3));
087        }
088
089        /*
090         * 2.4b. Spaces are considered part of a field and should not be ignored.
091         */
092        public void testSpacesArePartOfField() throws IOException {
093                final StringWriter out = new StringWriter();
094
095                try (Writer writer = getWriterForRfc4180(out, false)) {
096                        // write one line of data ... not in the correct order of fields
097                        writer.addRecordEntry("col1", "aaa ")
098                                        .addRecordEntry("col2", " bbb")
099                                        .addRecordEntry("col3", "c cc")
100                                        .nextRecord();
101                        writer.addRecordEntry("col3", "x x x")
102                                        .addRecordEntry("col2", "yy y")
103                                        .addRecordEntry("col1", " zzz ")
104                                        .nextRecord();
105
106                        writer.flush();
107                }
108
109                String result = out.toString();
110                assertThat(result, is(notNullValue()));
111                assertThat(result.length(), is(34));
112
113                String[] lines = result.split(LINE_SEPARATOR);
114                assertThat(lines, is(notNullValue()));
115                assertThat(lines.length, is(2));
116                assertThat(lines[0], is("aaa , bbb,c cc"));
117                assertThat(lines[1], is(" zzz ,yy y,x x x"));
118        }
119
120        /*
121         * 2.4c. The last field in the record must not be followed by a comma
122         */
123        public void testLastFieldShouldNotHaveDelimiter() throws IOException {
124                final StringWriter out = new StringWriter();
125
126                try (Writer writer = getWriterForRfc4180(out, false)) {
127                        // write one line of data ... not in the correct order of fields
128                        writer.addRecordEntry("col1", "aaa")
129                                        .addRecordEntry("col2", "bbb")
130                                        .addRecordEntry("col3", "ccc")
131                                        .nextRecord();
132                        writer.addRecordEntry("col3", "xxx")
133                                        .addRecordEntry("col2", "yyy")
134                                        .addRecordEntry("col1", "zzz")
135                                        .nextRecord();
136
137                        writer.flush();
138                }
139
140                String result = out.toString();
141                assertThat(result, is(notNullValue()));
142                assertThat(result.length(), is(26));
143
144                String[] lines = result.split(LINE_SEPARATOR);
145                assertThat(lines, is(notNullValue()));
146                assertThat(lines.length, is(2));
147                assertThat(lines[0].length(), is(11));
148                assertThat(lines[0].charAt(lines[0].length() -1), is(not(FIELD_DELIMITER)));
149                assertThat(lines[1].length(), is(11));
150                assertThat(lines[1].charAt(lines[0].length() -1), is(not(FIELD_DELIMITER)));
151        }
152
153
154        private Writer getWriterForRfc4180(java.io.Writer out, boolean autoPrintHeader) throws IOException {
155                WriterOptions options = WriterOptions.getInstance();
156                options.autoPrintHeader(autoPrintHeader);
157                options.setLineSeparator(LINE_SEPARATOR);
158
159                final DelimiterWriterFactory factory = new DelimiterWriterFactory(FIELD_DELIMITER, FIELD_QUALIFIER)//
160                                .addColumnTitle("col1")
161                                .addColumnTitle("col2")
162                                .addColumnTitle("col3");
163
164                return factory.createWriter(out, options);
165        }
166}
