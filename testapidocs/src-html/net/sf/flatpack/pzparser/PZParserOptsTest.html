097    public void testCaseSensitiveMetaData() {
098        DataSet ds;
099        final String cols = "COLUMN1,column2,Column3\r\n value1,value2,value3";
100        Parser p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
101
102        // check that column names are case sensitive
103        p.setColumnNamesCaseSensitive(true);
104        ds = p.parse();
105        ds.next();
106        try {
107            ds.getString("COLUMN2");
108            fail("Column was mapped as 'column2' and lookup was 'COLUMN2'...should fail with case sensitivity turned on");
109        } catch (final NoSuchElementException e) {
110            // this should happen since we are matching case
111        }
112
113        // check that column names are NOT case sensitive
114        p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
115        p.setColumnNamesCaseSensitive(false);
116        ds = p.parse();
117        ds.next();
118        try {
119            ds.getString("COLUMN2");
120        } catch (final NoSuchElementException e) {
121            fail("Column was mapped as 'column2' and lookup was 'COLUMN2'...should NOT fail with case sensitivity turned OFF");
122        }
123
124        // re-test the buffered reader
125        p = BuffReaderParseFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
126
127        // check that column names are case sensitive
128        p.setColumnNamesCaseSensitive(true);
129        ds = p.parse();
130        ds.next();
131        try {
132            ds.getString("COLUMN2");
133            fail("Column was mapped as 'column2' and lookup was 'COLUMN2'...should fail with case sensitivity turned on");
134        } catch (final NoSuchElementException e) {
135            // this should happen since we are matching case
136        }
137
138        // check that column names are NOT case sensitive
139        p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
140        p.setColumnNamesCaseSensitive(false);
141        ds = p.parse();
142        ds.next();
143        try {
144            ds.getString("COLUMN2");
145        } catch (final NoSuchElementException e) {
146            fail("Column was mapped as 'column2' and lookup was 'COLUMN2'...should NOT fail with case sensitivity turned OFF");
147        }

173    public void testStoreRawDataToDataError() {
174        DataSet ds;
175        final String cols = "column1,column2,column3\r\nVAL1,VAL2,VAL3,VAL4";
176        Parser p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
177        p.setStoreRawDataToDataError(true);
178        ds = p.parse();
179        Iterator errors = ds.getErrors().iterator();
180        DataError de = (DataError) errors.next();
181
182        assertNotNull("DataError should contain line data...", de.getRawData());
183
184        p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
185        p.setStoreRawDataToDataError(false);
186        ds = p.parse();
187        errors = ds.getErrors().iterator();
188        de = (DataError) errors.next();
189        assertNull("DataError should have <null> line data...", de.getRawData());
190    }

243    public void testFixedWidthMultipleRecordElementsInMapping() {
244        DataSet ds;
245        final String xml = "<?xml version=\"1.0\" encoding=\"ISO-8859-1\"?> \r\n" + "<!DOCTYPE PZMAP SYSTEM     \"pzfilereader.dtd\" > \r\n"
246                + "     <PZMAP>\r\n" + "                <RECORD id=\"header\" startPosition=\"1\" endPosition=\"1\" indicator=\"H\">"
247                + "                     <COLUMN name=\"recordtype\" length=\"1\" /> \r\n" + "                   <COLUMN name=\"headerdata1\" length=\"20\" /> \r\n" + "         </RECORD>"
248                + "             <COLUMN name=\"recordtype\" length=\"1\" /> \r\n" + "           <COLUMN name=\"detaildata1\" length=\"20\" /> \r\n" + " </PZMAP>";
249
250        final String cols = "HHEADER DATA         \r\n" + "DDETAIL DATA         \r\n";
251
252        final Parser p = DefaultParserFactory.getInstance().newFixedLengthParser(new StringReader(xml), new StringReader(cols));
253        ds = p.parse();
254        int i = 0;
255        while (ds.next()) {
256            if (i == 0) {
257                assertEquals("Checking to see if we have the header record...", ds.isRecordID("header"), true);
258                assertEquals("Checking header data", ds.getString("headerdata1"), "HEADER DATA");
259            } else {
260                assertEquals("Checking detail data", ds.getString("detaildata1"), "DETAIL DATA");
261            }
262
263            i++;
264        }
265
266    }

268    public void testSorting() {
269        DataSet ds;
270        String cols = "fname,lname,dob,anumber\r\npaul,zepernick,06/21/1981,2\r\nbenoit,xhenseval,05/01/1970,12";
271        Parser p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
272        ds = p.parse();
273
274        OrderBy order = new OrderBy();
275        order.addOrderColumn(new OrderColumn("fname", OrderColumn.ASC));
276        ds.orderRows(order);
277        ds.next();
278        assertEquals("benoit", ds.getString("fname"));
279
280        order = new OrderBy();
281        order.addOrderColumn(new OrderColumn("lname", OrderColumn.DESC));
282        ds.orderRows(order);
283        ds.next();
284        assertEquals("zepernick", ds.getString("lname"));
285
286        // test date sorting
287        order = new OrderBy();
288        OrderColumn column = new OrderColumn("dob", OrderColumn.ASC, OrderColumn.COLTYPE_DATE);
289        column.setDateFormatPattern("MM/dd/yyyy");
290        order.addOrderColumn(column);
291        ds.orderRows(order);
292        ds.next();
293        assertEquals("xhenseval", ds.getString("lname"));
294
295        // test numeric sorting
296        order = new OrderBy();
297        order.addOrderColumn(new OrderColumn("anumber", OrderColumn.DESC, OrderColumn.COLTYPE_NUMERIC));
298        ds.orderRows(order);
299        ds.next();
300        assertEquals("xhenseval", ds.getString("lname"));
301
302        // test bad date format & bad numeric data
303        // 06.21.1981 should default to 01/01/1900 since it does not match our date format
304        cols = "fname,lname,dob,anumber\r\npaul,zepernick,06.21.1981,not a number\r\nbenoit,xhenseval,05/01/1970,12";
305        p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
306        ds = p.parse();
307        order = new OrderBy();
308        column = new OrderColumn("dob", OrderColumn.ASC, OrderColumn.COLTYPE_DATE);
309        column.setDateFormatPattern("MM/dd/yyyy");
310        order.addOrderColumn(column);
311        ds.orderRows(order);
312        ds.next();
313        assertEquals("zepernick", ds.getString("lname"));
314
315        // not a number should get treated as a 0
316        order = new OrderBy();
317        order.addOrderColumn(new OrderColumn("anumber", OrderColumn.ASC, OrderColumn.COLTYPE_NUMERIC));
318        ds.orderRows(order);
319        ds.next();
320        assertEquals("zepernick", ds.getString("lname"));
321
322    }
