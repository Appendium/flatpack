001package net.sf.flatpack;
002
003import java.io.StringReader;
004import java.math.BigDecimal;
005import java.util.List;
006import java.util.stream.Collectors;
007
008import junit.framework.TestCase;
009
010/**
011 * JDK 8 Streaming test.
012 * @author Benoit Xhenseval
013 */
014public class StreamingTest extends TestCase {
015
016    private static class Test {
017        private String itemName;
018        private BigDecimal price;
019
020        public String getItemName() {
021            return itemName;
022        }
023
024        public void setItemName(final String itemName) {
025            this.itemName = itemName;
026        }
027
028        public BigDecimal getPrice() {
029            return price;
030        }
031
032        public void setPrice(final BigDecimal price) {
033            this.price = price;
034        }
035    }
036
037    public void testContains() {
038        final String cols = "item,price,purchaseDate\r\n"//
039                + "MacBook,1890.20,20140523\r\n"//
040                + "Surface3,850.00,20140524\r\n"//
041                ;
042        final Parser p = DefaultParserFactory.newCsvParser(new StringReader(cols));
043        final List<Test> ds = p.stream() //
044                .map(t -> {
045                    final Test r = new Test();
046                    r.setItemName(t.getString("item"));
047                    r.setPrice(t.getBigDecimal("price"));
048                    return r;
049                })// Mapping from Record to Test
050                .filter(t -> "Surface3".equals(t.getItemName())) // only keep the Surface3 (why???)
051                .collect(Collectors.toList());
052
053        // test record 1 with Data in file!
054        assertEquals("Size", 1, ds.size());
055        final Test test = ds.get(0);
056        assertEquals("Item", "Surface3", test.getItemName());
057        assertTrue("Price", new BigDecimal("850").compareTo(test.getPrice()) == 0);
058    }
059}
