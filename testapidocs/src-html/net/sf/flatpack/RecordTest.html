001package net.sf.flatpack;
002
003import java.io.StringReader;
004import java.math.BigDecimal;
005import java.text.ParseException;
006import java.util.Calendar;
007import java.util.Date;
008import java.util.Optional;
009
010import junit.framework.TestCase;
011import net.sf.flatpack.util.FPConstants;
012
013/**
014 * Test methods in the DataSet
015 *
016 * @author Paul Zepernick
017 */
018public class RecordTest extends TestCase {
019
020    public void testContains() {
021        final String cols = "stringCol,doubleCol,dateCol,bigDecimalCol,intCol\r\n"//
022                + "hello,2.20,20140523,123.45,6\r\n"//
023                + ",,,,"//
024                ;
025        final Parser p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
026        final StreamingDataSet ds = p.parseAsStream();
027        ds.next();
028        final Optional<Record> record1 = ds.getRecord();
029
030        // test record 1 with Data in file!
031        assertEquals("rec1 string", "hello", record1.get().getString("stringCol"));
032        assertTrue("rec1 doubleCol", Double.compare(2.2, record1.get().getDouble("doubleCol")) == 0);
033        try {
034            assertEquals("rec1 dateCol", new Date(114, Calendar.MAY, 23), record1.get().getDate("dateCol"));
035        } catch (final ParseException e) {
036            fail();
037        }
038        assertEquals("rec1 intCol", 6, record1.get().getInt("intCol"));
039        assertEquals("rec1 bigDecimalCol", new BigDecimal("123.45"), record1.get().getBigDecimal("bigDecimalCol"));
040
041        // NOW RECORD 2 with ALL defaults
042        ds.next();
043        final Optional<Record> record2 = ds.getRecord();
044        assertEquals("rec2 string", "Hi", record2.get().getString("stringCol", () -> "Hi"));
045        assertTrue("rec2 doubleCol", Double.compare(3.76, record2.get().getDouble("doubleCol", () -> 3.76d)) == 0);
046        try {
047            assertEquals("rec2 dateCol", new Date(114, Calendar.JUNE, 11), record2.get().getDate("dateCol", () -> new Date(114, Calendar.JUNE, 11)));
048        } catch (final ParseException e) {
049            fail();
050        }
051        assertEquals("rec2 intCol", 8, record2.get().getInt("intCol", () -> 8));
052        assertEquals("rec2 bigDecimalCol", new BigDecimal("555"), record2.get().getBigDecimal("bigDecimalCol", () -> new BigDecimal("555")));
053
054    }
055
056}
