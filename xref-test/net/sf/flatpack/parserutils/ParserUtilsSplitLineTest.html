34      /**
35       * Test without any line breaks
36       *
37       */
38      public void testNoLineBreaks() {
39          // loop down all delimiter qualifier pairs to test
40          for (final char[] element : DELIM_QUAL_PAIR) {
41              final char d = element[0];
42              final char q = element[1];
43  
44              final String txtToParse = UnitTestUtils.buildDelimString(DELIMITED_DATA_NO_BREAKS, d, q);
45  
46              final List splitLineResults = ParserUtils.splitLine(txtToParse, d, q, 10, false, false);
47  
48              // check to make sure we have the same amount of elements which were
49              // expected
50              assertEquals("Expected size (d = [" + d + "] q = [" + (q != 0 ? String.valueOf(q) : "") + "] txt [" + txtToParse + "])",
51                      DELIMITED_DATA_NO_BREAKS.length, splitLineResults.size());
52  
53              // loop through each value and compare what came back
54              for (int j = 0; j < DELIMITED_DATA_NO_BREAKS.length; j++) {
55                  assertEquals("Data Element Value Does Not Match (d = [" + d + "] q = [" + q + "] txt [" + txtToParse + "])",
56                          DELIMITED_DATA_NO_BREAKS[j], (String) splitLineResults.get(j));
57              }
58          }
59      }
