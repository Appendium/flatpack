34      public String parseRawData(String pzMapXML, String dataString) {
35          String rawData = null;
36          StringReader pzReader = new StringReader(pzMapXML);
37          StringReader lineReader = new StringReader(dataString);
38          try {
39              Parser parser = BuffReaderParseFactory.getInstance().newDelimitedParser(pzReader, lineReader, DELIMTER, QUALIFIER, false);
40              assertTrue("Parser is not an instance of " + BuffReaderDelimParser.class, parser instanceof BuffReaderDelimParser);
41  
42              BuffReaderDelimParser delimParser = (BuffReaderDelimParser) parser;
43              delimParser.setIgnoreExtraColumns(true);
44              delimParser.setHandlingShortLines(true);
45              delimParser.setStoreRawDataToDataSet(true);
46  
47              DefaultDataSet dataset = (DefaultDataSet) delimParser.parse();
48              Row parsedRow = delimParser.buildRow(dataset);
49              rawData = parsedRow.getRawData();
50          } finally {
51              pzReader.close();
52              lineReader.close();
53          }
54          return rawData;
55      }
