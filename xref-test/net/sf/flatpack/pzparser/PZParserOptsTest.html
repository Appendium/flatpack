268     public void testSorting() {
269         DataSet ds;
270         String cols = "fname,lname,dob,anumber\r\npaul,zepernick,06/21/1981,2\r\nbenoit,xhenseval,05/01/1970,12";
271         Parser p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
272         ds = p.parse();
273 
274         OrderBy order = new OrderBy();
275         order.addOrderColumn(new OrderColumn("fname", OrderColumn.ASC));
276         ds.orderRows(order);
277         ds.next();
278         assertEquals("benoit", ds.getString("fname"));
279 
280         order = new OrderBy();
281         order.addOrderColumn(new OrderColumn("lname", OrderColumn.DESC));
282         ds.orderRows(order);
283         ds.next();
284         assertEquals("zepernick", ds.getString("lname"));
285 
286         // test date sorting
287         order = new OrderBy();
288         OrderColumn column = new OrderColumn("dob", OrderColumn.ASC, OrderColumn.COLTYPE_DATE);
289         column.setDateFormatPattern("MM/dd/yyyy");
290         order.addOrderColumn(column);
291         ds.orderRows(order);
292         ds.next();
293         assertEquals("xhenseval", ds.getString("lname"));
294 
295         // test numeric sorting
296         order = new OrderBy();
297         order.addOrderColumn(new OrderColumn("anumber", OrderColumn.DESC, OrderColumn.COLTYPE_NUMERIC));
298         ds.orderRows(order);
299         ds.next();
300         assertEquals("xhenseval", ds.getString("lname"));
301 
302         // test bad date format & bad numeric data
303         // 06.21.1981 should default to 01/01/1900 since it does not match our date format
304         cols = "fname,lname,dob,anumber\r\npaul,zepernick,06.21.1981,not a number\r\nbenoit,xhenseval,05/01/1970,12";
305         p = DefaultParserFactory.getInstance().newDelimitedParser(new StringReader(cols), ',', FPConstants.NO_QUALIFIER);
306         ds = p.parse();
307         order = new OrderBy();
308         column = new OrderColumn("dob", OrderColumn.ASC, OrderColumn.COLTYPE_DATE);
309         column.setDateFormatPattern("MM/dd/yyyy");
310         order.addOrderColumn(column);
311         ds.orderRows(order);
312         ds.next();
313         assertEquals("zepernick", ds.getString("lname"));
314 
315         // not a number should get treated as a 0
316         order = new OrderBy();
317         order.addOrderColumn(new OrderColumn("anumber", OrderColumn.ASC, OrderColumn.COLTYPE_NUMERIC));
318         ds.orderRows(order);
319         ds.next();
320         assertEquals("zepernick", ds.getString("lname"));
