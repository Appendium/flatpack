033package net.sf.flatpack;
034
035import java.util.Properties;
036
037import net.sf.flatpack.ordering.OrderBy;
038
039public interface DataSet extends Record, RecordDataSet {
040
041    /**
042     * Goes to the top of the data set. This will put the pointer one record
043     * before the first in the set. Next() will have to be called to get the
044     * first record after this call.
045     */
046    void goTop();
047
048    /**
049     * Goes to the last record in the dataset
050     */
051    void goBottom();
052
053    /**
054     * Moves back to the previous record in the set return true if move was a
055     * success, false if not
056     *
057     * @return boolean
058     */
059    boolean previous();
060
061    /**
062     * Removes a row from the dataset. Once the row is removed the pointer will
063     * be sitting on the record previous to the deleted row.
064     */
065    void remove();
066
067    /**
068     * Returns the index the pointer is on for the array
069     *
070     * @return int
071     */
072    int getIndex();
073
074    /**
075     * Returns the total number of rows parsed in from the file
076     *
077     *
078     * @return int - Row Count
079     */
080    int getRowCount();
081
082    /**
083     * Returns true or false as to whether or not the line number contains an
084     * error. The import will skip the line if it contains an error and it will
085     * not be processed
086     *
087     * @param lineNo -
088     *            int line number
089     * @return boolean
090     */
091    boolean isAnError(int lineNo);
092
093    /**
094     * Orders the data by column(s) specified. This will reposition the cursor
095     * to the top of the DataSet when executed. This is currently not supported
096     * when specifying &lt;RECORD&gt; elements in the mapping. An exception will be
097     * thrown if this situation occurs
098     *
099     * @param ob -
100     *            OrderBy object
101     * @see net.sf.flatpack.ordering.OrderBy
102     * @see net.sf.flatpack.ordering.OrderColumn
103     */
104    void orderRows(OrderBy ob);
105
106    /**
107     * Sets data in the DataSet to lowercase
108     */
109    void setLowerCase();
110
111    /**
112     * Sets data in the DataSet to uppercase
113     */
114    void setUpperCase();
115
116    /**
117     * Sets the absolute position of the record pointer
118     *
119     * @param localPointer -
120     *            int
121     * @exception IndexOutOfBoundsException
122     */
123    void absolute(int localPointer);
124
125    /**
126     * Setting this to True will parse text as is and throw a
127     * NumberFormatException. Setting to false, which is the default, will
128     * remove any non numeric character from the field. The remaining numeric
129     * chars's will be returned. If it is an empty string,or there are no
130     * numeric chars, 0 will be returned for getInt() and getDouble()
131     *
132     * @param strictNumericParse
133     *            The strictNumericParse to set.
134     */
135    void setStrictNumericParse(boolean strictNumericParse);
136
137    /**
138     * Sets the properties from the pzconvert.properties file.
139     * This file specifies the PZConverter implementation to use
140     * for a particular class
141     *
142     * @param props
143     *          Property mapping for String to Object conversion
144     */
145    void setPZConvertProps(Properties props);
146
147    /**
148     * Changes the value of the given column only for the
149     * given row which the pointer is currently sitting on.
150     *
151     * @param column
152     *          Column name to set the value for
153     * @param value
154     *          Value to change the column to
155     */
156    void setValue(String column, String value);
157
158    /**
159     * Clears out the rows in memory from the last parse.
160     *
161     */
162    void clearRows();
163
164    /**
165     * Clears out the parse errors from memory
166     *
167     */
168    void clearErrors();
169
170    /**
171     * Clears both the errors and rows from memory
172     *
173     */
174    void clearAll();
175}
