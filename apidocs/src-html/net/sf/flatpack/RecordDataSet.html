001package net.sf.flatpack;
002
003import java.util.List;
004import java.util.Optional;
005
006/**
007 * Rather than treating a DataSet as a stateful class whereby we need to extract each column one by
008 * one, this interface allows you to extract a record at a time.
009 * @since 3.4
010 */
011public interface RecordDataSet {
012    /**
013     * Returns true if it has one more record. false if not
014     *
015     * @return boolean
016     */
017    boolean next();
018
019    Optional<Record> getRecord();
020
021    /**
022     * Returns A Collection Of DataErrors that happened during processing
023     *
024     * @return Vector
025     */
026    List<DataError> getErrors();
027
028    /**
029     * Returns total number of records which contained a parse error in the
030     * file.
031     *
032     * @return int - Record Error Count
033     */
034    int getErrorCount();
035
036}
