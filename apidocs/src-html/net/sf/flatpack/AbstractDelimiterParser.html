053public abstract class AbstractDelimiterParser extends AbstractParser {
054    private final static Logger LOGGER = LoggerFactory.getLogger(AbstractDelimiterParser.class);
055    private char delimiter = 0;
056    private char qualifier = 0;
057    private boolean ignoreFirstRecord = false;
058
059    private int lineCount = 0;

061    public AbstractDelimiterParser(final Reader dataSourceReader, final String dataDefinition, final char delimiter, final char qualifier,
062            final boolean ignoreFirstRecord) {
063        super(dataSourceReader, dataDefinition);
064        this.delimiter = delimiter;
065        this.qualifier = qualifier;
066        this.ignoreFirstRecord = ignoreFirstRecord;
067    }

069    public AbstractDelimiterParser(final Reader dataSourceReader, final char delimiter, final char qualifier, final boolean ignoreFirstRecord) {
070        super(dataSourceReader);
071        this.delimiter = delimiter;
072        this.qualifier = qualifier;
073        this.ignoreFirstRecord = ignoreFirstRecord;
074    }

076    @Override
077    protected DataSet doParse() {
078        try {
079            lineCount = 0;
080            return doDelimitedFile(getDataSourceReader(), shouldCreateMDFromFile());
081        } catch (final IOException e) {
082            LOGGER.error("error accessing/creating inputstream", e);
083        }
084        return null;
085    }
086
087    protected abstract boolean shouldCreateMDFromFile();
088
089    protected char getDelimiter() {
090        return delimiter;
091    }

094        this.delimiter = delimiter;
095    }

098        return ignoreFirstRecord;
099    }

102        this.ignoreFirstRecord = ignoreFirstRecord;

105    protected char getQualifier() {
106        return qualifier;
107    }
108
109    protected void setQualifier(final char qualifier) {
110        this.qualifier = qualifier;
111    }

114        return lineCount;
115    }

117    /*
118     * This is the new version of doDelimitedFile using InputStrem instead of
119     * File. This is more flexible especially it is working with WebStart.
120     *
121     * puts together the dataset for a DELIMITED file. This is used for PZ XML
122     * mappings, and SQL table mappings
123     */
124    private DataSet doDelimitedFile(final Reader dataSource, final boolean createMDFromFile) throws IOException {
125        if (dataSource == null) {
126            throw new IllegalArgumentException("dataSource is null");
127        }
128        BufferedReader br = null;
129        final DefaultDataSet ds = new DefaultDataSet(getPzMetaData(), this);
130        try {
131            // gather the conversion properties
132            ds.setPZConvertProps(ParserUtils.loadConvertProperties());
133
134            br = new BufferedReader(dataSource);
135
136            boolean processedFirst = false;
137            /** loop through each line in the file */
138            String line = null;
139            int estimatedColCount = FPConstants.SPLITLINE_SIZE_INIT;
140            while ((line = fetchNextRecord(br, getQualifier(), getDelimiter())) != null) {
141                // check to see if the user has elected to skip the first record
142                if (!processedFirst && isIgnoreFirstRecord()) {
143                    processedFirst = true;
144                    continue;
145                } else if (!processedFirst && createMDFromFile) {
146                    processedFirst = true;
147                    setPzMetaData(ParserUtils.getPZMetaDataFromFile(line, delimiter, qualifier, this, isAddSuffixToDuplicateColumnNames()));
148                    ds.setMetaData(getPzMetaData());
149                    continue;
150                }
151                // column values
152
153                // check number of Qualifier, if ODD number --> Incorrect!!!
154                if (oddNumberOfQualifier(line, getQualifier())) {
155                    addError(ds, "Odd number of Qualifier characters", lineCount, 1, isStoreRawDataToDataError() ? line : null);
156                    continue;
157                }
158
159                List<String> columns = ParserUtils.splitLine(line, getDelimiter(), getQualifier(), estimatedColCount, isPreserveLeadingWhitespace(),
160                        isPreserveTrailingWhitespace());
161                final String mdkey = ParserUtils.getCMDKeyForDelimitedFile(getPzMetaData(), columns);
162                final List<ColumnMetaData> metaData = ParserUtils.getColumnMetaData(mdkey, getPzMetaData());
163                final int columnCount = metaData.size();
164                estimatedColCount = columnCount;
165
166                if (columns.size() > columnCount) {
167                    // Incorrect record length on line log the error. Line
168                    // will not be included in the dataset log the error
169                    if (isIgnoreExtraColumns()) {
170                        // user has chosen to ignore the fact that we have too many columns in the data from
171                        // what the mapping has described. sublist the array to remove un-needed columns
172                        columns = columns.subList(0, columnCount);
173                        addError(ds, "Flatpack truncated line to correct number of columns", lineCount, 1, isStoreRawDataToDataError() ? line : null);
174                    } else {
175                        addError(ds, "Too many columns expected: " + columnCount + " Flatpack got: " + columns.size(), lineCount, 2,
176                                isStoreRawDataToDataError() ? line : null);
177                        continue;
178                    }
179                } else if (columns.size() < columnCount) {
180                    if (isHandlingShortLines()) {
181                        // We can pad this line out
182                        while (columns.size() < columnCount) {
183                            columns.add("");
184                        }
185
186                        // log a warning
187                        addError(ds, "Flatpack padded line to correct number of columns", lineCount, 1, isStoreRawDataToDataError() ? line : null);
188
189                    } else {
190                        addError(ds, "Too few columns expected: " + columnCount + " only got: " + columns.size(), lineCount, 2,
191                                isStoreRawDataToDataError() ? line : null);
192                        continue;
193                    }
194                }
195
196                final Row row = new Row();
197                row.setMdkey(mdkey.equals(FPConstants.DETAIL_ID) ? null : mdkey); // try
198                // to limit the memory use
199                row.setCols(columns);
200                row.setRowNumber(lineCount);
201                if (isFlagEmptyRows()) {
202                    // user has elected to have the parser flag rows that are empty
203                    row.setEmpty(ParserUtils.isListElementsEmpty(columns));
204                }
205                if (isStoreRawDataToDataSet()) {
206                    // user told the parser to keep a copy of the raw data in the row
207                    // WARNING potential for high memory usage here
208                    row.setRawData(line);
209                }
210
211                // add the row to the array
212                ds.addRow(row);
213
214            }
215        } finally {
216            if (br != null) {
217                br.close();
218            }
219            closeReaders();
220        }
221        return ds;
222    }
223
224    private boolean oddNumberOfQualifier(String line, char q) {
225        if (line == null || line.isEmpty()) {
226            return false;
227        }
228        int count = 0;
229        int idx = 0;
230        while ((idx = line.indexOf(q, idx)) != -1) {
231            count++;
232            idx++;
233        }
234
235        return count % 2 != 0;
236    }

238    /**
239     * Reads a record from a delimited file.  This will account for records which
240     * could span multiple lines.
241     * NULL will be returned when the end of the file is reached
242     *
243     * @param br
244     *          Open reader being used to read through the file
245     * @param qual
246     *          Qualifier being used for parse
247     * @param delim
248     *          Delimiter being used for parse
249     * @return String
250     *          Record from delimited file
251     *
252     */
253    protected String fetchNextRecord(final BufferedReader br, final char qual, final char delim) throws IOException {
254        String line = null;
255        final StringBuilder lineData = new StringBuilder();
256        final String linebreak = System.getProperty("line.separator");
257        boolean processingMultiLine = false;
258
259        while ((line = br.readLine()) != null) {
260            lineCount++;
261            final String trimmed = line.trim();
262            final int trimmedLen = trimmed.length();
263            if (!processingMultiLine && trimmed.length() == 0) {
264                // empty line skip past it, as long as it
265                // is not part of the multiline
266                continue;
267            }
268
269            // ********************************************************
270            // new functionality as of 2.1.0 check to see if we have
271            // any line breaks in the middle of the record, this will only
272            // be checked if we have specified a delimiter
273            // ********************************************************
274            final char[] chrArry = trimmed.toCharArray();
275            if (!processingMultiLine && delim > 0 && qual != FPConstants.NO_QUALIFIER) {
276                processingMultiLine = ParserUtils.isMultiLine(chrArry, delim, qual);
277            }
278
279            // check to see if we have reached the end of the linebreak in
280            // the record
281
282            final String trimmedLineData = lineData.toString().trim();
283            if (processingMultiLine && trimmedLineData.length() > 0 && trimmedLen > 0) {
284                // need to do one last check here. it is possible that the "
285                // could be part of the data
286                // excel will escape these with another quote; here is some
287                // data "" This would indicate
288                // there is more to the multiline
289                if (trimmed.charAt(trimmed.length() - 1) == qual && !trimmed.endsWith("" + qual + qual)) {
290                    // it is safe to assume we have reached the end of the
291                    // line break
292                    processingMultiLine = false;
293                    lineData.append(linebreak).append(line);
294                } else {
295                    // check to see if this is the last line of the record
296                    // looking for a qualifier followed by a delimiter
297                    lineData.append(linebreak).append(line);
298                    boolean qualiFound = false;
299                    for (final char element : chrArry) {
300                        if (qualiFound) {
301                            if (element == ' ') {
302                                continue;
303                            } else if (element == delim) {
304                                processingMultiLine = ParserUtils.isMultiLine(chrArry, delim, qual);
305                                break;
306                            }
307                            qualiFound = false;
308                        } else if (element == qual) {
309                            qualiFound = true;
310                        }
311                    }
312
313                    // check to see if we are still in multi line mode, if
314                    // so grab the next line
315                    if (processingMultiLine) {
316                        continue;
317                    }
318                }
319            } else {
320                // throw the line into lineData var.
321                // need to check to see if we need to insert a line break.
322                // The buffered reader excludes the breaks
323                lineData.append(trimmedLen == 0 ? linebreak : line);
324                if (processingMultiLine) {
325                    continue; // if we are working on a multiline rec, get
326                    // the data on the next line
327                }
328            }
329
330            break;
331        }
332
333        if (line == null && lineData.length() == 0) {
334            // eof
335            return null;
336        }
337
338        return lineData.toString();
339
340    }
