100    /**
101     * New method based on Reader. Reads the XMLDocument for a PZMetaData
102     * file from an InputStream, WebStart compatible. Parses the XML file, and
103     * returns a Map containing Lists of ColumnMetaData.
104     *
105     * @param xmlStreamReader
106     * @param pzparser
107     *          Can be null.  Allows additional opts to be set during the XML map read
108     * @return Map &lt;records&gt; with their corresponding
109     * @throws IOException
110     * @throws ParserConfigurationException 
111     * @throws SAXException 
112     */
113    public static Map<String, Object> parse(final Reader xmlStreamReader, final Parser pzparser)
114            throws IOException, ParserConfigurationException, SAXException {
115        if (xmlStreamReader == null) {
116            throw new NullPointerException("XML Reader Is Not Allowed To Be Null...");
117        }
118        final Map<String, Object> mdIndex = new LinkedHashMap<String, Object>(); // retain the same order
119
120        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
121        DocumentBuilder builder = factory.newDocumentBuilder();
122        builder.setEntityResolver(new ResolveLocalDTD());
123        org.w3c.dom.Document document = builder.parse(new InputSource(xmlStreamReader));
124        NodeList nodeList = document.getDocumentElement().getChildNodes();
125
126        final List<ColumnMetaData> columnResults = parseColumnElements(nodeList);
127
128        mdIndex.put(FPConstants.DETAIL_ID, columnResults); // always force detail
129        // to the top of
130        // the map no matter what
131        mdIndex.put(FPConstants.COL_IDX, ParserUtils.buidColumnIndexMap(columnResults, pzparser));
132
133        for (int i = 0; i < nodeList.getLength(); i++) {
134            Node node = nodeList.item(i);
135
136            final String nodeName = node.getNodeName();
137
138            if ("RECORD".equalsIgnoreCase(nodeName)) {
139                final NamedNodeMap attributes = node.getAttributes();
140                final Node id = attributes.getNamedItem("id");
141                if (id != null && FPConstants.DETAIL_ID.equalsIgnoreCase(id.getTextContent())) {
142                    // make sure the id attribute does not have a value of "detail" this
143                    // is the harcoded
144                    // value we are using to mark columns specified outside of a
145                    // <RECORD> element
146                    throw new IllegalArgumentException("The ID 'detail' on the <RECORD> element is reserved, please select another id");
147                }
148
149                final List<ColumnMetaData> columns = parseColumnElements(node.getChildNodes());
150                final XMLRecordElement xmlre = new XMLRecordElement();
151                xmlre.setColumns(columns, pzparser);
152                xmlre.setIndicator(getAttributeValue(attributes, "indicator"));
153                xmlre.setElementNumber(convertAttributeToInt(getAttributeValue(attributes, "elementNumber")));
154                xmlre.setStartPosition(convertAttributeToInt(getAttributeValue(attributes, "startPosition")));
155                xmlre.setEndPositition(convertAttributeToInt(getAttributeValue(attributes, "endPosition")));
156                xmlre.setElementCount(convertAttributeToInt(getAttributeValue(attributes, "elementCount")));
157                mdIndex.put(getAttributeValue(attributes, "id"), xmlre);
158                // make a column index for non detail records
159                mdIndex.put(FPConstants.COL_IDX + "_" + getAttributeValue(attributes, "id"), ParserUtils.buidColumnIndexMap(columns, pzparser));
160
161            }
162        }
163        if (showDebug) {
164            setShowDebug(mdIndex);
165        }
166        return mdIndex;
167    }
168
169    private static String getAttributeValue(NamedNodeMap attributes, String attributeName) {
170        final Node namedItem = attributes.getNamedItem(attributeName);
171        return namedItem != null ? namedItem.getTextContent() : null;
172    }
173
174    private static List<ColumnMetaData> parseColumnElements(NodeList nodeList) {
175        final Set<String> columnNames = new HashSet<>();
176        final List<ColumnMetaData> columnResults = new ArrayList<>();
177        for (int i = 0; i < nodeList.getLength(); i++) {
178            Node node = nodeList.item(i);
179
180            final String nodeName = node.getNodeName();
181
182            switch (nodeName) {
183            case "COLUMN":
184                final NamedNodeMap attributes = node.getAttributes();
185                System.out.println("COL:" + attributes.getNamedItem("name"));
186                final ColumnMetaData cmd = new ColumnMetaData();
187
188                // make sure the name attribute is present on the column
189                final String columnName = attributes.getNamedItem("name").getTextContent();
190                if (columnName == null) {
191                    throw new IllegalArgumentException("Name attribute is required on the column tag!");
192                }
193
194                // make sure the names in columnInfo are unique
195                if (columnNames.contains(columnName)) {
196                    throw new IllegalArgumentException("Duplicate name column '" + columnName + "'");
197                }
198
199                cmd.setColName(columnName);
200                columnNames.add(columnName);
201
202                // check to see if the column length can be set
203                if (attributes.getNamedItem("length") != null) {
204                    try {
205                        cmd.setColLength(Integer.parseInt(attributes.getNamedItem("length").getTextContent()));
206                    } catch (final Exception ex) {
207                        throw new IllegalArgumentException(
208                                "LENGTH ATTRIBUTE ON COLUMN ELEMENT MUST BE AN INTEGER.  GOT: " + attributes.getNamedItem("length").getTextContent(),
209                                ex);
210                    }
211                }
212                columnResults.add(cmd);
213                break;
214            }
215        }
216        return columnResults;
217    }

219    // helper to convert to integer
220    private static int convertAttributeToInt(final String attribute) {
221        if (attribute == null) {
222            return 0;
223        }
224
225        try {
226            return Integer.parseInt(attribute);
227        } catch (final Exception ignore) {
228            return 0;
229        }
230    }

242    private static void setShowDebug(final Map<String, Object> xmlResults) {
243        for (final Entry<String, Object> entry : xmlResults.entrySet()) {
244            XMLRecordElement xmlrecEle = null;
245            final String recordID = entry.getKey();
246            List<ColumnMetaData> columns = null;
247            if (recordID.equals(FPConstants.DETAIL_ID)) {
248                columns = (List<ColumnMetaData>) entry.getValue();
249            } else {
250                xmlrecEle = (XMLRecordElement) entry.getValue();
251                columns = xmlrecEle.getColumns();
252            }
253
254            LOGGER.debug(">>>>Column MD Id: " + recordID);
255            if (xmlrecEle != null) {
256                LOGGER.debug("Start Position: " + xmlrecEle.getStartPosition() + " " + "End Position: " + xmlrecEle.getEndPositition() + " "
257                        + "Element Number: " + xmlrecEle.getElementNumber() + " " + "Indicator: " + xmlrecEle.getIndicator());
258            }
259            for (final ColumnMetaData cmd : columns) {
260                LOGGER.debug("Column Name: " + cmd.getColName() + " LENGTH: " + cmd.getColLength());
261
262            }
263        }
264    }

266    /**
267     * New method based on Reader. Reads the XMLDocument for a PZMetaData
268     * file from an InputStream, WebStart compatible. Parses the XML file, and
269     * returns a Map containing Lists of ColumnMetaData.
270     *
271     * @param xmlStreamReader
272     * @param pzparser
273     *          Can be null.  Allows additional opts to be set during the XML map read
274     * @return Map &lt;records&gt; with their corresponding
275     * @throws IOException
276     * @throws SAXException 
277     * @throws ParserConfigurationException 
278     */
279    public static MetaData parseMap(final Reader xmlStreamReader, final Parser pzparser)
280            throws IOException, ParserConfigurationException, SAXException {
281        final Map map = parse(xmlStreamReader, pzparser);
282
283        final List<ColumnMetaData> col = (List<ColumnMetaData>) map.get(FPConstants.DETAIL_ID);
284        map.remove(FPConstants.DETAIL_ID);
285
286        final Map m = (Map) map.get(FPConstants.COL_IDX);
287        map.remove(FPConstants.COL_IDX);
288
289        // loop through the map and remove anything else that is an index of FPConstancts.COL_IDX + _
290        // these were put in for the writer.
291        // TODO maybe these shoudld be thrown into the MetaData instead of just discarded, but they are unused
292        // in the Reader the moment. This parseMap is not utilized in the writer so it is safe to remove them here
293        final Iterator entrySetIt = map.entrySet().iterator();
294        while (entrySetIt.hasNext()) {
295            final Entry e = (Entry) entrySetIt.next();
296            if (((String) e.getKey()).startsWith(FPConstants.COL_IDX + "_")) {
297                entrySetIt.remove();
298            }
299        }
300
301        return new MetaData(col, m, map);
302    }
