030public class CSVLarge {
031    private static final Logger LOG = LoggerFactory.getLogger(CSVLarge.class);
032
033    public static void main(final String[] args) {
034        try {
035            final Map settings = readSettings();
036            final String data = (String) settings.get("csvFile");
037            call(data);
038        } catch (final Exception e) {
039            LOG.error("issue", e);
040        }
041    }
042
043    public static String getDefaultDataFile() {
044        return "LargeSampleCSV.csv";
045    }
046
047    public static void call(final String data) throws Exception {
048        try (BuffReaderDelimParser pzparse = (BuffReaderDelimParser) BuffReaderParseFactory.getInstance().newDelimitedParser(new File(data), ',',
049                '"')) {
050
051            // delimited by a comma
052            // text qualified by double quotes
053            // ignore first record
054
055            final DataSet ds = pzparse.parse();
056            final long timeStarted = System.currentTimeMillis();
057            int totalCount = 0;
058            int tmpCount = 0;
059            while (ds.next()) {
060                totalCount++;
061                tmpCount++;
062                if (tmpCount >= 2500) {
063                    System.out.println("Read " + totalCount + " Records...");
064                    tmpCount = 0;
065                }
066            }
067            final long timeFinished = System.currentTimeMillis();
068
069            String timeMessage = "";
070
071            if (timeFinished - timeStarted < 1000) {
072                timeMessage = timeFinished - timeStarted + " Milleseconds...";
073            } else {
074                timeMessage = (timeFinished - timeStarted) / 1000 + " Seconds...";
075            }
076
077            System.out.println("");
078            System.out.println("********FILE PARSED IN: " + timeMessage + " ******");
079
080            if (ds.getErrors() != null && !ds.getErrors().isEmpty()) {
081                System.out.println("FOUND ERRORS IN FILE....");
082                for (int i = 0; i < ds.getErrors().size(); i++) {
083                    final DataError de = (DataError) ds.getErrors().get(i);
084                    System.out.println("Error: " + de.getErrorDesc() + " Line: " + de.getLineNo());
085                }
086            }
087        } catch (final Exception ex) {
088            LOG.error("Issue", ex);
089        }
090    }
091
092    private static Map readSettings() throws Exception {
093        final Map result = new HashMap();
094
095        try (FileReader fr = new FileReader("settings.properties"); BufferedReader br = new BufferedReader(fr)) {
096
097            String line = null;
098            while ((line = br.readLine()) != null) {
099                if (line.trim().length() == 0 || line.startsWith("#") || line.indexOf("=") == -1) {
100                    continue;
101                }
102
103                result.put(line.substring(0, line.indexOf("=")), line.substring(line.indexOf("=") + 1));
104            }
105        }
106
107        return result;
108    }
