008import java.io.File;
009import java.io.FileInputStream;
010
011import net.sf.flatpack.DataSet;
012import net.sf.flatpack.brparse.BuffReaderDelimParser;
013import net.sf.flatpack.brparse.BuffReaderParseFactory;
014
015/**
016 * @author zepernick
017 *
018 * TODO To change the template for this generated type comment go to Window -
019 * Preferences - Java - Code Style - Code Templates
020 */
021public class LargeDelimitedWithPZMap {
022    public static void main(final String[] args) throws Exception {
023        final String mapping = getDefaultMapping();
024        final String data = getDefaultDataFile();
025        call(mapping, data);
026
027    }
028
029    public static String getDefaultDataFile() {
030        return "PEOPLE-CommaDelimitedWithQualifier.txt";
031    }
032
033    public static String getDefaultMapping() {
034        return "PEOPLE-Delimited.pzmap.xml";
035    }
036
037    public static void call(final String mapping, final String data) throws Exception {
038        String[] colNames = null;
039        try (FileInputStream pzmap = new FileInputStream(new File(mapping));
040                FileInputStream fileToParse = new FileInputStream(new File(data));
041                BuffReaderDelimParser pzparse = (BuffReaderDelimParser) BuffReaderParseFactory.getInstance().newDelimitedParser(pzmap, fileToParse,
042                        ',', '"', true)) {
043            // delimited by a comma
044            // text qualified by double quotes
045            // ignore first record
046
047            final DataSet ds = pzparse.parse();
048
049            colNames = ds.getColumns();
050
051            while (ds.next()) {
052                for (final String colName : colNames) {
053                    System.out.println("COLUMN NAME: " + colName + " VALUE: " + ds.getString(colName));
054                }
055
056                System.out.println("===========================================================================");
057            }
058
059            if (ds.getErrors() != null && !ds.getErrors().isEmpty()) {
060                System.out.println("FOUND ERRORS IN FILE");
061            }
062        }
063    }
064}
