001package net.sf.flatpack.examples.createsamplecsv;
002
003import java.io.FileWriter;
004import java.io.PrintWriter;
005
006import org.slf4j.Logger;
007import org.slf4j.LoggerFactory;
008
009/*
010 * Created on Nov 30, 2005
011 *
012 */
013
014/**
015 * @author zepernick
016 *
017 * Creates a sample csv file with the specified number of columns and rows
018 */
019public class CSVTestFileCreator {
020    private static final Logger LOG = LoggerFactory.getLogger(CSVTestFileCreator.class);
021
022    public static void main(final String[] args) {
023        int cols = 0;
024        int rows = 0;
025
026        if (args.length != 2) {
027            printUsage();
028            return;
029        }
030
031        try {
032            cols = Integer.parseInt(args[0]);
033            rows = Integer.parseInt(args[1]);
034        } catch (final Exception ex) {
035            printUsage();
036            return;
037        }
038
039        createFile(cols, rows);
040    }
041
042    public static void createFile(final int cols, final int rows) {
043        createFile(cols, rows, "SampleCSV.csv");
044    }
045
046    public static void createFile(final int cols, final int rows, final String filename) {
047        FileWriter fw = null;
048        PrintWriter out = null;
049        try {
050
051            fw = new FileWriter(filename);
052            out = new PrintWriter(fw);
053
054            // write the column names across the top of the file
055            for (int i = 1; i <= cols; i++) {
056                if (i > 1) {
057                    out.write(",");
058                }
059                out.write("\"column " + i + "\"");
060            }
061            out.write("\r\n");
062            out.flush();
063
064            // write the rows
065            for (int i = 1; i <= rows; i++) {
066                for (int j = 1; j <= cols; j++) {
067                    if (j > 1) {
068                        out.write(",");
069                    }
070                    out.write("\"data " + j + "\"");
071                }
072
073                out.write("\r\n");
074                out.flush();
075                if (i % 100000 == 0) {
076                    System.out.print(".");
077                }
078            }
079
080        } catch (final Exception ex) {
081            LOG.error("Issue", ex);
082        } finally {
083            try {
084                if (out != null) {
085                    out.close();
086                }
087                if (fw != null) {
088                    fw.close();
089                }
090            } catch (final Exception ignore) {
091            }
092
093        }
094    }
095
096    private static void printUsage() {
097        System.out.println("INVALID USAGE...");
098        System.out.println("PARAMETER 1 = # OF COLUMNS");
099        System.out.println("PARAMETER 2 = # OF ROWS");
100        System.out.println("Example - java CSVTestFileCreator 10 100");
101    }
102}
