008import java.io.File;
009import java.util.Iterator;
010
011import net.sf.flatpack.DataError;
012import net.sf.flatpack.DataSet;
013import net.sf.flatpack.DefaultParserFactory;
014import net.sf.flatpack.Parser;
015import net.sf.flatpack.ordering.OrderBy;
016import net.sf.flatpack.ordering.OrderColumn;
017
018/**
019 * @author zepernick
020 *
021 * TODO To change the template for this generated type comment go to Window -
022 * Preferences - Java - Code Style - Code Templates
023 */
024public class DelimitedWithPZMapErrors {
025    public static void main(final String[] args) throws Exception {
026        final String mapping = getDefaultMapping();
027        final String data = getDefaultDataFile();
028        call(mapping, data);
029
030    }
031
032    public static String getDefaultDataFile() {
033        return "PEOPLE-CommaDelimitedWithQualifierAndHeaderTrailerRecError.txt";
034    }
035
036    public static String getDefaultMapping() {
037        return "PEOPLE-DelimitedWithHeaderTrailer.pzmap.xml";
038    }
039
040    public static void call(final String mapping, final String data) throws Exception {
041        // delimited by a comma
042        // text qualified by double quotes
043        // ignore first record
044        final Parser pzparser = DefaultParserFactory.getInstance().newDelimitedParser(new File(mapping), new File(data), ',', '"', true);
045        final DataSet ds = pzparser.parse();
046        // re order the data set by last name
047        final OrderBy orderby = new OrderBy();
048        orderby.addOrderColumn(new OrderColumn("CITY", false));
049        orderby.addOrderColumn(new OrderColumn("LASTNAME", true));
050        ds.orderRows(orderby);
051
052        final String[] colNames = ds.getColumns();
053
054        while (ds.next()) {
055
056            if (ds.isRecordID("header")) {
057                System.out.println(">>>>found header");
058                System.out.println("COLUMN NAME: INDICATOR VALUE: " + ds.getString("RECORDINDICATOR"));
059                System.out.println("COLUMN NAME: HEADERDATA VALUE: " + ds.getString("HEADERDATA"));
060                System.out.println("===========================================================================");
061                continue;
062            }
063
064            if (ds.isRecordID("trailer")) {
065                System.out.println(">>>>found trailer");
066                System.out.println("COLUMN NAME: INDICATOR VALUE: " + ds.getString("RECORDINDICATOR"));
067                System.out.println("COLUMN NAME: TRAILERDATA VALUE: " + ds.getString("TRAILERDATA"));
068                System.out.println("===========================================================================");
069                continue;
070            }
071
072            for (final String colName : colNames) {
073                System.out.println("COLUMN NAME: " + colName + " VALUE: " + ds.getString(colName));
074            }
075
076            System.out.println("===========================================================================");
077        }
078
079        System.out.println(">>>>>>ERRORS!!!");
080        final Iterator errors = ds.getErrors().iterator();
081        while (errors.hasNext()) {
082            final DataError dataError = (DataError) errors.next();
083
084            System.out.println("ERROR: " + dataError.getErrorDesc() + " LINE NUMBER: " + dataError.getLineNo());
085        }
086
087    }
088}
