008import java.io.File;
009
010import net.sf.flatpack.DataError;
011import net.sf.flatpack.DataSet;
012import net.sf.flatpack.DefaultParserFactory;
013import net.sf.flatpack.Parser;
014import net.sf.flatpack.ordering.OrderBy;
015import net.sf.flatpack.ordering.OrderColumn;
016import net.sf.flatpack.util.ExcelTransformer;
017
018/**
019 * @author zepernick
020 *
021 * TODO To change the template for this generated type comment go to Window -
022 * Preferences - Java - Code Style - Code Templates
023 */
024public class DelimitedFileExportToExcel {
025    public static void main(final String[] args) throws Exception {
026        final String mapping = getDefaultMapping();
027        final String data = getDefaultDataFile();
028        call(mapping, data);
029
030    }
031
032    public static String getDefaultDataFile() {
033        return "PEOPLE-CommaDelimitedWithQualifier.txt";
034    }
035
036    public static String getDefaultMapping() {
037        return "PEOPLE-Delimited.pzmap.xml";
038    }
039
040    public static void call(final String mapping, final String data) throws Exception {
041        // delimited by a comma
042        // text qualified by double quotes
043        // ignore first record
044        final Parser pzparser = DefaultParserFactory.getInstance().newDelimitedParser(new File(mapping), new File(data), ',', '"', true);
045        final DataSet ds = pzparser.parse();
046
047        // re order the data set by last name
048        final OrderBy orderby = new OrderBy();
049        orderby.addOrderColumn(new OrderColumn("CITY", false));
050        orderby.addOrderColumn(new OrderColumn("LASTNAME", true));
051        ds.orderRows(orderby);
052
053        if (ds.getErrors() != null && !ds.getErrors().isEmpty()) {
054            for (int i = 0; i < ds.getErrors().size(); i++) {
055                final DataError de = (DataError) ds.getErrors().get(i);
056                System.out.println("Error Msg: " + de.getErrorDesc() + " Line: " + de.getLineNo());
057            }
058        }
059
060        // lets write this file out to excel
061        final File xlFile = new File("MyExcelExport.xls");
062        final ExcelTransformer xlTransformer = new ExcelTransformer(ds, xlFile);
063        xlTransformer.writeExcelFile();
064        System.out.println("Excel Workbook Written To: " + xlFile.getAbsolutePath());
065
066    }
067}
