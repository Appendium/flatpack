008import java.io.File;
009
010import net.sf.flatpack.DataSet;
011import net.sf.flatpack.DefaultParserFactory;
012import net.sf.flatpack.Parser;
013
014/**
015 * @author zepernick
016 *
017 * TODO To change the template for this generated type comment go to Window -
018 * Preferences - Java - Code Style - Code Templates
019 */
020public class FixedLengthWithPZMap {
021    public static void main(final String[] args) throws Exception {
022        final String mapping = getDefaultMapping();
023        final String data = getDefaultDataFile();
024        call(mapping, data);
025
026    }
027
028    public static String getDefaultDataFile() {
029        return "PEOPLE-FixedLength.txt";
030    }
031
032    public static String getDefaultMapping() {
033        return "PEOPLE-FixedLength.pzmap.xml";
034    }
035
036    public static void call(final String mapping, final String data) throws Exception {
037        final Parser pzparser = DefaultParserFactory.getInstance().newFixedLengthParser(new File(mapping), new File(data));
038        final DataSet ds = pzparser.parse();
039
040        final String[] colNames = ds.getColumns();
041
042        while (ds.next()) {
043            for (final String colName : colNames) {
044                System.out.println("COLUMN NAME: " + colName + " VALUE: " + ds.getString(colName));
045            }
046
047            System.out.println("===========================================================================");
048        }
049
050    }
051}
