008import java.io.File;
009
010import net.sf.flatpack.DataSet;
011import net.sf.flatpack.DefaultParserFactory;
012import net.sf.flatpack.Parser;
013
014/**
015 * @author zepernick
016 *
017 * TODO To change the template for this generated type comment go to Window -
018 * Preferences - Java - Code Style - Code Templates
019 */
020public class CSVHeaderAndTrailer {
021    public static void main(final String[] args) throws Exception {
022        call(getDefaultMapping(), getDefaultDataFile());
023    }
024
025    public static String getDefaultMapping() {
026        return "PEOPLE-DelimitedWithHeaderTrailer.pzmap.xml";
027    }
028
029    public static String getDefaultDataFile() {
030        return "PEOPLE-CommaDelimitedWithQualifier.txt";
031    }
032
033    public static void call(final String mapping, final String data) throws Exception {
034        final File mapFile = new File(mapping);
035        final File dataFile = new File(data);
036        // delimited by a comma
037        // text qualified by double quotes
038        // ignore first record
039        final Parser pzparser = DefaultParserFactory.getInstance().newDelimitedParser(mapFile, dataFile, ',', '\"', true);
040        final DataSet ds = pzparser.parse();
041        while (ds.next()) {
042
043            if (ds.isRecordID("header")) {
044                System.out.println(">>>>>>Found Header Record");
045                System.out.println("COLUMN NAME: RECORDINDICATOR VALUE: " + ds.getString("RECORDINDICATOR"));
046                System.out.println("COLUMN NAME: HEADERDATA VALUE: " + ds.getString("HEADERDATA"));
047                System.out.println("===========================================================================");
048                continue;
049            }
050
051            if (ds.isRecordID("trailer")) {
052                System.out.println(">>>>>>Found Trailer Record");
053                System.out.println("COLUMN NAME: RECORDINDICATOR VALUE: " + ds.getString("RECORDINDICATOR"));
054                System.out.println("COLUMN NAME: TRAILERDATA VALUE: " + ds.getString("TRAILERDATA"));
055                System.out.println("===========================================================================");
056                continue;
057            }
058
059            System.out.println("COLUMN NAME: FIRSTNAME VALUE: " + ds.getString("FIRSTNAME"));
060            System.out.println("COLUMN NAME: LASTNAME VALUE: " + ds.getString("LASTNAME"));
061            System.out.println("COLUMN NAME: ADDRESS VALUE: " + ds.getString("ADDRESS"));
062            System.out.println("COLUMN NAME: CITY VALUE: " + ds.getString("CITY"));
063            System.out.println("COLUMN NAME: STATE VALUE: " + ds.getString("STATE"));
064            System.out.println("COLUMN NAME: ZIP VALUE: " + ds.getString("ZIP"));
065            System.out.println("===========================================================================");
066        }
067
068        if (ds.getErrors() != null && !ds.getErrors().isEmpty()) {
069            System.out.println("FOUND ERRORS IN FILE");
070        }
071    }
072}
