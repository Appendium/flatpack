008import java.io.File;
009import java.text.SimpleDateFormat;
010
011import net.sf.flatpack.DataSet;
012import net.sf.flatpack.DefaultParserFactory;
013import net.sf.flatpack.Parser;
014
015/**
016 * @author zepernick
017 *
018 * TODO To change the template for this generated type comment go to Window -
019 * Preferences - Java - Code Style - Code Templates
020 */
021public class NumericsAndDates {
022    public static void main(final String[] args) throws Exception {
023        final String mapping = getDefaultMapping();
024        final String data = getDefaultDataFile();
025        call(mapping, data);
026    }
027
028    public static String getDefaultDataFile() {
029        return "INVENTORY-CommaDelimitedWithQualifier.txt";
030    }
031
032    public static String getDefaultMapping() {
033        return "INVENTORY-Delimited.pzmap.xml";
034    }
035
036    public static void call(final String mapping, final String data) throws Exception {
037        // wll provide a clean format for printing the date to the screen
038        final SimpleDateFormat sdf = new SimpleDateFormat("MM/dd/yyyy");
039
040        // delimited by a comma
041        // text qualified by double quotes
042        // ignore first record
043        final Parser pzparser = DefaultParserFactory.getInstance().newDelimitedParser(new File(mapping), new File(data), ',', '\"', true);
044        final DataSet ds = pzparser.parse();
045        // demonstrates the casting abilities of FlatPack
046        while (ds.next()) {
047            System.out.println("Item Desc: " + ds.getString("ITEM_DESC") + " (String)");
048            System.out.println("In Stock: " + ds.getInt("IN_STOCK") + " (int)");
049            System.out.println("Price: " + ds.getDouble("PRICE") + " (double)");
050            System.out.println("Received Dt: " + sdf.format(ds.getDate("LAST_RECV_DT")) + " (Date)");
051            System.out.println("===========================================================================");
052        }
053    }
054}
