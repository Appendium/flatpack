001package net.sf.flatpack.examples.lowlevelparse;
002
003import java.io.BufferedReader;
004import java.io.File;
005import java.io.FileReader;
006import java.util.List;
007
008import org.slf4j.Logger;
009import org.slf4j.LoggerFactory;
010
011import net.sf.flatpack.util.ParserUtils;
012
013/*
014 * Created on Nov 27, 2005
015 *
016 */
017
018/**
019 * @author zepernick
020 *
021 */
022public class LowLevelParse {
023    private static final Logger LOG = LoggerFactory.getLogger(LowLevelParse.class);
024
025    public static void main(final String[] args) {
026        final String data = getDefaultDataFile();
027        try {
028            call(data);
029        } catch (final Exception e) {
030            LOG.error("issue", e);
031        }
032    }
033
034    public static String getDefaultDataFile() {
035        return "PEOPLE-CommaDelimitedWithQualifier.txt";
036    }
037
038    public static void call(final String data) throws Exception {
039        final File textFile = new File(data);
040        String line = null;
041        List elements = null;
042
043        try (FileReader fr = new FileReader(textFile); BufferedReader br = new BufferedReader(fr)) {
044
045            while ((line = br.readLine()) != null) {
046                if (line.trim().length() == 0) {
047                    continue;
048                }
049
050                // tell the parser to split using a comma delimiter with a "
051                // text qualifier. The text qualifier is optional, it can be
052                // null
053                // or empty
054                elements = ParserUtils.splitLine(line, ',', '"', 10, false, false);
055
056                for (int i = 0; i < elements.size(); i++) {
057                    System.out.println("Column " + i + ": " + (String) elements.get(i));
058                }
059
060                System.out.println("===========================================================================");
061            }
062        } catch (final Exception ex) {
063            LOG.error("issue", ex);
064        }
065
066    }
067}
