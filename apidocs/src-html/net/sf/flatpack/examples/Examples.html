231    public void doStringBuffer() {
232        final int repeat = ConsoleMenu.getInt("How many times?", 100000);
233        final int characters = ConsoleMenu.getInt("How many char?", 20);
234
235        long start = System.currentTimeMillis();
236        for (int i = 0; i < repeat; i++) {
237            final StringBuffer sb = new StringBuffer();
238            for (int u = 0; u < 1000; u++) {
239                sb.append("h");
240            }
241        }
242        long stop = System.currentTimeMillis();
243
244        System.out.println("Creating new SB " + (stop - start) + " ms.");
245
246        start = System.currentTimeMillis();
247        final StringBuffer sb = new StringBuffer();
248        for (int i = 0; i < repeat; i++) {
249            for (int u = 0; u < characters; u++) {
250                sb.append("h");
251            }
252            sb.delete(0, sb.length());
253        }
254        stop = System.currentTimeMillis();
255
256        System.out.println("Deleting existing SB " + (stop - start) + " ms.");
257
258    }

260    public void doTestParsers() {
261        final int repeat = ConsoleMenu.getInt("How many Rows?", 1000);
262        final int numberOfCols = ConsoleMenu.getInt("How many columns?", 100);
263        final boolean qualif = ConsoleMenu.getBoolean("With qualifier?", true);
264
265        final StringBuffer aRow = new StringBuffer();
266        for (int i = 0; i < numberOfCols; i++) {
267            if (qualif) {
268                aRow.append("\"");
269            }
270            aRow.append("Column ").append(i);
271            if (qualif) {
272                aRow.append("\"");
273            }
274        }
275
276        final String line = aRow.toString();
277
278        final long start = System.currentTimeMillis();
279        for (int i = 0; i < repeat; i++) {
280            ParserUtils.splitLine(line, ',', '\"', FPConstants.SPLITLINE_SIZE_INIT, false, false);
281        }
282        final long stop = System.currentTimeMillis();
283
284        System.out.println("ParserUtil " + (stop - start) + " ms.");
285
286    }
