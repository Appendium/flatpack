077    /**
078     * Returns an ArrayList of items in a delimited string. If there is no
079     * qualifier around the text, the qualifier parameter can be left null, or
080     * empty. There should not be any line breaks in the string. Each line of
081     * the file should be passed in individually.
082     * Elements which are not qualified will have leading and trailing white
083     * space removed.  This includes unqualified elements, which may be
084     * contained in an unqualified parse: "data",  data  ,"data"
085     *
086     * Special thanks to Benoit for contributing this much improved speedy parser :0)
087     *
088     * @author Benoit Xhenseval
089     * @param line -
090     *            String of data to be parsed
091     * @param delimiter -
092     *            Delimiter seperating each element
093     * @param qualifier -
094     *            qualifier which is surrounding the text
095     * @param initialSize -
096     *            intial capacity of the List size
097     * @param preserveLeadingWhitespace
098     *            Keep any leading spaces
099     * @param preserveTrailingWhitespace
100     *            Keep any trailing spaces
101     * @return List
102     */
103    public static List<String> splitLine(final String line, final char delimiter, final char qualifier, final int initialSize,
104            final boolean preserveLeadingWhitespace, final boolean preserveTrailingWhitespace) {
105        final List<String> list = new ArrayList<String>(initialSize);
106
107        if (delimiter == 0) {
108            list.add(line);
109            return list;
110        } else if (line == null) {
111            return list;
112        }
113
114        String trimmedLine;
115        if (delimiter == '\t' || delimiter == ' ') {
116            // skip the trim for these delimiters, doing the trim will mess up the parse
117            // on empty records which contain just the delimiter
118            trimmedLine = line;
119        } else {
120            trimmedLine = line;
121            if (!preserveLeadingWhitespace) {
122                trimmedLine = ParserUtils.lTrim(line);
123            }
124            if (!preserveTrailingWhitespace) {
125                trimmedLine = ParserUtils.rTrim(line);
126            }
127        }
128
129        final int size = trimmedLine.length();
130
131        if (size == 0) {
132            list.add("");
133            return list;
134        }
135
136        boolean insideQualifier = false;
137        char previousChar = 0;
138        int startBlock = 0;
139        int endBlock = 0;
140        boolean blockWasInQualifier = false;
141
142        final String doubleQualifier = "" + qualifier + qualifier;
143        for (int i = 0; i < size; i++) {
144
145            final char currentChar = trimmedLine.charAt(i);
146            if (currentChar != delimiter && currentChar != qualifier) {
147                previousChar = currentChar;
148                endBlock = i + 1;
149                continue;
150            }
151
152            if (currentChar == delimiter) {
153                // we've found the delimiter (eg ,)
154                if (!insideQualifier) {
155                    String trimmed = trimmedLine.substring(startBlock, endBlock > startBlock ? endBlock : startBlock + 1);
156                    if (!blockWasInQualifier) {
157                        if (!preserveLeadingWhitespace) {
158                            trimmed = ParserUtils.lTrim(trimmed);
159                        }
160                        if (!preserveTrailingWhitespace) {
161                            trimmed = ParserUtils.rTrim(trimmed);
162                        }
163                    }
164
165                    if (trimmed == null || trimmed.length() == 1 && (trimmed.charAt(0) == delimiter || trimmed.charAt(0) == qualifier)) {
166                        list.add("");
167                    } else {
168                        list.add(trimmed);
169                    }
170                    blockWasInQualifier = false;
171                    startBlock = i + 1;
172                }
173            } else if (currentChar == qualifier) {
174                if (!insideQualifier && previousChar != qualifier) {
175                    if (previousChar == delimiter || previousChar == 0 || previousChar == ' ') {
176                        insideQualifier = true;
177                        startBlock = i + 1;
178                    } else {
179                        endBlock = i + 1;
180                    }
181                } else {
182                    if (i + 1 < size && delimiter != ' ') {
183                        // this is used to allow unescaped qualifiers to be contained within the element
184                        // do not run this check is a space is being used as a delimiter
185                        // we don't want to trim the delimiter off
186                        // loop until we find a char that is not a space, or we reach the end of the line.
187                        int start = i + 1;
188                        char charToCheck = trimmedLine.charAt(start);
189                        while (charToCheck == ' ') {
190                            start++;
191                            if (start == size) {
192                                break;
193                            }
194                            charToCheck = trimmedLine.charAt(start);
195                        }
196
197                        if (charToCheck != delimiter) {
198                            previousChar = currentChar;
199                            endBlock = i + 1;
200                            continue;
201                        }
202
203                    }
204                    insideQualifier = false;
205                    blockWasInQualifier = true;
206                    endBlock = i;
207                    // last column (e.g. finishes with ")
208                    if (i == size - 1) {
209                        String str = trimmedLine.substring(startBlock, size - 1);
210                        str = str.replaceAll(doubleQualifier, String.valueOf(qualifier));
211                        list.add(str);
212                        startBlock = i + 1;
213                    }
214                }
215            }
216            previousChar = currentChar;
217        }
218
219        if (startBlock < size) {
220            String str = trimmedLine.substring(startBlock, size);
221            str = str.replaceAll(doubleQualifier, String.valueOf(qualifier));
222            if (blockWasInQualifier) {
223                if (str.charAt(str.length() - 1) == qualifier) {
224                    list.add(str.substring(0, str.length() - 1));
225                } else {
226                    list.add(str);
227                }
228            } else {
229                String s = str;
230                if (!preserveLeadingWhitespace) {
231                    s = ParserUtils.lTrim(s);
232                }
233                if (!preserveTrailingWhitespace) {
234                    s = ParserUtils.rTrim(s);
235                }
236                list.add(s);
237            }
238        } else if (trimmedLine.charAt(size - 1) == delimiter) {
239            list.add("");
240        }
241
242        return list;
243    }

262    /**
263     * Removes empty space from the beginning of a string
264     *
265     * @param value -
266     *            to be trimmed
267     * @return String
268     */
269    public static String lTrim(final String value) {
270        if (value == null) {
271            return null;
272        }
273
274        String trimmed = value;
275        int offset = 0;
276        final int maxLength = value.length();
277        while (offset < maxLength && (value.charAt(offset) == ' ' || value.charAt(offset) == '\t')) {
278            offset++;
279        }
280
281        if (offset > 0) {
282            trimmed = value.substring(offset);
283        }
284
285        return trimmed;
286    }

314    /**
315     * Removes empty space from the end of a string
316     *
317     * @param value -
318     *            to be trimmed
319     * @return String
320     */
321    public static String rTrim(final String value) {
322        if (value == null) {
323            return null;
324        }
325
326        String trimmed = value;
327        int offset = value.length() - 1;
328        while (offset > -1 && (value.charAt(offset) == ' ' || value.charAt(offset) == '\t')) {
329            offset--;
330        }
331
332        if (offset < value.length() - 1) {
333            trimmed = value.substring(0, offset + 1);
334        }
335
336        return trimmed;
337    }

368        final StringBuilder s = new StringBuilder();
369        for (int i = 0; i < theString.length(); i++) {

379    /**
380     * Returns a list of ColumnMetaData objects. This is for use with delimited
381     * files. The first line of the file which contains data will be used as the
382     * column names
383     *
384     * @param line
385     * @param delimiter
386     * @param qualifier
387     * @param p
388     *          PZParser used to specify additional option when working with the ColumnMetaData. Can be null
389     * @param addSuffixToDuplicateColumnNames 
390     * @return PZMetaData
391     */
392    public static MetaData getPZMetaDataFromFile(final String line, final char delimiter, final char qualifier, final Parser p,
393            boolean addSuffixToDuplicateColumnNames) {
394        final List<ColumnMetaData> results = new ArrayList<ColumnMetaData>();
395        final Set<String> dupCheck = new HashSet<String>();
396
397        final List<String> lineData = splitLine(line, delimiter, qualifier, FPConstants.SPLITLINE_SIZE_INIT, false, false);
398        for (final String colName : lineData) {
399            final ColumnMetaData cmd = new ColumnMetaData();
400            String colNameToUse = colName;
401            if (dupCheck.contains(colNameToUse)) {
402                if (!addSuffixToDuplicateColumnNames) {
403                    throw new FPException("Duplicate Column Name In File: " + colNameToUse);
404                } else {
405                    int count = 2;
406                    while (dupCheck.contains(colNameToUse + count)) {
407                        count++;
408                    }
409                    colNameToUse = colName + count;
410                }
411            }
412            cmd.setColName(colNameToUse);
413            results.add(cmd);
414            dupCheck.add(cmd.getColName());
415        }
416
417        return new MetaData(results, buidColumnIndexMap(results, p));
418    }

420    /**
421     * Determines if the given line is the first part of a multiline record.  It does this by verifying that the
422     * qualifer on the last element is not closed
423     *
424     * @param chrArry -
425     *            char data of the line
426     * @param delimiter -
427     *            delimiter being used
428     * @param qualifier -
429     *            qualifier being used
430     * @return boolean
431     */
432    public static boolean isMultiLine(final char[] chrArry, final char delimiter, final char qualifier) {
433
434        // check if the last char is the qualifier, if so then this a good
435        // chance it is not multiline
436        if (chrArry[chrArry.length - 1] != qualifier) {
437            // could be a potential line break
438            boolean qualiFound = false;
439            for (int i = chrArry.length - 1; i >= 0; i--) {
440                if (chrArry[i] == ' ') {
441                    continue;
442                }
443
444                // check to see if we can find a qualifier followed by a
445                // delimiter
446                // remember we are working are way backwards on the line
447                if (qualiFound) {
448                    if (chrArry[i] == delimiter) {
449                        // before deciding if this is the begining of a qualified new line
450                        // I think we have to go back to the beginning of the line and see if we are inside a qualified
451                        // field or not?
452                        // return true;
453                        boolean qualifiedContent = chrArry[0] == qualifier;
454                        for (int index = 0; index < chrArry.length; index++) {
455                            char currentChar = chrArry[index];
456                            qualifiedContent = currentChar == qualifier;
457                            if (qualifiedContent) {
458                                // go until first occurence of closing qualifierdelimiter combination
459                                for (; index < chrArry.length; index++) {
460                                    if (chrArry[index] == delimiter && chrArry[++index] == qualifier) {
461                                        qualifiedContent = false;
462                                    }
463                                }
464                            }
465                        }
466                        return qualifiedContent;
467                    }
468                    // guard against multiple qualifiers in the sequence [ ,""We ]
469                    qualiFound = chrArry[i] == qualifier;
470                } else if (chrArry[i] == delimiter) {
471                    // if we have a delimiter followed by a qualifier, then we
472                    // have moved on to a new element and this could not be multiline.
473                    // start a new loop here in case there is
474                    // space between the delimiter and qualifier
475                    for (int j = i - 1; j >= 0; j--) {
476                        if (chrArry[j] == ' ') {
477                            continue;
478                        } else if (chrArry[j] == qualifier) {
479                            return false;
480                        }
481                        break;
482                    }
483
484                } else if (chrArry[i] == qualifier) {
485                    qualiFound = true;
486                }
487            }
488        } else {
489            // we have determined that the last char on the line is a qualifier.
490            // This most likely means
491            // that this is not multiline, however we must account for the
492            // following scenario
493            // data,data,"
494            // data
495            // /data"
496            for (int i = chrArry.length - 1; i >= 0; i--) {
497                if (i == chrArry.length - 1 || chrArry[i] == ' ') {
498                    // skip the first char, or any spaces we come across between
499                    // the delimiter and qualifier
500                    continue;
501                }
502                if (chrArry[i] == delimiter) {
503                    // before deciding if this is the begining of a qualified new line
504                    // I think we have to go back to the beginning of the line and see if we are inside a qualified
505                    // field or not?
506                    // return true;
507                    boolean qualifiedContent = chrArry[0] == qualifier;
508                    for (int index = 0; index < chrArry.length; index++) {
509                        char currentChar = chrArry[index];
510                        qualifiedContent = currentChar == qualifier;
511                        if (qualifiedContent) {
512                            // go until first occurence of closing qualifierdelimiter combination
513                            for (; index < chrArry.length; index++) {
514                                if (chrArry[index] == delimiter && chrArry[++index] == qualifier) {
515                                    qualifiedContent = false;
516                                }
517                            }
518                        }
519                    }
520                    return qualifiedContent;
521                }
522                break;
523            }
524        }
525
526        return false;
527    }

529    public static Map<String, Integer> calculateRecordLengths(final MetaData columnMD) {
530        final Map<String, Integer> recordLengths = new HashMap<String, Integer>();
531
532        // first the basic columns
533        int recordLength = 0;
534        for (final ColumnMetaData cmd : columnMD.getColumnsNames()) {
535            recordLength += cmd.getColLength();
536        }
537
538        recordLengths.put(FPConstants.DETAIL_ID, recordLength);
539
540        final Iterator<Entry<String, XMLRecordElement>> columnMDIt = columnMD.xmlRecordIterator();
541        while (columnMDIt.hasNext()) {
542            final Entry<String, XMLRecordElement> entry = columnMDIt.next();
543            final List<ColumnMetaData> cmds = entry.getValue().getColumns();
544
545            recordLength = 0;
546            for (final ColumnMetaData cmd : cmds) {
547                recordLength += cmd.getColLength();
548            }
549
550            recordLengths.put(entry.getKey(), recordLength);
551        }
552
553        return recordLengths;
554
555    }

557    public static String getCMDKeyForDelimitedFile(final MetaData columnMD, final List<String> lineElements) {
558        if (!columnMD.isAnyRecordFormatSpecified()) {
559            // no <RECORD> elments were specifed for this parse, just return the
560            // detail id
561            return FPConstants.DETAIL_ID;
562        }
563        final Iterator<Entry<String, XMLRecordElement>> mapEntries = columnMD.xmlRecordIterator();
564        // loop through the XMLRecordElement objects and see if we need a
565        // different MD object
566        while (mapEntries.hasNext()) {
567            final Entry<String, XMLRecordElement> entry = mapEntries.next();
568            final XMLRecordElement recordXMLElement = entry.getValue();
569
570            if (recordXMLElement.getElementCount() > 0 && recordXMLElement.getElementCount() == lineElements.size()) {
571                // determing which <record> mapping to use by the number of elements
572                // contained on the line
573                return entry.getKey();
574            } else if (recordXMLElement.getElementNumber() > lineElements.size()) {
575                // make sure the element referenced in the mapping exists
576                continue;
577            }
578
579            final String lineElement = lineElements.get(recordXMLElement.getElementNumber() - 1);
580            if (lineElement.equals(recordXMLElement.getIndicator())) {
581                // we found the MD object we want to return
582                return entry.getKey();
583            }
584
585        }
586
587        // must be a detail line
588        return FPConstants.DETAIL_ID;
589    }

599    public static int getColumnIndex(final String key, final MetaData columnMD, final String colName, final boolean columNameCaseSensitive) {
600        int idx = -1;
601        String column = colName;
602        if (!columNameCaseSensitive) {
603            column = colName.toLowerCase(Locale.getDefault());
604        }
605        idx = columnMD.getColumnIndex(key, column);
606
607        if (idx < 0) {
608            throw new NoSuchElementException(
609                    "Column [" + column + "] does not exist, check case/spelling." + (key != null ? " key:[" + key + "]" : ""));
610        }
611        return idx;
612    }

661    /**
662     * <p>
663     * Returns padding using the specified delimiter repeated to a given length.
664     * </p>
665     *
666     * <pre>
667     *                PZStringUtils.padding(0, 'e')  = &quot;&quot;
668     *                PZStringUtils.padding(3, 'e')  = &quot;eee&quot;
669     *                PZStringUtils.padding(-2, 'e') = IndexOutOfBoundsException
670     * </pre>
671     *
672     * <p>
673     * Note: this method doesn't not support padding with <a
674     * href="http://www.unicode.org/glossary/#supplementary_character">Unicode
675     * Supplementary Characters</a> as they require a pair of <code>char</code>s
676     * to be represented.
677     * </p>
678     *
679     * @param repeat
680     *            number of times to repeat delim
681     * @param padChar
682     *            character to repeat
683     * @return String with repeated character
684     * @throws IndexOutOfBoundsException
685     */
686    public static String padding(final int repeat, final char padChar) {
687        if (repeat < 0) {
688            throw new IndexOutOfBoundsException("Cannot pad a negative amount: " + repeat);
689        }
690        final char[] buf = new char[repeat];
691        for (int i = 0; i < buf.length; i++) {
692            buf[i] = padChar;
693        }
694        return new String(buf);
695    }

697    /**
698     * Build a map of name/position based on a list of ColumnMetaData.
699     *
700     * @author Benoit Xhenseval
701     * @author Paul Zepernick
702     * @param columns
703     * @param p
704     *         Reference to Parser which can provide additional options on how the
705     *         map should be build.  This can be NULL.
706     * @return a new Map
707     */
708    public static Map<String, Integer> buidColumnIndexMap(final List<ColumnMetaData> columns, final Parser p) {
709        Map<String, Integer> map = null;
710        if (columns != null && !columns.isEmpty()) {
711            map = new HashMap<String, Integer>();
712            int idx = 0;
713            for (final ColumnMetaData meta : columns) {
714                String colName = meta.getColName();
715                if (p != null && !p.isColumnNamesCaseSensitive()) {
716                    // user has selected to make column names case sensitive
717                    // on lookups
718                    colName = colName.toLowerCase(Locale.getDefault());
719                }
720                map.put(colName, idx++);
721            }
722        }
723        return map;
724    }

726    /**
727     * Removes chars from the String that could not
728     * be parsed into a Long value
729     *
730     *      PZStringUtils.stripNonLongChars("1000.25") = "1000"
731     *
732     * Method will truncate everything to the right of the decimal
733     * place when encountered.
734     *
735     * @param value
736     * @return String
737     */
738    public static String stripNonLongChars(final String value) {
739        final StringBuilder newString = new StringBuilder();
740
741        for (int i = 0; i < value.length(); i++) {
742            final char c = value.charAt(i);
743            if (c == '.') {
744                // stop if we hit a decimal point
745                break;
746            } else if (c >= '0' && c <= '9' || c == '-') {
747                newString.append(c);
748            }
749        }
750        // check to make sure we do not have a single length string with
751        // just a minus sign
752        final int sLen = newString.length();
753        final String s = newString.toString();
754        if (sLen == 0 || sLen == 1 && "-".equals(s)) {
755            return "0";
756        }
757
758        return newString.toString();
759    }

768    public static String stripNonDoubleChars(final String value) {

776        }
777        final int sLen = newString.length();
778        final String s = newString.toString();
779        if (sLen == 0 || sLen == 1 && (".".equals(s) || "-".equals(s))) {
780            return "0";
781        }
782
783        return newString.toString();
784    }

843        } catch (final InstantiationException ex) {
844            throw new FPConvertException(ex);
845        } catch (final ClassNotFoundException ex) {
846            throw new FPConvertException(ex);
847        }

866    /**
867     * Returns a definition of pz column metadata from a given
868     * pz datastructure held in an SQL database
869     *
870     * @param con
871     *          Database connection containing the Datafile and Datastructure
872     *          tables
873     * @param dataDefinition
874     *          Name of the data definition stored in the Datafile table
875     * @param parser
876     *                  Instance of the parser being used for the file.  It will be checked to get the table names
877     *                  for the DATASTRUCTURE table and DATAFILE table.
878     * @throws SQLException
879     * @return List
880     */
881    public static List<ColumnMetaData> buildMDFromSQLTable(final Connection con, final String dataDefinition, final Parser parser)
882            throws SQLException {
883        final List<ColumnMetaData> cmds = new ArrayList<ColumnMetaData>();
884        final String dfTbl = parser != null ? parser.getDataFileTable() : "DATAFILE";
885        final String dsTbl = parser != null ? parser.getDataStructureTable() : "DATASTRUCTURE";
886        final StringBuilder sqlSb = new StringBuilder();
887
888        sqlSb.append("SELECT * FROM ").append(dfTbl).append(" INNER JOIN ").append(dsTbl).append(" ON ").append(dfTbl).append(".DATAFILE_NO = ")
889                .append(dsTbl).append(".DATAFILE_NO " + "WHERE DATAFILE_DESC = ? ORDER BY DATASTRUCTURE_COL_ORDER");
890
891        try (PreparedStatement stmt = con.prepareStatement(sqlSb.toString())) { // always use PreparedStatement
892            // as the DB can do clever things.
893            stmt.setString(1, dataDefinition);
894            try (ResultSet rs = stmt.executeQuery()) {
895
896                int recPosition = 1;
897                // put array of columns together. These will be used to put together
898                // the dataset when reading in the file
899                while (rs.next()) {
900
901                    final ColumnMetaData column = new ColumnMetaData();
902                    column.setColName(rs.getString("DATASTRUCTURE_COLUMN"));
903                    column.setColLength(rs.getInt("DATASTRUCTURE_LENGTH"));
904                    column.setStartPosition(recPosition);
905                    column.setEndPosition(recPosition + rs.getInt("DATASTRUCTURE_LENGTH") - 1);
906                    recPosition += rs.getInt("DATASTRUCTURE_LENGTH");
907
908                    cmds.add(column);
909                }
910            }
911            if (cmds.isEmpty()) {
912                throw new FPException("Data File Key [" + dataDefinition + "] Is Not In The database OR No Columns Specified In Table");
913            }
914        }
915        return cmds;
916    }
