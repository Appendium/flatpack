033package net.sf.flatpack.util;
034
035import java.util.ArrayList;
036import java.util.Iterator;
037import java.util.List;
038import java.util.Map.Entry;
039
040import net.sf.flatpack.structure.ColumnMetaData;
041import net.sf.flatpack.xml.MetaData;
042import net.sf.flatpack.xml.XMLRecordElement;
043
044/**
045 * Collection of parser utilities related to the parsing of fixed width files.
046 *
047 * @author paul zepernick
048 */
049public final class FixedWidthParserUtils {
050    private FixedWidthParserUtils() {
051    }
052
053    /**
054     * Splits up a fixed width line of text
055     *
056     * @param columnMetaData
057     *            Collection of ColumnMetaData to parse the line
058     * @param lineToParse
059     *            Line of text to be parsed against the ColumnMetaData
060     * @param preserveLeadingWhitespace
061     *            Keep any leading spaces
062     * @param preserveTrailingWhitespace
063     *            Keep any trailing spaces
064     * @return List Collection of Strings. Each element representing a column
065     */
066    public static List<String> splitFixedText(final List<ColumnMetaData> columnMetaData, final String lineToParse,
067            final boolean preserveLeadingWhitespace, final boolean preserveTrailingWhitespace) {
068        final List<String> splitResult = new ArrayList<String>();
069        int recPosition = 1;
070        for (final ColumnMetaData colMetaDataObj : columnMetaData) {
071            String tempValue = lineToParse.substring(recPosition - 1, recPosition + colMetaDataObj.getColLength() - 1);
072            recPosition += colMetaDataObj.getColLength();
073            // make sure that we preserve leading and trailing spaces as user has requested
074            // This was previously issuing a trim()
075            if (!preserveLeadingWhitespace) {
076                tempValue = ParserUtils.lTrim(tempValue);
077            }
078            if (!preserveTrailingWhitespace) {
079                tempValue = ParserUtils.rTrim(tempValue);
080            }
081
082            splitResult.add(tempValue);
083        }
084
085        return splitResult;
086    }
087
088    /**
089     * Returns the key to the list of ColumnMetaData objects. Returns the
090     * correct MetaData per the mapping file and the data contained on the line
091     *
092     *
093     * @param columnMD
094     * @param line
095     * @return List - ColumMetaData
096     */
097    public static String getCMDKey(final MetaData columnMD, final String line) {
098        if (!columnMD.isAnyRecordFormatSpecified()) {
099            // no <RECORD> elements were specified for this parse, just return the
100            // detail id
101            return FPConstants.DETAIL_ID;
102        }
103        final Iterator<Entry<String, XMLRecordElement>> mapEntries = columnMD.xmlRecordIterator();
104        // loop through the XMLRecordElement objects and see if we need a
105        // different MD object
106        while (mapEntries.hasNext()) {
107            final Entry<String, XMLRecordElement> entry = mapEntries.next();
108            final XMLRecordElement recordXMLElement = entry.getValue();
109
110            if (recordXMLElement.getEndPositition() > line.length()) {
111                // make sure our substring is not going to fail
112                continue;
113            }
114            final int subfrm = recordXMLElement.getStartPosition() - 1; // convert
115            // to 0
116            // based
117            final int subto = recordXMLElement.getEndPositition();
118            if (line.substring(subfrm, subto).equals(recordXMLElement.getIndicator())) {
119                // we found the MD object we want to return
120                return entry.getKey();
121            }
122
123        }
124
125        // must be a detail line
126        return FPConstants.DETAIL_ID;
127    }
128}
