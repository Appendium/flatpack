004package net.sf.flatpack.util;
005
006import java.util.ArrayList;
007import java.util.Collection;
008import java.util.HashMap;
009import java.util.List;
010import java.util.Map;
011
012import net.sf.flatpack.structure.ColumnMetaData;
013
014/**
015 * Helper class to create ColumnMetaData and Mapping for the Writer Factory.
016 * @author Benoit Xhenseval
017 */
018public final class FlatpackWriterUtil {
019    private FlatpackWriterUtil() {
020    }
021
022    /**
023     * Creates a Mapping for a WriterFactory for the given list of columns.
024     * @param colsAsCsv comma-separated column names
025     * @return a map to be used in, for instance, DelimiterWriterFactory
026     */
027    public static Map<String, Object> buildParametersForColumns(final String colsAsCsv) {
028        final Map<String, Object> mapping = new HashMap<>();
029        mapping.put(FPConstants.DETAIL_ID, buildColumns(colsAsCsv));
030        return mapping;
031    }
032
033    /**
034     * Create a new list of ColumnMetaData based on a CSV list of column titles.
035     * @param colsAsCsv
036     * @return new list
037     */
038    public static List<ColumnMetaData> buildColumns(final String colsAsCsv) {
039        final List<ColumnMetaData> listCol = new ArrayList<>();
040        buildColumns(listCol, colsAsCsv);
041        return listCol;
042    }
043
044    public static void buildColumns(final List<ColumnMetaData> listCol, final Collection<String> cols) {
045        if (cols != null) {
046            cols.forEach(s -> listCol.add(new ColumnMetaData(s)));
047        }
048    }
049
050    public static void buildColumns(final List<ColumnMetaData> listCol, final String cols) {
051        if (cols != null) {
052            for (final String s : cols.split(",")) {
053                listCol.add(new ColumnMetaData(s));
054            }
055        }
056    }
057
058    public static void buildColumns(final List<ColumnMetaData> listCol, final String... cols) {
059        if (cols != null) {
060            for (final String s : cols) {
061                listCol.add(new ColumnMetaData(s));
062            }
063        }
064    }
065}
