001package net.sf.flatpack.util;
002
003/**
004 * Thanks to the Apache Commons Contributors.
005 *
006 * @author Jakarta Commons
007 */
008public final class FPStringUtils {
009    private FPStringUtils() {
010    }
011
012    /**
013     * <p>Checks if a String is whitespace, empty ("") or null.</p>
014     *
015     * <pre>
016     * PZStringUtils.isBlank(null)      = true
017     * PZStringUtils.isBlank("")        = true
018     * PZStringUtils.isBlank(" ")       = true
019     * PZStringUtils.isBlank("bob")     = false
020     * PZStringUtils.isBlank("  bob  ") = false
021     * </pre>
022     *
023     * @param str  the String to check, may be null
024     * @return <code>true</code> if the String is null, empty or whitespace
025     * @since 2.0
026     */
027    public static boolean isBlank(final String str) {
028        final int strLen = str == null ? 0 : str.length();
029        if (strLen == 0) {
030            return true;
031        }
032        for (int i = 0; i < strLen; i++) {
033            if (!Character.isWhitespace(str.charAt(i))) {
034                return false;
035            }
036        }
037        return true;
038    }
039
040    /**
041     * <p>Checks if a String is not empty (""), not null and not whitespace only.</p>
042     *
043     * <pre>
044     * PZStringUtils.isNotBlank(null)      = false
045     * PZStringUtils.isNotBlank("")        = false
046     * PZStringUtils.isNotBlank(" ")       = false
047     * PZStringUtils.isNotBlank("bob")     = true
048     * PZStringUtils.isNotBlank("  bob  ") = true
049     * </pre>
050     *
051     * @param str  the String to check, may be null
052     * @return <code>true</code> if the String is
053     *  not empty and not null and not whitespace
054     * @since 2.0
055     */
056    public static boolean isNotBlank(final String str) {
057        return !FPStringUtils.isBlank(str);
058    }
059
060}
