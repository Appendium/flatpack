001package net.sf.flatpack.util;
002
003/**
004 * Generic exception for FlatPack
005 *
006 * @author Paul Zepernick
007 */
008public class FPException extends RuntimeException {
009    private static final long serialVersionUID = -4269317129024968528L;
010
011    public FPException() {
012    }
013
014    /**
015     * @param message
016     */
017    public FPException(final String message) {
018        super(message);
019    }
020
021    /**
022     * @param cause
023     */
024    public FPException(final Throwable cause) {
025        super(cause);
026    }
027
028    /**
029     * @param message
030     * @param cause
031     */
032    public FPException(final String message, final Throwable cause) {
033        super(message, cause);
034    }
035}
