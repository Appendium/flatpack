001package net.sf.flatpack.writer;
002
003import java.io.IOException;
004import java.math.BigDecimal;
005import java.util.Arrays;
006import java.util.List;
007import java.util.Map;
008
009import net.sf.flatpack.structure.ColumnMetaData;
010import net.sf.flatpack.util.FPConstants;
011
012/**
013 *
014 * @author Dirk Holmes and Holger Holger Hoffstatte
015 */
016public class FixedLengthWriter extends AbstractWriter {
017    private static final int MAX_CHAR_TO_USE_LOOP = 8;
018    private final Map<?, ?> columnMapping;
019    private final char padChar;
020
021    protected FixedLengthWriter(final Map parsedMapping, final java.io.Writer output, final char padChar) throws IOException {
022        super(output);
023        this.columnMapping = parsedMapping;
024        this.padChar = padChar;
025    }
026
027    @Override
028    public Writer addRecordEntry(final String columnName, final Object value) {
029        if (value != null) {
030            final ColumnMetaData metaData = this.getColumnMetaData(columnName);
031
032            final String valueString = value.toString();
033            if (valueString.length() > metaData.getColLength()) {
034                throw new IllegalArgumentException(valueString + " exceeds the maximum length for column " + columnName + "("
035                        + metaData.getColLength() + ")");
036            }
037        }
038        return super.addRecordEntry(columnName, value);
039    }
040
041    @Override
042    public Writer nextRecord() throws IOException {
043        for (final ColumnMetaData element : getColumnMetaData()) {
044            final Object value = this.getRowMap().get(element.getColName());
045            this.write(this.formattedValue(value, element));
046        }
047
048        return super.nextRecord();
049    }
050
051    protected char[] formattedValue(final Object val, final ColumnMetaData element) {
052        Object value = val;
053        if (value == null) {
054            // TODO DO: maybe have a way to substitute default values here?
055            value = "";
056        }
057
058        String stringValue;
059        if (value instanceof BigDecimal) {
060            final BigDecimal bd = (BigDecimal) value;
061            stringValue = bd.signum() == 0 ? "0" : bd.toPlainString();
062        } else {
063            stringValue = value.toString();
064        }
065
066        final int stringLength = stringValue.length();
067
068        final int columnLength = element.getColLength();
069        final char[] formattedValue = new char[columnLength];
070
071        /*
072         * Copy the contents of the original string; for Strings up to ~8-10
073         * characters this loop is consistently faster than String.getChars().
074         * Short Strings are usually the majority of values in fixed-width data
075         * columns.
076         */
077        final int numCharacters = Math.min(stringLength, columnLength);
078        if (numCharacters < MAX_CHAR_TO_USE_LOOP) {
079            for (int i = 0; i < numCharacters; i++) {
080                formattedValue[i] = stringValue.charAt(i);
081            }
082        } else {
083            stringValue.getChars(0, numCharacters, formattedValue, 0);
084        }
085
086        if (stringLength < columnLength) {
087            // pad if necessary
088            Arrays.fill(formattedValue, stringLength, columnLength, padChar);
089        }
090
091        return formattedValue;
092    }
093
094    @Override
095    protected boolean validateColumnTitle(final String columnTitle) {
096        final Map<String, Integer> columnNameToIndex = (Map<String, Integer>) columnMapping.get(FPConstants.COL_IDX);
097        return columnNameToIndex.keySet().contains(columnTitle);
098    }
099
100    @Override
101    public Writer printFooter() {
102        return this;
103    }
104
105    @Override
106    public Writer printHeader() {
107        return this;
108    }
109
110    /**
111     * @return List of ColumnMetaData objects describing the mapping defined in
112     *         the XML file.
113     */
114    private List<ColumnMetaData> getColumnMetaData() {
115        return (List<ColumnMetaData>) columnMapping.get(FPConstants.DETAIL_ID);
116    }
117
118    private ColumnMetaData getColumnMetaData(final String columnName) {
119        for (final ColumnMetaData element : getColumnMetaData()) {
120            if (element.getColName().equals(columnName)) {
121                return element;
122            }
123        }
124
125        throw new IllegalArgumentException("Column \"" + columnName + "\" unknown");
126    }
127}
