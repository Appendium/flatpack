001package net.sf.flatpack.writer;
002
003import java.io.IOException;
004import java.math.BigDecimal;
005import java.util.ArrayList;
006import java.util.Iterator;
007import java.util.List;
008import java.util.Map;
009
010import net.sf.flatpack.structure.ColumnMetaData;
011import net.sf.flatpack.util.FPConstants;
012
013/**
014 *
015 * @author Dirk Holmes and Holger Holger Hoffstatte
016 * @since 4.0 public methods are more fluent
017 */
018public class DelimiterWriter extends AbstractWriter {
019    private final char delimiter;
020    private final char qualifier;
021    private List<String> columnTitles = null;
022    private boolean columnTitlesWritten = false;
023
024    protected DelimiterWriter(final Map columnMapping, final java.io.Writer output, final char delimiter, final char qualifier,
025            final WriterOptions options) throws IOException {
026        super(output);
027        this.delimiter = delimiter;
028        this.qualifier = qualifier;
029
030        columnTitles = new ArrayList<String>();
031        final List<ColumnMetaData> columns = (List<ColumnMetaData>) columnMapping.get(FPConstants.DETAIL_ID);
032        for (final ColumnMetaData cmd : columns) {
033            columnTitles.add(cmd.getColName());
034        }
035        // write the column headers
036        if (options.isAutoPrintHeader()) {
037            printHeader();
038        }
039    }
040
041    protected void writeWithDelimiter(final Object value) throws IOException {
042        this.write(value);
043        this.write(delimiter);
044    }
045
046    @Override
047    protected void write(final Object value) throws IOException {
048        String stringValue = "";
049
050        if (value != null) {
051            // TODO DO: format the value
052            if (value instanceof BigDecimal) {
053                final BigDecimal bd = (BigDecimal) value;
054                stringValue = bd.signum() == 0 ? "0" : bd.toPlainString();
055            } else {
056                stringValue = value.toString();
057            }
058        }
059
060        final boolean foundQualifier = qualifier != FPConstants.NO_QUALIFIER && stringValue.indexOf(qualifier) != -1;
061        final boolean needsQuoting = stringValue.indexOf(delimiter) != -1 //
062                || foundQualifier //
063                || stringValue.indexOf('\n') != -1;
064        // || stringValue.split("\r\n|\r|\n").length > 1;
065
066        if (needsQuoting) {
067            super.write(qualifier);
068        }
069
070        if (foundQualifier) {
071            final char[] chars = stringValue.toCharArray();
072            for (final char character : chars) {
073                if (character == qualifier) {
074                    super.write(qualifier);
075                }
076                super.write(character);
077            }
078        } else {
079            super.write(stringValue);
080        }
081
082        if (needsQuoting) {
083            super.write(qualifier);
084        }
085    }
086
087    protected void writeColumnTitles() throws IOException {
088        final Iterator<String> titleIter = columnTitles.iterator();
089        while (titleIter.hasNext()) {
090            final String title = titleIter.next();
091
092            if (titleIter.hasNext()) {
093                this.writeWithDelimiter(title);
094            } else {
095                this.write(title);
096            }
097        }
098    }
099
100    protected void writeRow() throws IOException {
101        final Iterator<String> titlesIter = columnTitles.iterator();
102        while (titlesIter.hasNext()) {
103            final String columnTitle = titlesIter.next();
104            if (getRowMap() != null) {
105                if (titlesIter.hasNext()) {
106                    writeWithDelimiter(getRowMap().get(columnTitle));
107                } else {
108                    write(getRowMap().get(columnTitle));
109                }
110            }
111        }
112    }
113
114    @Override
115    public final Writer nextRecord() throws IOException {
116        this.writeRow();
117
118        return super.nextRecord();
119    }
120
121    @Override
122    public Writer printFooter() throws IOException {
123        // TODO DO: implement footer handling
124        return this;
125    }
126
127    @Override
128    public Writer printHeader() throws IOException {
129        if (!columnTitlesWritten) {
130            this.writeColumnTitles();
131            columnTitlesWritten = true;
132            return super.nextRecord();
133        }
134        return this;
135    }
136
137    @Override
138    protected boolean validateColumnTitle(final String columnTitle) {
139        return columnTitles.contains(columnTitle);
140    }
141}
