001package net.sf.flatpack.writer;
002
003import java.io.IOException;
004import java.io.Reader;
005import java.util.ArrayList;
006import java.util.Collections;
007import java.util.HashMap;
008import java.util.Map;
009
010import net.sf.flatpack.InitialisationException;
011import net.sf.flatpack.util.FPConstants;
012import net.sf.flatpack.xml.MapParser;
013
014/**
015 *
016 * @author Dirk Holmes and Holger Holger Hoffstatte
017 */
018public abstract class AbstractWriterFactory implements WriterFactory {
019    private Map mapping;
020
021    protected AbstractWriterFactory() {
022        super();
023
024        mapping = new HashMap<String, Object>();
025        mapping.put(FPConstants.DETAIL_ID, new ArrayList());
026        mapping.put(FPConstants.COL_IDX, new HashMap());
027    }
028
029    protected AbstractWriterFactory(final Map<String, ?> mapping) {
030        super();
031        this.mapping = mapping;
032    }
033
034    protected AbstractWriterFactory(final Reader mappingSrc) throws IOException {
035        this();
036
037        try {
038            mapping = MapParser.parse(mappingSrc, null);
039        } catch (final Exception jde) {
040            throw new InitialisationException(jde);
041        }
042    }
043
044    protected Map<String, Object> getColumnMapping() {
045        // TODO DO: return deep mutable clone here or better: make the Map a
046        // first class
047        // citizen of the library
048        return Collections.unmodifiableMap(mapping);
049    }
050
051    @Override
052    public abstract Writer createWriter(java.io.Writer out) throws IOException;
053}
