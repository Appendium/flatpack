001package net.sf.flatpack.writer;

003import java.io.IOException;
004
005/**
006 *
007 * @author Dirk Holmes and Holger Holger Hoffstatte

009public interface Writer extends AutoCloseable {

011    Writer printHeader() throws IOException;
012

014    Writer printFooter() throws IOException;
015

017
018    Writer nextRecord() throws IOException;
019
020    Writer flush() throws IOException;
021

024}
