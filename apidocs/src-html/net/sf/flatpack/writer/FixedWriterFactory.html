001package net.sf.flatpack.writer;
002
003import java.io.IOException;
004import java.io.Reader;
005import java.util.Map;
006
007/**
008 *
009 * @author Dirk Holmes and Holger Holger Hoffstatte
010 */
011public class FixedWriterFactory extends AbstractWriterFactory {
012    public static final char DEFAULT_PADDING_CHARACTER = ' ';
013
014    private final char pad;
015
016    public FixedWriterFactory(final Map mapping) {
017        super(mapping);
018        this.pad = DEFAULT_PADDING_CHARACTER;
019    }
020
021    public FixedWriterFactory(final Reader mappingSrc) throws IOException {
022        this(mappingSrc, DEFAULT_PADDING_CHARACTER);
023    }
024
025    public FixedWriterFactory(final Reader mappingSrc, final char fillChar) throws IOException {
026        super(mappingSrc);
027        this.pad = fillChar;
028    }
029
030    @Override
031    public Writer createWriter(final java.io.Writer output) throws IOException {
032        return new FixedLengthWriter(this.getColumnMapping(), output, pad);
033    }
034}
