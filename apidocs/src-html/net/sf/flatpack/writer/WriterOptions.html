001package net.sf.flatpack.writer;
002
003/**
004 * Defines options for various Writer behaviours
005 *
006 * @author Paul Zepernick
007 */
008public class WriterOptions {
009
010    private boolean autoPrintHeader = true;
011    private String lineSeparator = System.lineSeparator();
012
013    /**
014     * Returns a DelimiterWriterOptions instance
015     *
016     * @return DelimiterWriterOptions
017     */
018    public static WriterOptions getInstance() {
019        return new WriterOptions();
020    }
021
022    /**
023     * @return the noColumnMappings
024     */
025    public boolean isAutoPrintHeader() {
026        return autoPrintHeader;
027    }
028
029    /**
030     * When this is set to true, the addRecordEntry(column, value) will throw an exception.  You
031     * must use addRecordEntry(value).
032     *
033     * @param autoPrintHeader the autoPrintHeader to set
034     */
035    public WriterOptions autoPrintHeader(final boolean autoPrintHeader) {
036        this.autoPrintHeader = autoPrintHeader;
037        return this;
038    }
039
040    /**
041     * Get the current line separator. Default is the system line separator.
042     * @return
043     */
044    public String getLineSeparator() {
045        return lineSeparator;
046    }
047
048    /**
049     * Set the line leperator.
050     * @param lineSeparator the line separator
051     * @return
052     */
053    public WriterOptions setLineSeparator(String lineSeparator) {
054        this.lineSeparator = lineSeparator;
055        return this;
056    }
057}
