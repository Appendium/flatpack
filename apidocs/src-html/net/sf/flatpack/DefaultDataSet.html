103    /*
104     * (non-Javadoc)
105     *
106     * @see net.sf.flatpack.DataSet#getColumns()
107     */
108    @Override
109    public String[] getColumns() {
110        String[] array = null;
111
112        if (/* columnMD != null || */metaData != null) {
113            final List<ColumnMetaData> cmds = metaData.getColumnsNames();
114            array = new String[cmds.size()];
115            for (int i = 0; i < cmds.size(); i++) {
116                final ColumnMetaData column = cmds.get(i);
117                array[i] = column.getColName();
118            }
119        }
120
121        return array;
122    }

307    /*

309     *
310     * @see net.sf.flatpack.DataSet#isAnError(int)

312    @Override
313    public boolean isAnError(final int lineNo) {
314        for (int i = 0; i < errors.size(); i++) {
315            if (errors.get(i).getLineNo() == lineNo && errors.get(i).getErrorLevel() > 1) {
316                return true;
317            }
318        }
319        return false;
320    }

322    /*
323     * (non-Javadoc)
324     *
325     * @see net.sf.flatpack.DataSet#next()
326     */
327    @Override
328    public boolean next() {
329        if (pointer < rows.size() && pointer + 1 != rows.size()) {
330            pointer++;
331            currentRecord = new RowRecord(rows.get(pointer), metaData, parser.isColumnNamesCaseSensitive(), pzConvertProps, strictNumericParse,
332                    upperCase, lowerCase, parser.isNullEmptyStrings());
333            return true;
334        }
335        currentRecord = null;
336        return false;
337    }

349    @Override
350    public void orderRows(final OrderBy ob) {
351        if (ob != null) {
352            ob.setMetaData(getMetaData());
353            ob.setParser(parser);
354            Collections.sort(rows, ob);
355            goTop();
356        }
357    }

363     */
364    @Override
365    public boolean previous() {
366        if (pointer <= 0) {
367            currentRecord = null;
368            return false;
369        }
370        pointer--;
371        currentRecord = new RowRecord(rows.get(pointer), metaData, parser.isColumnNamesCaseSensitive(), pzConvertProps, strictNumericParse, upperCase,
372                lowerCase, parser.isNullEmptyStrings());
373        return true;
374    }

410    /**
411     * Sets the absolute position of the record pointer
412     *
413     * @param localPointer
414     *            - int
415     * @exception IndexOutOfBoundsException
416     */
417    @Override
418    public void absolute(final int localPointer) {
419        if (localPointer < 0 || localPointer >= rows.size()) {
420            throw new IndexOutOfBoundsException("INVALID POINTER LOCATION: " + localPointer);
421        }
422
423        pointer = localPointer;
424        currentRecord = new RowRecord(rows.get(pointer), metaData, parser.isColumnNamesCaseSensitive(), pzConvertProps, strictNumericParse, upperCase,
425                lowerCase, parser.isNullEmptyStrings());
426    }

493    @Override
494    public String toString() {
495        final StringBuilder buf = new StringBuilder();
496        buf.append("Errors:").append(errors.size()).append(NEW_LINE);
497        buf.append("Rows:").append(rows.size()).append(NEW_LINE);
498        buf.append("Position:").append(pointer).append(NEW_LINE);
499        buf.append("Conversion Props:").append(pzConvertProps).append(NEW_LINE);
500        buf.append("MetaData:").append(metaData).append(NEW_LINE);
501        return buf.toString();
502    }

504    @Override
505    public boolean contains(final String column) {
506        if (pointer == -1) {
507            throw new IndexOutOfBoundsException("dataset on invalid row. need to call next()");
508        }
509        return currentRecord.contains(column);
510    }

512    /**
513     * @throws FPInvalidUsageException
514     *             Parser.isFlagEmptyRows() must be set to true before using
515     *             this
516     * @throws FPException
517     *             if cursor is on an invalid row
518     */
519    @Override
520    public boolean isRowEmpty() {
521        if (!parser.isFlagEmptyRows()) {
522            // flag empty rows needs to be set for this functionality
523            // throw an exception
524            throw new FPInvalidUsageException("Parser.isFlagEmptyRows(true) must be set before using isRowEmpty()");
525        }
526
527        if (pointer < 0) {
528            throw new FPException("Cursor on invalid row..  Make sure next() is called and returns true");
529        }
530
531        return rows.get(pointer).isEmpty();
532    }

545        }
546
547        if (pointer < 0) {
548            throw new FPException("Cursor on invalid row.. Make sure next() is called and returns true");
549        }
550
551        return rows.get(pointer).getRawData();
552    }
