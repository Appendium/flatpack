033package net.sf.flatpack;
034
035import java.io.File;
036import java.io.FileNotFoundException;
037import java.io.FileReader;
038import java.io.IOException;
039import java.io.InputStream;
040import java.io.InputStreamReader;
041import java.io.Reader;
042
043import net.sf.flatpack.xml.MapParser;
044
045
046/**
047 * @author xhensevb
048 *
049 */
050public class DelimiterParser extends AbstractDelimiterParser {
051    private InputStream pzmapXMLStream = null;
052    private File pzmapXML = null;
053    private Reader pzmapReader;
054
055    // this InputStream and file can be removed after support for
056    // file and inputstream is removed from the parserfactory. The
057    // methods have been deprecated..pz
058    private InputStream dataSourceStream = null;
059    private File dataSource = null;
060
061    public DelimiterParser(final File pzmapXML, final File dataSource, final char delimiter, final char qualifier, final boolean ignoreFirstRecord) {
062        super(null, delimiter, qualifier, ignoreFirstRecord);
063        this.pzmapXML = pzmapXML;
064        this.dataSource = dataSource;
065    }
066
067    public DelimiterParser(final InputStream pzmapXMLStream, final InputStream dataSourceStream, final char delimiter, final char qualifier,
068            final boolean ignoreFirstRecord) {
069        super(null, delimiter, qualifier, ignoreFirstRecord);
070        this.pzmapXMLStream = pzmapXMLStream;
071        this.dataSourceStream = dataSourceStream;
072    }
073
074    public DelimiterParser(final File dataSource, final char delimiter, final char qualifier, final boolean ignoreFirstRecord) {
075        super(null, delimiter, qualifier, ignoreFirstRecord);
076        this.dataSource = dataSource;
077    }
078
079    public DelimiterParser(final InputStream dataSourceStream, final char delimiter, final char qualifier, final boolean ignoreFirstRecord) {
080        super(null, delimiter, qualifier, ignoreFirstRecord);
081        this.dataSourceStream = dataSourceStream;
082    }
083
084    public DelimiterParser(final Reader dataSourceReader, final char delimiter, final char qualifier, final boolean ignoreFirstRecord) {
085        super(dataSourceReader, delimiter, qualifier, ignoreFirstRecord);
086    }
087
088    public DelimiterParser(final Reader dataSourceReader, final Reader pzmapReader, final char delimiter, final char qualifier,
089            final boolean ignoreFirstRecord) {
090        super(dataSourceReader, delimiter, qualifier, ignoreFirstRecord);
091        this.pzmapReader = pzmapReader;
092    }
093
094    @Override
095    protected void init() {
096        try {
097            // check to see if the user is using a File or InputStream. This is
098            // here for backwards compatibility
099            initStreamOrSource();
100
101            boolean closeMapReader = false;
102            if (pzmapXML != null) {
103                this.pzmapReader = new FileReader(pzmapXML);
104                closeMapReader = true;
105            } else if (pzmapXMLStream != null) {
106                this.pzmapReader = new InputStreamReader(pzmapXMLStream);
107                closeMapReader = true;
108            }
109
110            if (this.pzmapReader != null) {
111                try {
112                    setPzMetaData(MapParser.parseMap(this.pzmapReader, this));
113                } finally {
114                    if (closeMapReader) {
115                        // only close the reader if it is one we created
116                        // otherwise we will let the user handle it
117                        this.pzmapReader.close();
118                    }
119                }
120            }
121
122            setInitialised(true);
123        } catch (final IOException e) {
124            throw new InitialisationException(e);
125        } catch (final Exception e) {
126            throw new InitialisationException(e);
127        }
128    }
129
130    private void initStreamOrSource() throws FileNotFoundException {
131        if (dataSourceStream != null) {
132            final Reader r = new InputStreamReader(dataSourceStream);
133            setDataSourceReader(r);
134            addToCloseReaderList(r);
135        } else if (dataSource != null) {
136            final Reader r = new FileReader(dataSource);
137            setDataSourceReader(r);
138            addToCloseReaderList(r);
139        }
140    }
141
142    @Override
143    protected boolean shouldCreateMDFromFile() {
144        return pzmapReader == null;
145    }
146}
