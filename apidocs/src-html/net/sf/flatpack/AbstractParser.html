162    @Override
163    public final DataSet parse() {
164        if (!initialised) {
165            init();
166        }
167        return doParse();
168    }
169
170    @Override
171    public final StreamingDataSet parseAsStream() {
172        return new StreamingRecord(parse());
173    }
174
175    @Override
176    public final Stream<Record> stream() {
177        return new StreamingRecord(parse()).stream();
178    }
179
180    protected abstract DataSet doParse();
181
182    protected abstract void init();
183
184    protected void closeReaders() throws IOException {
185        if (readersToClose != null) {
186            for (final Reader r : readersToClose) {
187                r.close();
188            }
189        }
190    }
191
192    // adds a reader to the close list. the list will be processed after parsing
193    // is completed.
194    protected void addToCloseReaderList(final Reader r) {
195        if (readersToClose == null) {
196            readersToClose = new ArrayList<Reader>();
197        }
198        readersToClose.add(r);
199    }
200
201    protected void addToMetaData(final List<ColumnMetaData> columns) {
202        if (pzMetaData == null) {
203            pzMetaData = new MetaData(columns, ParserUtils.buidColumnIndexMap(columns, this));
204        } else {
205            pzMetaData.setColumnsNames(columns);
206            pzMetaData.setColumnIndexMap(ParserUtils.buidColumnIndexMap(columns, this));
207        }
208    }
209

226    /**
227     * Adds a new error to this DataSet. These can be collected, and retrieved
228     * after processing
229     *
230     * @param ds
231     *            the dataset
232     * @param errorDesc
233     *            String description of error
234     * @param lineNo
235     *            line number error occurred on
236     * @param errorLevel
237     *            errorLevel 1,2,3 1=warning 2=error 3= severe error
238     */
239    protected void addError(final DefaultDataSet ds, final String errorDesc, final int lineNo, final int errorLevel) {
240        addError(ds, errorDesc, lineNo, errorLevel, null);
241    }
242
243    /**
244     * Adds a new error to this DataSet. These can be collected, and retrieved
245     * after processing
246     *
247     * @param errorDesc
248     *            String description of error
249     * @param lineNo
250     *            line number error occurred on
251     * @param errorLevel
252     *            errorLevel 1,2,3 1=warning 2=error 3= severe error'
253     * @param lineData
254     *            Data of the line which failed the parse
255     */
256    protected void addError(final DefaultDataSet ds, final String errorDesc, final int lineNo, final int errorLevel, final String lineData) {
257        if (errorLevel == 1 && isIgnoreParseWarnings()) {
258            // user has selected to not log warnings in the parser
259            return;
260        }
261        ds.addError(new DataError(errorDesc, lineNo, errorLevel, lineData));
262    }
