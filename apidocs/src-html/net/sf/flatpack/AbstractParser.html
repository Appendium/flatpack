107    protected void initStreamOrSource(InputStream dataSourceStream, File dataSource) throws FileNotFoundException {
108        if (dataSourceStream != null) {
109            final Reader r = new InputStreamReader(dataSourceStream);
110            setDataSourceReader(r);
111            addToCloseReaderList(r);
112        } else if (dataSource != null) {
113            final Reader r = new FileReader(dataSource);
114            setDataSourceReader(r);
115            addToCloseReaderList(r);
116        }
117    }

173    @Override
174    public final DataSet parse() {
175        if (!initialised) {
176            init();

181    @Override

191    protected abstract DataSet doParse();
192
193    protected abstract void init();
194
195    protected void closeReaders() throws IOException {
196        if (readersToClose != null) {
197            for (final Reader r : readersToClose) {
198                r.close();
199            }
200        }
201    }
202
203    // adds a reader to the close list. the list will be processed after parsing
204    // is completed.
205    protected void addToCloseReaderList(final Reader r) {
206        if (readersToClose == null) {
207            readersToClose = new ArrayList<Reader>();
208        }
209        readersToClose.add(r);
210    }
211
212    protected void addToMetaData(final List<ColumnMetaData> columns) {
213        if (pzMetaData == null) {
214            pzMetaData = new MetaData(columns, ParserUtils.buidColumnIndexMap(columns, this));
215        } else {
216            pzMetaData.setColumnsNames(columns);
217            pzMetaData.setColumnIndexMap(ParserUtils.buidColumnIndexMap(columns, this));
218        }
219    }

237    /**
238     * Adds a new error to this DataSet. These can be collected, and retrieved
239     * after processing
240     *
241     * @param ds
242     *            the dataset
243     * @param errorDesc
244     *            String description of error
245     * @param lineNo
246     *            line number error occurred on
247     * @param errorLevel
248     *            errorLevel 1,2,3 1=warning 2=error 3= severe error
249     */
250    protected void addError(final DefaultDataSet ds, final String errorDesc, final int lineNo, final int errorLevel) {
251        addError(ds, errorDesc, lineNo, errorLevel, null);
252    }
253
254    /**
255     * Adds a new error to this DataSet. These can be collected, and retrieved
256     * after processing
257     *
258     * @param errorDesc
259     *            String description of error
260     * @param lineNo
261     *            line number error occurred on
262     * @param errorLevel
263     *            errorLevel 1,2,3 1=warning 2=error 3= severe error'
264     * @param lineData
265     *            Data of the line which failed the parse
266     */
267    protected void addError(final DefaultDataSet ds, final String errorDesc, final int lineNo, final int errorLevel, final String lineData) {
268        if (errorLevel == 1 && isIgnoreParseWarnings()) {
269            // user has selected to not log warnings in the parser
270            return;
271        }
272        ds.addError(new DataError(errorDesc, lineNo, errorLevel, lineData));
273    }
