033package net.sf.flatpack.brparse;
034
035import java.io.BufferedReader;
036import java.io.File;
037import java.io.IOException;
038import java.io.InputStream;
039import java.io.Reader;
040import java.util.List;
041import java.util.Map;
042
043import net.sf.flatpack.DataSet;
044import net.sf.flatpack.DefaultDataSet;
045import net.sf.flatpack.FixedLengthParser;
046import net.sf.flatpack.structure.ColumnMetaData;
047import net.sf.flatpack.structure.Row;
048import net.sf.flatpack.util.FPConstants;
049import net.sf.flatpack.util.FixedWidthParserUtils;
050import net.sf.flatpack.util.ParserUtils;
051
052import org.slf4j.Logger;
053import org.slf4j.LoggerFactory;
054
055/**
056 *
057 *
058 * @author Paul Zepernick
059 */
060public class BuffReaderFixedParser extends FixedLengthParser implements InterfaceBuffReaderParse {
061    private BufferedReader br = null;
062
063    private int lineCount = 0;
064
065    private Map recordLengths = null;
066
067    private static final Logger LOGGER = LoggerFactory.getLogger(BuffReaderFixedParser.class);
068
069    /**
070     *
071     * @param pzmapXMLStream
072     * @param dataSourceStream
073     */
074    public BuffReaderFixedParser(final InputStream pzmapXMLStream, final InputStream dataSourceStream) {
075        super(pzmapXMLStream, dataSourceStream);
076    }
077
078    /**
079     *
080     * @param pzmapXML
081     * @param dataSource
082     */
083    public BuffReaderFixedParser(final File pzmapXML, final File dataSource) {
084        super(pzmapXML, dataSource);
085    }
086
087    /**
088     *
089     *
090     * @param pzmapXML
091     * @param dataSource
092     */
093    public BuffReaderFixedParser(final Reader pzmapXML, final Reader dataSource) {
094        super(pzmapXML, dataSource);
095    }
096
097    protected BuffReaderFixedParser(final Reader dataSourceReader, final String dataDefinition) {
098        super(dataSourceReader, dataDefinition);
099    }
100
101    @Override
102    protected DataSet doParse() {
103        final DataSet ds = new BuffReaderDataSet(getPzMetaData(), this);
104        lineCount = 0;
105        recordLengths = ParserUtils.calculateRecordLengths(getPzMetaData());
106        try {
107            // gather the conversion properties
108            ds.setPZConvertProps(ParserUtils.loadConvertProperties());
109
110            br = new BufferedReader(getDataSourceReader());
111
112            return ds;
113
114        } catch (final IOException ex) {
115            LOGGER.error("error accessing/creating inputstream", ex);
116        }
117
118        return null;
119    }
120
121    /**
122     * Reads in the next record on the file and return a row
123     *
124     * @param ds
125     * @return Row
126     */
127    @Override
128    public Row buildRow(final DefaultDataSet ds) {
129        String line = null;
130        try {
131            while ((line = br.readLine()) != null) {
132                lineCount++;
133                // empty line skip past it
134                if (line.trim().length() == 0) {
135                    continue;
136                }
137
138                final String mdkey = FixedWidthParserUtils.getCMDKey(getPzMetaData(), line);
139
140                final Row row = new Row();
141                row.setRowNumber(lineCount);
142                row.setMdkey(mdkey.equals(FPConstants.DETAIL_ID) ? null : mdkey);
143
144                final List<ColumnMetaData> cmds = ParserUtils.getColumnMetaData(mdkey, getPzMetaData());
145
146                final int recordLength = ((Integer) recordLengths.get(mdkey)).intValue();
147
148                if (line.length() > recordLength) {
149                    // Incorrect record length on line log the error. Line will not
150                    // be included in the
151                    // dataset
152                    if (isIgnoreExtraColumns()) {
153                        addError(ds, "TRUNCATED LINE TO CORRECT LENGTH", lineCount, 1);
154                        // user has chosen to ignore the fact that we have too many bytes in the fixed
155                        // width file. Truncate the line to the correct length
156                        row.addColumn(FixedWidthParserUtils.splitFixedText(cmds, line.substring(0, recordLength), isPreserveLeadingWhitespace(), isPreserveTrailingWhitespace()));
157                    } else {
158                        addError(ds, "LINE TOO LONG. LINE IS " + line.length() + " LONG. SHOULD BE " + recordLength, lineCount, 2,
159                                isStoreRawDataToDataError() ? line : null);
160                        continue;
161                    }
162                } else if (line.length() < recordLength) {
163                    if (isHandlingShortLines()) {
164                        // log a warning
165                        addError(ds, "PADDED LINE TO CORRECT RECORD LENGTH", lineCount, 1);
166                        // We can pad this line out
167                        row.addColumn(FixedWidthParserUtils.splitFixedText(cmds, line + ParserUtils.padding(recordLength - line.length(), ' '), isPreserveLeadingWhitespace(), isPreserveTrailingWhitespace()));
168
169                    } else {
170                        addError(ds, "LINE TOO SHORT. LINE IS " + line.length() + " LONG. SHOULD BE " + recordLength, lineCount, 2,
171                                isStoreRawDataToDataError() ? line : null);
172                        continue;
173                    }
174                } else {
175                    row.addColumn(FixedWidthParserUtils.splitFixedText(cmds, line, isPreserveLeadingWhitespace(), isPreserveTrailingWhitespace()));
176                }
177
178                if (isFlagEmptyRows()) {
179                    // user has elected to have the parser flag rows that are empty
180                    row.setEmpty(ParserUtils.isListElementsEmpty(row.getCols()));
181                }
182                if (isStoreRawDataToDataSet()) {
183                    // user told the parser to keep a copy of the raw data in the row
184                    // WARNING potential for high memory usage here
185                    row.setRawData(line);
186                }
187
188                return row;
189            }
190
191        } catch (final IOException e) {
192            throw new RuntimeException("Error Fetching Record From File...", e);
193        }
194
195        return null;
196    }
197
198    /**
199     * Closes out the file readers
200     *
201     *@throws IOException
202     */
203    public void close() throws IOException {
204        if (br != null) {
205            br.close();
206        }
207    }
208
209    // try to clean up the file handles automatically if
210    // the close was not called
211    @Override
212    protected void finalize() throws Throwable {
213        try {
214            close();
215        } catch (final IOException ex) {
216            LOGGER.warn("Problem trying to auto close file handles...", ex);
217        } finally {
218            super.finalize();
219        }
220    }
221}
