050import org.slf4j.Logger;
051import org.slf4j.LoggerFactory;
052
053public class BuffReaderDelimParser extends DelimiterParser implements InterfaceBuffReaderParse {
054    private BufferedReader br;
055
056    private boolean processedFirst = false;
057
058    private static final Logger LOGGER = LoggerFactory.getLogger(BuffReaderDelimParser.class);
059
060    public BuffReaderDelimParser(final File pzmapXML, final File dataSource, final char delimiter, final char qualifier,
061            final boolean ignoreFirstRecord) {
062        super(pzmapXML, dataSource, delimiter, qualifier, ignoreFirstRecord);
063    }
064
065    public BuffReaderDelimParser(final InputStream pzmapXMLStream, final InputStream dataSourceStream, final char delimiter, final char qualifier,
066            final boolean ignoreFirstRecord) {
067        super(pzmapXMLStream, dataSourceStream, delimiter, qualifier, ignoreFirstRecord);
068    }
069
070    public BuffReaderDelimParser(final File dataSource, final char delimiter, final char qualifier, final boolean ignoreFirstRecord) {
071        super(dataSource, delimiter, qualifier, ignoreFirstRecord);
072    }
073
074    public BuffReaderDelimParser(final InputStream dataSourceStream, final char delimiter, final char qualifier, final boolean ignoreFirstRecord) {
075        super(dataSourceStream, delimiter, qualifier, ignoreFirstRecord);
076    }
077
078    public BuffReaderDelimParser(final Reader pzmapXML, final Reader dataSource, final char delimiter, final char qualifier,
079            final boolean ignoreFirstRecord) {
080        super(dataSource, pzmapXML, delimiter, qualifier, ignoreFirstRecord);
081    }
082
083    public BuffReaderDelimParser(final Reader dataSourceStream, final char delimiter, final char qualifier, final boolean ignoreFirstRecord) {
084        super(dataSourceStream, delimiter, qualifier, ignoreFirstRecord);
085    }
086
087    @Override
088    protected DataSet doParse() {
089        final DataSet ds = new BuffReaderDataSet(getPzMetaData(), this);
090        try {
091            // gather the conversion properties
092            ds.setPZConvertProps(ParserUtils.loadConvertProperties());
093
094            br = new BufferedReader(getDataSourceReader());
095
096            return ds;
097
098        } catch (final IOException ex) {
099            LOGGER.error("error accessing/creating inputstream", ex);
100        }
101
102        return null;
103    }
104
105    /**
106     * Reads in the next record on the file and return a row
107     *
108     * @param ds
109     * @return Row
110     */
111    @Override
112    public Row buildRow(final DefaultDataSet ds) {
113        /** loop through each line in the file */
114        while (true) {
115            String line;
116            try {
117                line = fetchNextRecord(br, getQualifier(), getDelimiter());
118            } catch (final IOException e) {
119                throw new RuntimeException("Error Fetching Record From File...", e);
120            }
121
122            if (line == null) {
123                return null;
124            }
125
126            // check to see if the user has elected to skip the first record
127            if (!processedFirst && isIgnoreFirstRecord()) {
128                processedFirst = true;
129                continue;
130            } else if (!processedFirst && shouldCreateMDFromFile()) {
131                processedFirst = true;
132                setPzMetaData(ParserUtils.getPZMetaDataFromFile(line, getDelimiter(), getQualifier(), this, isAddSuffixToDuplicateColumnNames()));
133                ds.setMetaData(getPzMetaData());
134                continue;
135            }
136
137            // TODO
138            // seems like we may want to try doing something like this. I have my reservations because
139            // it is possible that we don't get a "detail" id and this might generate NPE
140            // is it going to create too much overhead to do a null check here as well???
141            List<String> columns = ParserUtils.splitLine(line, getDelimiter(), getQualifier(), FPConstants.SPLITLINE_SIZE_INIT,
142                    isPreserveLeadingWhitespace(), isPreserveTrailingWhitespace());
143            final String mdkey = ParserUtils.getCMDKeyForDelimitedFile(getPzMetaData(), columns);
144            final List<ColumnMetaData> cmds = ParserUtils.getColumnMetaData(mdkey, getPzMetaData());
145            final int columnCount = cmds.size();
146            // DEBUG
147
148            // Incorrect record length on line log the error. Line
149            // will not be included in the dataset
150            if (columns.size() > columnCount) {
151                if (isIgnoreExtraColumns()) {
152                    // user has choosen to ignore the fact that we have too many columns in the data from
153                    // what the mapping has described. sublist the array to remove un-needed columns
154                    columns = columns.subList(0, columnCount);
155                    addError(ds, "TRUNCATED LINE TO CORRECT NUMBER OF COLUMNS", getLineCount(), 1);
156                } else {
157                    // log the error
158                    addError(ds, "TOO MANY COLUMNS WANTED: " + columnCount + " GOT: " + columns.size(), getLineCount(), 2,
159                            isStoreRawDataToDataError() ? line : null);
160                    continue;
161                }
162            } else if (columns.size() < columnCount) {
163                if (isHandlingShortLines()) {
164                    // We can pad this line out
165                    while (columns.size() < columnCount) {
166                        columns.add("");
167                    }
168
169                    // log a warning
170                    addError(ds, "PADDED LINE TO CORRECT NUMBER OF COLUMNS", getLineCount(), 1);
171
172                } else {
173                    addError(ds, "TOO FEW COLUMNS WANTED: " + columnCount + " GOT: " + columns.size(), getLineCount(), 2,
174                            isStoreRawDataToDataError() ? line : null);
175                    continue;
176                }
177            }
178
179            final Row row = new Row();
180            row.setMdkey(mdkey.equals(FPConstants.DETAIL_ID) ? null : mdkey); // try
181            // to limit the memory use
182            row.setCols(columns);
183            row.setRowNumber(getLineCount());
184
185            if (isFlagEmptyRows()) {
186                // user has elected to have the parser flag rows that are empty
187                row.setEmpty(ParserUtils.isListElementsEmpty(columns));
188            }
189            if (isStoreRawDataToDataSet()) {
190                // user told the parser to keep a copy of the raw data in the row
191                // WARNING potential for high memory usage here
192                row.setRawData(line);
193            }
194
195            return row;
196        }
197    }
198
199    /**
200     * Closes out the file readers
201     *
202     *@throws IOException
203     */
204    public void close() throws IOException {
205        if (br != null) {
206            br.close();
207            br = null;
208        }
209    }
210
211    // try to clean up the file handles automatically if
212    // the close was not called
213    @Override
214    protected void finalize() throws Throwable {
215        try {
216            close();
217        } catch (final IOException ex) {
218            LOGGER.warn("Problem trying to auto close file handles...", ex);
219        } finally {
220            super.finalize();
221        }
222    }
