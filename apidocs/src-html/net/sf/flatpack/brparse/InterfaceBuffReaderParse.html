001package net.sf.flatpack.brparse;
002
003import net.sf.flatpack.DefaultDataSet;
004import net.sf.flatpack.structure.Row;
005
006/**
007 * All buffered reader parsers should implement this interface and provide
008 * an implementation for the buildRow
009 *
010 * @author Paul Zepernick
011 */
012public interface InterfaceBuffReaderParse extends AutoCloseable {
013
014    /**
015     * Builds a row into the DataSet using the current record from the File
016     *
017     * @param ds
018     * @return Row object
019     */
020    public Row buildRow(final DefaultDataSet ds);
021
022}
