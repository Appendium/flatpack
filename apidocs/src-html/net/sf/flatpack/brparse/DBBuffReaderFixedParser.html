001package net.sf.flatpack.brparse;
002
003import java.io.FileNotFoundException;
004import java.io.Reader;
005import java.sql.Connection;
006import java.sql.SQLException;
007import java.util.List;
008
009import net.sf.flatpack.InitialisationException;
010import net.sf.flatpack.structure.ColumnMetaData;
011import net.sf.flatpack.util.ParserUtils;
012
013public class DBBuffReaderFixedParser extends BuffReaderFixedParser implements InterfaceBuffReaderParse {
014    private final Connection con;
015
016    public DBBuffReaderFixedParser(final Connection con, final Reader dataSourceReader, final String dataDefinition) {
017        super(dataSourceReader, dataDefinition);
018        this.con = con;
019    }
020
021    @Override
022    protected void init() {
023        try {
024
025            final List<ColumnMetaData> cmds = ParserUtils.buildMDFromSQLTable(con, getDataDefinition(), this);
026            addToMetaData(cmds);
027
028            if (cmds.isEmpty()) {
029                throw new FileNotFoundException("DATA DEFINITION CAN NOT BE FOUND IN THE DATABASE " + getDataDefinition());
030            }
031            setInitialised(true);
032        } catch (final SQLException e) {
033            throw new InitialisationException(e);
034        } catch (final FileNotFoundException e) {
035            throw new InitialisationException(e);
036        }
037    }
038
039    protected boolean shouldCreateMDFromFile() {
040        // The MetaData should always be pulled from the DB for this implementation
041        return false;
042    }
043}
