033package net.sf.flatpack.brparse;
034
035import java.io.File;
036import java.io.InputStream;
037import java.io.Reader;
038import java.sql.Connection;
039
040import net.sf.flatpack.Parser;
041import net.sf.flatpack.ParserFactory;
042
043/**
044 * Provides a PZParser which obtains records directly from
045 * a BufferedReader as an alternative to reading the
046 * entire file into memory.
047 *
048 * Database column mappings are not supported by this factory
049 * at the present time.  This class is meant to mimic the LargeDataSet
050 * class of pre 3.0 versions, which did not support database mappings
051 * either.  A RuntimeExcpetion will be thrown if trying to obtain a parser
052 * for a database map.
053 *
054 * @author Paul Zepernick
055 */
056public class BuffReaderParseFactory implements ParserFactory {
057    private static final BuffReaderParseFactory INSTANCE = new BuffReaderParseFactory();
058
059    public static ParserFactory getInstance() {
060        return INSTANCE;
061    }
062
063    /**
064     * Not supported at this time.
065     */
066    @Override
067    public Parser newFixedLengthParser(final Connection con, final File dataSource, final String dataDefinition) {
068        throw new UnsupportedOperationException("Not supported...");
069    }
070
071    /**
072     * Not supported at this time.
073     */
074    @Override
075    public Parser newFixedLengthParser(final Connection con, final InputStream dataSourceStream, final String dataDefinition) {
076        throw new UnsupportedOperationException("Not supported...");
077    }
078
079    /*
080     * (non-Javadoc)
081     *
082     * @see net.sf.flatpack.PZParserFactory#newParser(java.io.File,
083     *      java.io.File)
084     */
085    @Override
086    public Parser newFixedLengthParser(final File pzmapXML, final File dataSource) {
087        return new BuffReaderFixedParser(pzmapXML, dataSource);
088    }
089
090    @Override
091    public Parser newFixedLengthParser(final Connection con, final Reader dataSource, final String dataDefinition) {
092        return new DBBuffReaderFixedParser(con, dataSource, dataDefinition);
093    }
094
095    @Override
096    public Parser newFixedLengthParser(final Reader pzmapXMLStream, final Reader dataSource) {
097        return new BuffReaderFixedParser(pzmapXMLStream, dataSource);
098    }
099
100    /*
101     * (non-Javadoc)
102     *
103     * @see net.sf.flatpack.PZParserFactory#newParser(java.io.InputStream,
104     *      java.io.InputStream)
105     */
106    @Override
107    public Parser newFixedLengthParser(final InputStream pzmapXMLStream, final InputStream dataSourceStream) {
108        return new BuffReaderFixedParser(pzmapXMLStream, dataSourceStream);
109    }
110
111    @Override
112    public Parser newDelimitedParser(final Connection con, final InputStream dataSourceStream, final String dataDefinition, final char delimiter,
113            final char qualifier, final boolean ignoreFirstRecord) {
114        throw new UnsupportedOperationException("Not supported.  Use 'Reader' Constructor Instead Of InputStream.");
115    }
116
117    /*
118     * (non-Javadoc)
119     *
120     * @see net.sf.flatpack.PZParserFactory#newParser(java.io.File,
121     *      java.io.File, char, char, boolean)
122     */
123    @Override
124    public Parser newDelimitedParser(final File pzmapXML, final File dataSource, final char delimiter, final char qualifier,
125            final boolean ignoreFirstRecord) {
126        return new BuffReaderDelimParser(pzmapXML, dataSource, delimiter, qualifier, ignoreFirstRecord);
127    }
128
129    /*
130     * (non-Javadoc)
131     *
132     * @see net.sf.flatpack.PZParserFactory#newParser(java.io.InputStream,
133     *      java.io.InputStream, char, char, boolean)
134     */
135    @Override
136    public Parser newDelimitedParser(final InputStream pzmapXMLStream, final InputStream dataSourceStream, final char delimiter,
137            final char qualifier, final boolean ignoreFirstRecord) {
138        return new BuffReaderDelimParser(pzmapXMLStream, dataSourceStream, delimiter, qualifier, ignoreFirstRecord);
139    }
140
141    /*
142     * (non-Javadoc)
143     *
144     * @see net.sf.flatpack.PZParserFactory#newParser(java.io.File, char,
145     *      char)
146     */
147    @Override
148    public Parser newDelimitedParser(final File dataSource, final char delimiter, final char qualifier) {
149        return new BuffReaderDelimParser(dataSource, delimiter, qualifier, false);
150    }
151
152    /*
153     * (non-Javadoc)
154     *
155     * @see net.sf.flatpack.PZParserFactory#newParser(java.io.InputStream,
156     *      char, char)
157     */
158    @Override
159    public Parser newDelimitedParser(final InputStream dataSourceStream, final char delimiter, final char qualifier) {
160        return new BuffReaderDelimParser(dataSourceStream, delimiter, qualifier, false);
161    }
162
163    /**
164     * Not supported at this time.
165     */
166    @Override
167    public Parser newDelimitedParser(final Connection con, final Reader dataSource, final String dataDefinition, final char delimiter,
168            final char qualifier, final boolean ignoreFirstRecord) {
169        return new DBBuffReaderDelimParser(con, dataSource, dataDefinition, delimiter, qualifier, ignoreFirstRecord);
170    }
171
172    @Override
173    public Parser newDelimitedParser(final Reader dataSource, final char delimiter, final char qualifier) {
174        return new BuffReaderDelimParser(dataSource, delimiter, qualifier, false);
175    }
176
177    @Override
178    public Parser newDelimitedParser(final Reader pzmapXML, final Reader dataSource, final char delimiter, final char qualifier,
179            final boolean ignoreFirstRecord) {
180        return new BuffReaderDelimParser(pzmapXML, dataSource, delimiter, qualifier, ignoreFirstRecord);
181    }
182}
