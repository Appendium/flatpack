033package net.sf.flatpack;
034
035import java.io.FileNotFoundException;
036import java.io.InputStream;
037import java.io.InputStreamReader;
038import java.io.Reader;
039import java.sql.Connection;
040import java.sql.SQLException;
041import java.util.List;
042
043import net.sf.flatpack.structure.ColumnMetaData;
044import net.sf.flatpack.util.ParserUtils;
045
046/**
047 * @author xhensevb
048 * @author zepernick
049 *
050 */
051public class DBDelimiterParser extends AbstractDelimiterParser {
052    private final Connection con;
053
054    private InputStream dataSourceStream;
055
056    public DBDelimiterParser(final Connection con, final InputStream dataSourceStream, final String dataDefinition, final char delimiter,
057            final char qualifier, final boolean ignoreFirstRecord) {
058        super(null, dataDefinition, delimiter, qualifier, ignoreFirstRecord);
059        this.con = con;
060        this.dataSourceStream = dataSourceStream;
061    }
062
063    public DBDelimiterParser(final Connection con, final Reader dataSourceReader, final String dataDefinition, final char delimiter,
064            final char qualifier, final boolean ignoreFirstRecord) {
065        super(dataSourceReader, dataDefinition, delimiter, qualifier, ignoreFirstRecord);
066        this.con = con;
067    }
068
069    @Override
070    protected void init() {
071        try {
072            // check to see if the user is using a InputStream. This is
073            // here for backwards compatibility
074            if (dataSourceStream != null) {
075                final Reader r = new InputStreamReader(dataSourceStream);
076                setDataSourceReader(r);
077                addToCloseReaderList(r);
078            }
079
080            final List<ColumnMetaData> cmds = ParserUtils.buildMDFromSQLTable(con, getDataDefinition(), this);
081            addToMetaData(cmds);
082
083            if (cmds.isEmpty()) {
084                throw new FileNotFoundException("DATA DEFINITION CAN NOT BE FOUND IN THE DATABASE " + getDataDefinition());
085            }
086            setInitialised(true);
087        } catch (final SQLException | FileNotFoundException e) {
088            throw new InitialisationException(e);
089        }
090    }
091
092    @Override
093    protected boolean shouldCreateMDFromFile() {
094        // The MetaData should always be pulled from the DB for this implementation
095        return false;
096    }
097}
