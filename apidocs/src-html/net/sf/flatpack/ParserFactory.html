035import java.io.File;

040/**
041 * Factory definitions for creating a PZParser (delimiter or fixed length). The
042 * creation of a parser will not start the parsing. It should not fail either
043 * (unless DB issues etc).
044 *
045 * @author Benoit Xhenseval
046 */
047public interface ParserFactory {
048    /**
049     * Constructs a new DataSet using the database table file layout method.
050     * This is used for a FIXED LENGTH text file.
051     *
052     * The user is responsible for closing the DB connection.
053     *
054     * @param con -
055     *            Connection to database with DATAFILE and DATASTRUCTURE tables,
056     *            user is responsible for closing it.
057     * @param dataSource -
058     *            Fixed length file to read from
059     * @param dataDefinition -
060     *            Name of dataDefinition in the DATAFILE table DATAFILE_DESC
061     *            column
062     * @deprecated Please use the newFixedLengthParser(Connection, Reader, String).  The File can be
063     *            wrapped in a "new FileReader(File)"
064     * @return PZParser
065     */
066    @Deprecated
067    Parser newFixedLengthParser(Connection con, File dataSource, String dataDefinition);
068
069    /**
070     * Constructs a new DataSet using the database table file layout method.
071     * This is used for a FIXED LENGTH text file.
072     *
073     * The user is responsible for closing the DB connection and InputStream.
074     *
075     * @param con -
076     *            Connection to database with DATAFILE and DATASTRUCTURE tables,
077     *            user is responsible for closing it.
078     * @param dataSourceStream -
079     *            text file datasource InputStream to read from
080     * @param dataDefinition -
081     *            Name of dataDefinition in the DATAFILE table DATAFILE_DESC
082     *            column
083     * @deprecated Please use the newFixedLengthParser(Connection, Reader, String).  The InputStream can be
084     *            wrapped in a "new InputStreamReader(InputStream)"
085     * @return PZParser
086     */
087    @Deprecated
088    Parser newFixedLengthParser(Connection con, InputStream dataSourceStream, String dataDefinition);
089
090    /**
091     * Constructs a new DataSet using the PZMAP XML file layout method. This is
092     * used for a FIXED LENGTH text file.
093     *
094     * @param pzmapXML -
095     *            Reference to the xml file holding the pzmap
096     * @param dataSource -
097     *            Delimited file to read from
098     * @deprecated Please use the newFixedLengthParser(Reader, Reader).  The File can be
099     *            wrapped in a "new FileReader(InputStream)"
100     * @return PZParser
101     */
102    @Deprecated
103    Parser newFixedLengthParser(File pzmapXML, File dataSource);
104
105    /**
106     * New constructor based on InputStream. Constructs a new DataSet using the
107     * PZMAP XML file layout method. This is used for a FIXED LENGTH text file.
108     *
109     * The user is responsible for closing the InputStreams.
110     *
111     * @param pzmapXMLStream -
112     *            Reference to the xml file InputStream holding the pzmap, user
113     *            must close them after use.
114     * @param dataSourceStream -
115     *            Delimited file InputStream to read from, user must close them
116     *            after use.
117     * @deprecated Please use the newFixedLengthParser(Reader, Reader).  The InputStream can be
118     *            wrapped in a "new InputStreamReader(InputStream)"
119     * @return PZParser
120     */
121    @Deprecated
122    Parser newFixedLengthParser(InputStream pzmapXMLStream, InputStream dataSourceStream);
123
124    /**
125     * Constructs a new DataSet using the database table file layout method.
126     * This is used for a FIXED LENGTH text file.
127     *
128     * The user is responsible for closing the DB connection.
129     *
130     * @param con -
131     *            Connection to database with DATAFILE and DATASTRUCTURE tables,
132     *            user is responsible for closing it.
133     * @param dataSource -
134     *            Fixed length file to read from
135     * @param dataDefinition -
136     *            Name of dataDefinition in the DATAFILE table DATAFILE_DESC
137     *            column
138     * @return PZParser
139     */
140    Parser newFixedLengthParser(Connection con, Reader dataSource, String dataDefinition);

143     * New constructor based on Reader. Constructs a new DataSet using the

145     *
146     * The user is responsible for closing the Readers.
147     *
148     * @param pzmapXMLStream -
149     *            Reference to the xml Reader holding the pzmap, user
150     *            must close them after use.
151     * @param dataSource -
152     *            Delimited file Reader to read from, user must close them
153     *            after use.
154     * @return PZParser
155     */
156    Parser newFixedLengthParser(Reader pzmapXMLStream, Reader dataSource);

164    /**
165     * Constructs a new DataSet using the
166     * database table file layout method. This is used for a DELIMITED text
167     * file. esacpe sequence reference: \n newline <br>
168     * \t tab <br>
169     * \b backspace <br>
170     * \r return <br>
171     * \f form feed <br> \\ backslash <br> \' single quote <br> \" double quote
172     *
173     * The user is responsible for closing the DB connection and InputStream.
174     *
175     * @param con -
176     *            Connection to database with DATAFILE and DATASTRUCTURE tables,
177     *            user must close it when done.
178     * @param dataSource -
179     *            text file datasource InputStream to read from, user must close
180     *            it when done.
181     * @param dataDefinition -
182     *            Name of dataDefinition in the DATAFILE table DATAFILE_DESC
183     *            column
184     * @param delimiter -
185     *            Char the file is delimited By
186     * @param qualifier -
187     *            Char text is qualified by
188     * @param ignoreFirstRecord -
189     *            skips the first line that contains data in the file
190     * @deprecated Please use the newDelimitedParser(Connection, Reader, String, char, char, boolean).
191     *            The InputStream can be wrapped in a "new InputStreamReader(InputStream)"
192     * @return PZParser
193     */
194    @Deprecated
195    Parser newDelimitedParser(Connection con, InputStream dataSource, String dataDefinition, char delimiter, char qualifier, boolean ignoreFirstRecord);
196

201     * \t tab <br>
202     * \b backspace <br>

206     * The user is responsible for closing the DB connection and InputStream.
207     *

209     *            Connection to database with DATAFILE and DATASTRUCTURE tables,

211     * @param dataSource -
212     *            text file datasource InputStream to read from, user must close

214     * @param dataDefinition -
