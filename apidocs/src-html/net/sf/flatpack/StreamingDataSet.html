001package net.sf.flatpack;
002
003import java.util.Iterator;
004import java.util.NoSuchElementException;
005import java.util.Optional;
006import java.util.Spliterator;
007import java.util.Spliterators;
008import java.util.stream.Stream;
009import java.util.stream.StreamSupport;
010
011/**
012 * New with jdk8, define stream() methods. You should
013 * start using this.
014 *
015 * @author Benoit Xhenseval
016 * @since 3.4
017 */
018public interface StreamingDataSet extends RecordDataSet {
019    /**
020     * @since 4.0
021     * @return a stream of Records
022     */
023    default Stream<Record> stream() {
024        return StreamSupport
025                .stream(Spliterators.spliteratorUnknownSize(spliterator(), Spliterator.ORDERED | Spliterator.NONNULL | Spliterator.IMMUTABLE), false);
026    }
027
028    /**
029     * @since 4.0
030     * @return a stream of Records
031     */
032    default Stream<Record> parallelStream() {
033        return StreamSupport
034                .stream(Spliterators.spliteratorUnknownSize(spliterator(), Spliterator.ORDERED | Spliterator.NONNULL | Spliterator.IMMUTABLE), true);
035    }
036
037    default Iterator<Record> spliterator() {
038        return new Iterator<Record>() {
039            Optional<Record> nextData = Optional.empty();
040
041            @Override
042            public boolean hasNext() {
043                if (nextData.isPresent()) {
044                    return true;
045                } else {
046                    if (StreamingDataSet.this.next()) {
047                        nextData = getRecord();
048                    } else {
049                        nextData = Optional.empty();
050                    }
051                    return nextData.isPresent();
052                }
053            }
054
055            @Override
056            public Record next() {
057                if (nextData.isPresent() || hasNext()) {
058                    final Record line = nextData.isPresent() ? nextData.get() : null;
059                    nextData = Optional.empty();
060                    return line;
061                } else {
062                    throw new NoSuchElementException();
063                }
064            }
065        };
066    }
067}
