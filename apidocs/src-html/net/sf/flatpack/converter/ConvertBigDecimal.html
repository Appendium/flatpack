001package net.sf.flatpack.converter;
002
003import java.math.BigDecimal;
004
005import net.sf.flatpack.util.ParserUtils;
006
007/**
008 * Returns a BigDecimal
009 * Non numeric chars are removed from the string
010 * before converting
011 *
012 * @author Paul Zepernick
013 */
014public class ConvertBigDecimal implements Converter {
015    /*
016     * (non-Javadoc)
017     *
018     * @see net.sf.flatpack.converter#convertValue(java.lang.String)
019     */
020    @Override
021    public Object convertValue(final String valueToConvert) {
022        return new BigDecimal(ParserUtils.stripNonDoubleChars(valueToConvert));
023    }
024}
