034package net.sf.flatpack.converter;
035
036/**
037 * Exception thrown when a conversion error occurs
038 *
039 * @author Paul Zepernick
040 */
041public class FPConvertException extends RuntimeException {
042    private static final long serialVersionUID = 1L;
043
044    public FPConvertException() {
045        super();
046    }
047
048    /**
049     * @param message
050     */
051    public FPConvertException(final String message) {
052        super(message);
053    }
054
055    /**
056     * @param cause
057     */
058    public FPConvertException(final Throwable cause) {
059        super(cause);
060    }
061
062    /**
063     * @param message
064     * @param cause
065     */
066    public FPConvertException(final String message, final Throwable cause) {
067        super(message, cause);
068    }
069
070}
