034package net.sf.flatpack.converter;
035
036import net.sf.flatpack.util.ParserUtils;
037
038/**
039 * Returns a Double
040 * Non numeric chars are removed from the string
041 * before converting
042 *
043 * @author Paul Zepernick
044 */
045public class ConvertDouble implements Converter {
046
047    /*
048     * (non-Javadoc)
049     *
050     * @see net.sf.flatpack.converter#convertValue(java.lang.String)
051     */
052    @Override
053    public Object convertValue(final String valueToConvert) {
054        return new Double(ParserUtils.stripNonDoubleChars(valueToConvert));
055    }
056
057}
