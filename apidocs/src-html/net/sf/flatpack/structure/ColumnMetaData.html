015package net.sf.flatpack.structure;
016
017/**
018 * @author Paul zepernick
019 *
020 * Holds information about the columns in the data set. This is an improvment
021 * over 1.0.x. This information was being repeated for every row, causing a
022 * spike in memory usage
023 *
024 * @version 2.0
025 */
026public class ColumnMetaData {
027
028    /** Column Name */
029    private String colName = null;
030
031    /** column length */
032    private int colLength = 0;
033
034    /** starting position */
035    private int startPosition = 0;
036
037    /** ending position */
038    private int endPosition = 0;
039
040    public ColumnMetaData() {
041        super();
042    }
043
044    /**
045     * @param colName column name
046     */
047    public ColumnMetaData(final String colName) {
048        super();
049        this.colName = colName;
050    }
051
052    /**
053     * Returns the colLength.
054     *
055     * @return int
056     */
057    public int getColLength() {
058        return colLength;
059    }
060
061    /**
062     * Returns the colName.
063     *
064     * @return String
065     */
066    public String getColName() {
067        return colName;
068    }
069
070    /**
071     * Returns the endPosition.
072     *
073     * @return int
074     */
075    public int getEndPosition() {
076        return endPosition;
077    }
078
079    /**
080     * Returns the startPosition.
081     *
082     * @return int
083     */
084    public int getStartPosition() {
085        return startPosition;
086    }
087
088    /**
089     * Sets the colLength.
090     *
091     * @param colLength
092     *            The colLength to set
093     */
094    public void setColLength(final int colLength) {
095        this.colLength = colLength;
096    }
097
098    /**
099     * Sets the colName.
100     *
101     * @param colName
102     *            The colName to set
103     */
104    public void setColName(final String colName) {
105        this.colName = colName;
106    }
107
108    /**
109     * Sets the endPosition.
110     *
111     * @param endPosition
112     *            The endPosition to set
113     */
114    public void setEndPosition(final int endPosition) {
115        this.endPosition = endPosition;
116    }
117
118    /**
119     * Sets the startPosition.
120     *
121     * @param startPosition
122     *            The startPosition to set
123     */
124    public void setStartPosition(final int startPosition) {
125        this.startPosition = startPosition;
126    }
127
128    @Override
129    public String toString() {
130        final StringBuilder buf = new StringBuilder();
131        buf.append("Name:").append(colName).append(" Length:").append(colLength).append(" Start:").append(startPosition);
132        buf.append(" End:").append(endPosition).append(System.getProperty("line.separator"));
133        return buf.toString();
134    }
135}
