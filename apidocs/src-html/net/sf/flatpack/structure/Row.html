033package net.sf.flatpack.structure;
034
035import java.util.ArrayList;
036import java.util.List;
037
038/**
039 * This holds a collection of columns and their values along with the actual
040 * row number they appear in the flat file
041 *
042 * @author Paul Zepernick
043 */
044public class Row {
045    /** List to hold all columns that exist in the row */
046    private List<String> cols;
047
048    /** Row number in flat file */
049    private int rowNumber;
050
051    /** key to the MD for this row, null will indicate it is "detail" MD */
052    private String mdkey;
053
054    private boolean empty;
055
056    private String rawData;
057
058    /**
059     * Constructs a new Row
060     *
061     */
062    public Row() {
063        cols = new ArrayList<String>();
064    }
065
066    /**
067     * Adds a column to a row
068     *
069     * @param colValue -
070     *            String value to add to the row
071     */
072    public void addColumn(final String colValue) {
073        cols.add(colValue);
074    }
075
076    /**
077     * Appends the List of Strings to the existing columns in the row
078     *
079     * @param columns -
080     *            List of Strings to append to the row
081     */
082    public void addColumn(final List<String> columns) {
083        if (cols == null) {
084            cols = new ArrayList<String>();
085        }
086        cols.addAll(columns);
087    }
088
089    /**
090     * Returns the value of a column for a specified column name
091     *
092     * @param colPosition -
093     *            int position of the column in the array
094     * @return String value of column
095     */
096    public String getValue(final int colPosition) {
097        return cols.get(colPosition);
098    }
099
100    /**
101     * Set the value of a column for a specified column name
102     *
103     * @param columnIndex -
104     *            column number to change
105     * @param value -
106     *            String column value
107     */
108    public void setValue(final int columnIndex, final String value) {
109        cols.set(columnIndex, value);
110    }
111
112    /**
113     * Returns the rowNumber.
114     *
115     * @return int
116     */
117    public int getRowNumber() {
118        return rowNumber;
119    }
120
121    /**
122     * Sets the rowNumber.
123     *
124     * @param rowNumber
125     *            The rowNumber to set
126     */
127    public void setRowNumber(final int rowNumber) {
128        this.rowNumber = rowNumber;
129    }
130
131    /**
132     * Returns the cols for the row.
133     *
134     * @return Vector
135     */
136    public List<String> getCols() {
137        return cols;
138    }
139
140    /**
141     * Set the columns for the row.
142     *
143     * @param cols -
144     *            Vector of Strings
145     */
146    public void setCols(final List<String> cols) {
147        this.cols = cols;
148    }
149
150    /**
151     * @return Returns the mdkey.
152     */
153    public String getMdkey() {
154        return mdkey;
155    }
156
157    /**
158     * @param mdkey
159     *            The mdkey to set.
160     */
161    public void setMdkey(final String mdkey) {
162        this.mdkey = mdkey;
163    }
164
165    /**
166     * All columns in the row are empty
167     *
168     * @return the empty
169     */
170    public boolean isEmpty() {
171        return empty;
172    }
173
174    /**
175     * When true, all columns in the row are empty
176     *
177     * @param empty the empty to set
178     */
179    public void setEmpty(final boolean empty) {
180        this.empty = empty;
181    }
182
183    /**
184     * Raw data used to create the columns for the row
185     *
186     * @return the rawData
187     */
188    public String getRawData() {
189        return rawData;
190    }
191
192    /**
193     * Raw data used to create the columns for the row
194     *
195     * @param rawData the rawData to set
196     */
197    public void setRawData(final String rawData) {
198        this.rawData = rawData;
199    }
200}
