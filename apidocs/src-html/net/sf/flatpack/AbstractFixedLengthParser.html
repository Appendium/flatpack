033package net.sf.flatpack;
034
035import java.io.BufferedReader;
036import java.io.IOException;
037import java.io.Reader;
038import java.util.List;
039import java.util.Map;
040
041import net.sf.flatpack.structure.ColumnMetaData;
042import net.sf.flatpack.structure.Row;
043import net.sf.flatpack.util.FPConstants;
044import net.sf.flatpack.util.FixedWidthParserUtils;
045import net.sf.flatpack.util.ParserUtils;
046
047import org.slf4j.Logger;
048import org.slf4j.LoggerFactory;
049
050/**
051 * @author xhensevb
052 *
053 */
054public abstract class AbstractFixedLengthParser extends AbstractParser {
055    private static final Logger LOGGER = LoggerFactory.getLogger(AbstractFixedLengthParser.class);
056
057    protected AbstractFixedLengthParser(final Reader dataSourceReader, final String dataDefinition) {
058        super(dataSourceReader, dataDefinition);
059    }
060
061    protected AbstractFixedLengthParser(final Reader dataSourceReader) {
062        super(dataSourceReader);
063    }
064
065    @Override
066    protected DataSet doParse() {
067        try {
068            return doFixedLengthFile(getDataSourceReader());
069        } catch (final IOException e) {
070            LOGGER.error("error accessing/reading data", e);
071        }
072        return null;
073    }
074
075    /*
076     * This is the new version of doDelimitedFile using InputStrem instead of
077     * File. This is more flexible especially it is working with WebStart.
078     *
079     * puts together the dataset for fixed length file. This is used for PZ XML
080     * mappings, and SQL table mappings
081     */
082    private DataSet doFixedLengthFile(final Reader dataSource) throws IOException {
083        BufferedReader br = null;
084
085        final DefaultDataSet ds = new DefaultDataSet(getPzMetaData(), this);
086
087        try {
088            // gather the conversion properties
089            ds.setPZConvertProps(ParserUtils.loadConvertProperties());
090
091            final Map<String, Integer> recordLengths = ParserUtils.calculateRecordLengths(getPzMetaData());
092
093            // Read in the flat file
094            br = new BufferedReader(dataSource);
095            String line = null;
096            int lineCount = 0;
097            // map of record lengths corresponding to the ID's in the columnMD
098            // array loop through each line in the file
099            while ((line = br.readLine()) != null) {
100                lineCount++;
101                // empty line skip past it
102                if (line.trim().length() == 0) {
103                    continue;
104                }
105
106                final String mdkey = FixedWidthParserUtils.getCMDKey(getPzMetaData(), line);
107                final int recordLength = recordLengths.get(mdkey);
108
109                if (line.length() > recordLength) {
110                    // Incorrect record length on line log the error. Line will not
111                    // be included in the
112                    // dataset
113                    if (isIgnoreExtraColumns()) {
114                        // user has choosen to ignore the fact that we have too many bytes in the fixed
115                        // width file. Truncate the line to the correct length
116                        line = line.substring(0, recordLength);
117                        addError(ds, "TRUNCATED LINE TO CORRECT LENGTH", lineCount, 1);
118                    } else {
119                        addError(ds, "LINE TOO LONG. LINE IS " + line.length() + " LONG. SHOULD BE " + recordLength, lineCount, 2,
120                                isStoreRawDataToDataError() ? line : null);
121                        continue;
122                    }
123                } else if (line.length() < recordLength) {
124                    if (isHandlingShortLines()) {
125                        // We can pad this line out
126                        line += ParserUtils.padding(recordLength - line.length(), ' ');
127
128                        // log a warning
129                        addError(ds, "PADDED LINE TO CORRECT RECORD LENGTH", lineCount, 1);
130
131                    } else {
132                        addError(ds, "LINE TOO SHORT. LINE IS " + line.length() + " LONG. SHOULD BE " + recordLength, lineCount, 2,
133                                isStoreRawDataToDataError() ? line : null);
134                        continue;
135                    }
136                }
137
138                final Row row = new Row();
139                row.setMdkey(mdkey.equals(FPConstants.DETAIL_ID) ? null : mdkey); // try
140
141                final List<ColumnMetaData> cmds = ParserUtils.getColumnMetaData(mdkey, getPzMetaData());
142                row.addColumn(FixedWidthParserUtils.splitFixedText(cmds, line, isPreserveLeadingWhitespace(), isPreserveTrailingWhitespace()));
143                row.setRowNumber(lineCount);
144
145                if (isFlagEmptyRows()) {
146                    // user has elected to have the parser flag rows that are empty
147                    row.setEmpty(ParserUtils.isListElementsEmpty(row.getCols()));
148                }
149                if (isStoreRawDataToDataSet()) {
150                    // user told the parser to keep a copy of the raw data in the row
151                    // WARNING potential for high memory usage here
152                    row.setRawData(line);
153                }
154
155                // add the row to the array
156                ds.addRow(row);
157            }
158        } finally {
159            if (br != null) {
160                br.close();
161            }
162            closeReaders();
163        }
164        return ds;
165    }
166}
